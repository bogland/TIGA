74  public Skeleton (SkeletonData data) {

76  this.data = data;
77 
78  bones = new ExposedList<Bone>(data.bones.Count);
79  foreach (BoneData boneData in data.bones) {
80  Bone bone;
81  if (boneData.parent == null) {
82  bone = new Bone(boneData, this, null);
83  } else {
84  Bone parent = bones.Items[boneData.parent.index];
85  bone = new Bone(boneData, this, parent);
86  parent.children.Add(bone);
87  }
88  bones.Add(bone);
89  }
90 
91  slots = new ExposedList<Slot>(data.slots.Count);
92  drawOrder = new ExposedList<Slot>(data.slots.Count);
93  foreach (SlotData slotData in data.slots) {

95  Slot slot = new Slot(slotData, bone);
96  slots.Add(slot);
97  drawOrder.Add(slot);
98  }

100  ikConstraints = new ExposedList<IkConstraint>(data.ikConstraints.Count);

102  ikConstraints.Add(new IkConstraint(ikConstraintData, this));

118  public void UpdateCache () {
119  ExposedList<IUpdatable> updateCache = this.updateCache;
120  updateCache.Clear();
121  this.updateCacheReset.Clear();
122 
123  ExposedList<Bone> bones = this.bones;
124  for (int i = 0, n = bones.Count; i < n; i++)
125  bones.Items[i].sorted = false;
126 
127  ExposedList<IkConstraint> ikConstraints = this.ikConstraints;
128  var transformConstraints = this.transformConstraints;
129  var pathConstraints = this.pathConstraints;
130  int ikCount = IkConstraints.Count, transformCount = transformConstraints.Count, pathCount = pathConstraints.Count;
131  int constraintCount = ikCount + transformCount + pathCount;
132  //outer:
133  for (int i = 0; i < constraintCount; i++) {
134  for (int ii = 0; ii < ikCount; ii++) {
135  IkConstraint constraint = ikConstraints.Items[ii];
136  if (constraint.data.order == i) {
137  SortIkConstraint(constraint);
138  goto continue_outer; //continue outer;
139  }
140  }
141  for (int ii = 0; ii < transformCount; ii++) {
142  TransformConstraint constraint = transformConstraints.Items[ii];
143  if (constraint.data.order == i) {
144  SortTransformConstraint(constraint);
145  goto continue_outer; //continue outer;
146  }
147  }
148  for (int ii = 0; ii < pathCount; ii++) {
149  PathConstraint constraint = pathConstraints.Items[ii];
150  if (constraint.data.order == i) {
151  SortPathConstraint(constraint);
152  goto continue_outer; //continue outer;
153  }
154  }
155  continue_outer: {}
156  }
157 
158  for (int i = 0, n = bones.Count; i < n; i++)
159  SortBone(bones.Items[i]);
160  }

162  private void SortIkConstraint (IkConstraint constraint) {
163  Bone target = constraint.target;
164  SortBone(target);
165 
166  var constrained = constraint.bones;
167  Bone parent = constrained.Items[0];
168  SortBone(parent);
169 
170  if (constrained.Count > 1) {
171  Bone child = constrained.Items[constrained.Count - 1];
172  if (!updateCache.Contains(child))
173  updateCacheReset.Add(child);
174  }
175 
176  updateCache.Add(constraint);
177 
178  SortReset(parent.children);
179  constrained.Items[constrained.Count - 1].sorted = true;
180  }

182  private void SortPathConstraint (PathConstraint constraint) {
183  Slot slot = constraint.target;
184  int slotIndex = slot.data.index;
185  Bone slotBone = slot.bone;
186  if (skin != null) SortPathConstraintAttachment(skin, slotIndex, slotBone);
187  if (data.defaultSkin != null && data.defaultSkin != skin)
188  SortPathConstraintAttachment(data.defaultSkin, slotIndex, slotBone);
189  for (int ii = 0, nn = data.skins.Count; ii < nn; ii++)
190  SortPathConstraintAttachment(data.skins.Items[ii], slotIndex, slotBone);
191 
192  Attachment attachment = slot.attachment;
193  if (attachment is PathAttachment) SortPathConstraintAttachment(attachment, slotBone);
194 
195  var constrained = constraint.bones;
196  int boneCount = constrained.Count;
197  for (int i = 0; i < boneCount; i++)
198  SortBone(constrained.Items[i]);
199 
200  updateCache.Add(constraint);
201 
202  for (int i = 0; i < boneCount; i++)
203  SortReset(constrained.Items[i].children);
204  for (int i = 0; i < boneCount; i++)
205  constrained.Items[i].sorted = true;
206  }

208  private void SortTransformConstraint (TransformConstraint constraint) {
209  SortBone(constraint.target);
210 
211  var constrained = constraint.bones;
212  int boneCount = constrained.Count;
213  if (constraint.data.local) {
214  for (int i = 0; i < boneCount; i++) {
215  Bone child = constrained.Items[i];
216  SortBone(child.parent);
217  if (!updateCache.Contains(child)) updateCacheReset.Add(child);
218  }
219  } else {
220  for (int i = 0; i < boneCount; i++)
221  SortBone(constrained.Items[i]);
222  }
223 
224  updateCache.Add(constraint);
225 
226  for (int i = 0; i < boneCount; i++)
227  SortReset(constrained.Items[i].children);
228  for (int i = 0; i < boneCount; i++)
229  constrained.Items[i].sorted = true;
230  }

237  private void SortPathConstraintAttachment (Attachment attachment, Bone slotBone) {
238  if (!(attachment is PathAttachment)) return;
239  int[] pathBones = ((PathAttachment)attachment).bones;
240  if (pathBones == null)
241  SortBone(slotBone);
242  else {
243  var bones = this.bones;
244  for (int i = 0, n = pathBones.Length; i < n;) {
245  int nn = pathBones[i++];
246  nn += i;
247  while (i < nn)
248  SortBone(bones.Items[pathBones[i++]]);
249  }
250  }
251  }

257  bone.sorted = true;

497  public void GetBounds (out float x, out float y, out float width, out float height, ref float[] vertexBuffer) {
498  float[] temp = vertexBuffer;
499  temp = temp ?? new float[8];
500  var drawOrderItems = this.drawOrder.Items;
501  float minX = int.MaxValue, minY = int.MaxValue, maxX = int.MinValue, maxY = int.MinValue;
502  for (int i = 0, n = this.drawOrder.Count; i < n; i++) {
503  Slot slot = drawOrderItems[i];
504  int verticesLength = 0;
505  float[] vertices = null;
506  Attachment attachment = slot.attachment;
507  var regionAttachment = attachment as RegionAttachment;
508  if (regionAttachment != null) {
509  verticesLength = 8;
510  vertices = temp;
511  if (vertices.Length < 8) vertices = temp = new float[8];
512  regionAttachment.ComputeWorldVertices(slot.bone, temp, 0);
513  } else {
514  var meshAttachment = attachment as MeshAttachment;
515  if (meshAttachment != null) {
516  MeshAttachment mesh = meshAttachment;
517  verticesLength = mesh.WorldVerticesLength;
518  vertices = temp;
519  if (vertices.Length < verticesLength) vertices = temp = new float[verticesLength];
520  mesh.ComputeWorldVertices(slot, 0, verticesLength, temp, 0);
521  }
522  }
523 
524  if (vertices != null) {
525  for (int ii = 0; ii < verticesLength; ii += 2) {
526  float vx = vertices[ii], vy = vertices[ii + 1];
527  minX = Math.Min(minX, vx);
528  minY = Math.Min(minY, vy);
529  maxX = Math.Max(maxX, vx);
530  maxY = Math.Max(maxY, vy);
531  }
532  }
533  }
534  x = minX;
535  y = minY;
536  width = maxX - minX;
537  height = maxY - minY;
538  vertexBuffer = temp;
539  }
540  }
