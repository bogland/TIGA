142  public void UpdateWorldTransform (float x, float y, float rotation, float scaleX, float scaleY, float shearX, float shearY) {
143  ax = x;
144  ay = y;
145  arotation = rotation;
146  ascaleX = scaleX;
147  ascaleY = scaleY;
148  ashearX = shearX;
149  ashearY = shearY;
150  appliedValid = true;
151  Skeleton skeleton = this.skeleton;
152 
153  Bone parent = this.parent;
154  if (parent == null) { // Root bone.
155  float rotationY = rotation + 90 + shearY;
156  float la = MathUtils.CosDeg(rotation + shearX) * scaleX;
157  float lb = MathUtils.CosDeg(rotationY) * scaleY;
158  float lc = MathUtils.SinDeg(rotation + shearX) * scaleX;
159  float ld = MathUtils.SinDeg(rotationY) * scaleY;
160  if (skeleton.flipX) {
161  x = -x;
162  la = -la;
163  lb = -lb;
164  }
165  if (skeleton.flipY != yDown) {
166  y = -y;
167  lc = -lc;
168  ld = -ld;
169  }
170  a = la;
171  b = lb;
172  c = lc;
173  d = ld;
174  worldX = x + skeleton.x;
175  worldY = y + skeleton.y;
176  return;
177  }
178 
179  float pa = parent.a, pb = parent.b, pc = parent.c, pd = parent.d;
180  worldX = pa * x + pb * y + parent.worldX;
181  worldY = pc * x + pd * y + parent.worldY;
182 
183  switch (data.transformMode) {
184  case TransformMode.Normal: {
185  float rotationY = rotation + 90 + shearY;
186  float la = MathUtils.CosDeg(rotation + shearX) * scaleX;
187  float lb = MathUtils.CosDeg(rotationY) * scaleY;
188  float lc = MathUtils.SinDeg(rotation + shearX) * scaleX;
189  float ld = MathUtils.SinDeg(rotationY) * scaleY;
190  a = pa * la + pb * lc;
191  b = pa * lb + pb * ld;
192  c = pc * la + pd * lc;
193  d = pc * lb + pd * ld;
194  return;
195  }
196  case TransformMode.OnlyTranslation: {
197  float rotationY = rotation + 90 + shearY;
198  a = MathUtils.CosDeg(rotation + shearX) * scaleX;
199  b = MathUtils.CosDeg(rotationY) * scaleY;
200  c = MathUtils.SinDeg(rotation + shearX) * scaleX;
201  d = MathUtils.SinDeg(rotationY) * scaleY;
202  break;
203  }
204  case TransformMode.NoRotationOrReflection: {
205  float s = pa * pa + pc * pc, prx;
206  if (s > 0.0001f) {
207  s = Math.Abs(pa * pd - pb * pc) / s;
208  pb = pc * s;
209  pd = pa * s;
210  prx = MathUtils.Atan2(pc, pa) * MathUtils.RadDeg;
211  } else {
212  pa = 0;
213  pc = 0;
214  prx = 90 - MathUtils.Atan2(pd, pb) * MathUtils.RadDeg;
215  }
216  float rx = rotation + shearX - prx;
217  float ry = rotation + shearY - prx + 90;
218  float la = MathUtils.CosDeg(rx) * scaleX;
219  float lb = MathUtils.CosDeg(ry) * scaleY;
220  float lc = MathUtils.SinDeg(rx) * scaleX;
221  float ld = MathUtils.SinDeg(ry) * scaleY;
222  a = pa * la - pb * lc;
223  b = pa * lb - pb * ld;
224  c = pc * la + pd * lc;
225  d = pc * lb + pd * ld;
226  break;
227  }
228  case TransformMode.NoScale:
229  case TransformMode.NoScaleOrReflection: {
230  float cos = MathUtils.CosDeg(rotation), sin = MathUtils.SinDeg(rotation);
231  float za = pa * cos + pb * sin;
232  float zc = pc * cos + pd * sin;
233  float s = (float)Math.Sqrt(za * za + zc * zc);
234  if (s > 0.00001f) s = 1 / s;
235  za *= s;
236  zc *= s;
237  s = (float)Math.Sqrt(za * za + zc * zc);
238  float r = MathUtils.PI / 2 + MathUtils.Atan2(zc, za);
239  float zb = MathUtils.Cos(r) * s;
240  float zd = MathUtils.Sin(r) * s;
241  float la = MathUtils.CosDeg(shearX) * scaleX;
242  float lb = MathUtils.CosDeg(90 + shearY) * scaleY;
243  float lc = MathUtils.SinDeg(shearX) * scaleX;
244  float ld = MathUtils.SinDeg(90 + shearY) * scaleY;
245  if (data.transformMode != TransformMode.NoScaleOrReflection? pa * pd - pb* pc< 0 : skeleton.flipX != skeleton.flipY) {
246  zb = -zb;
247  zd = -zd;
248  }
249  a = za * la + zb * lc;
250  b = za * lb + zb * ld;
251  c = zc * la + zd * lc;
252  d = zc * lb + zd * ld;
253  return;
254  }
255  }
256 
257  if (skeleton.flipX) {
258  a = -a;
259  b = -b;
260  }
261  if (skeleton.flipY != Bone.yDown) {
262  c = -c;
263  d = -d;
264  }
265  }

284  internal void UpdateAppliedTransform () {
285  appliedValid = true;
286  Bone parent = this.parent;
287  if (parent == null) {
288  ax = worldX;
289  ay = worldY;
290  arotation = MathUtils.Atan2(c, a) * MathUtils.RadDeg;
291  ascaleX = (float)Math.Sqrt(a * a + c * c);
292  ascaleY = (float)Math.Sqrt(b * b + d * d);
293  ashearX = 0;
294  ashearY = MathUtils.Atan2(a * b + c * d, a * d - b * c) * MathUtils.RadDeg;
295  return;
296  }
297  float pa = parent.a, pb = parent.b, pc = parent.c, pd = parent.d;
298  float pid = 1 / (pa * pd - pb * pc);
299  float dx = worldX - parent.worldX, dy = worldY - parent.worldY;
300  ax = (dx * pd * pid - dy * pb * pid);
301  ay = (dy * pa * pid - dx * pc * pid);
302  float ia = pid * pd;
303  float id = pid * pa;
304  float ib = pid * pb;
305  float ic = pid * pc;
306  float ra = ia * a - ib * c;
307  float rb = ia * b - ib * d;
308  float rc = id * c - ic * a;
309  float rd = id * d - ic * b;
310  ashearX = 0;
311  ascaleX = (float)Math.Sqrt(ra * ra + rc * rc);
312  if (ascaleX > 0.0001f) {
313  float det = ra * rd - rb * rc;
314  ascaleY = det / ascaleX;
315  ashearY = MathUtils.Atan2(ra * rb + rc * rd, det) * MathUtils.RadDeg;
316  arotation = MathUtils.Atan2(rc, ra) * MathUtils.RadDeg;
317  } else {
318  ascaleX = 0;
319  ascaleY = (float)Math.Sqrt(rb * rb + rd * rd);
320  ashearY = 0;
321  arotation = 90 - MathUtils.Atan2(rd, rb) * MathUtils.RadDeg;
322  }
323  }

325  public void WorldToLocal (float worldX, float worldY, out float localX, out float localY) {
326  float a = this.a, b = this.b, c = this.c, d = this.d;
327  float invDet = 1 / (a * d - b * c);
328  float x = worldX - this.worldX, y = worldY - this.worldY;
329  localX = (x * d * invDet - y * b * invDet);
330  localY = (y * a * invDet - x * c * invDet);
331  }
