68  internal static int BinarySearch (float[] values, float target, int step) {
69  int low = 0;
70  int high = values.Length / step - 2;
71  if (high == 0) return step;
72  int current = (int)((uint)high >> 1);
73  while (true) {
74  if (values[(current + 1) * step] <= target)
75  low = current + 1;
76  else
77  high = current;
78  if (low == high) return (low + 1) * step;
79  current = (int)((uint)(low + high) >> 1);
80  }
81  }
82 
84  internal static int BinarySearch (float[] values, float target) {
85  int low = 0;
86  int high = values.Length - 2;
87  if (high == 0) return 1;
88  int current = (int)((uint)high >> 1);
89  while (true) {
90  if (values[(current + 1)] <= target)
91  low = current + 1;
92  else
93  high = current;
94  if (low == high) return (low + 1);
95  current = (int)((uint)(low + high) >> 1);
96  }
97  }

99  internal static int LinearSearch (float[] values, float target, int step) {
100  for (int i = 0, last = values.Length - step; i <= last; i += step)
101  if (values[i] > target) return i;
102  return -1;
103  }

180  public void SetCurve (int frameIndex, float cx1, float cy1, float cx2, float cy2) {
181  float tmpx = (-cx1 * 2 + cx2) * 0.03f, tmpy = (-cy1 * 2 + cy2) * 0.03f;
182  float dddfx = ((cx1 - cx2) * 3 + 1) * 0.006f, dddfy = ((cy1 - cy2) * 3 + 1) * 0.006f;
183  float ddfx = tmpx * 2 + dddfx, ddfy = tmpy * 2 + dddfy;
184  float dfx = cx1 * 0.3f + tmpx + dddfx * 0.16666667f, dfy = cy1 * 0.3f + tmpy + dddfy * 0.16666667f;
185 
186  int i = frameIndex * BEZIER_SIZE;
187  float[] curves = this.curves;
188  curves[i++] = BEZIER;
189 
190  float x = dfx, y = dfy;
191  for (int n = i + BEZIER_SIZE - 1; i < n; i += 2) {
192  curves[i] = x;
193  curves[i + 1] = y;
194  dfx += ddfx;
195  dfy += ddfy;
196  ddfx += dddfx;
197  ddfy += dddfy;
198  x += dfx;
199  y += dfy;
200  }
201  }

203  public float GetCurvePercent (int frameIndex, float percent) {
204  percent = MathUtils.Clamp (percent, 0, 1);
205  float[] curves = this.curves;
206  int i = frameIndex * BEZIER_SIZE;
207  float type = curves[i];
208  if (type == LINEAR) return percent;
209  if (type == STEPPED) return 0;
210  i++;
211  float x = 0;
212  for (int start = i, n = i + BEZIER_SIZE - 1; i < n; i += 2) {
213  x = curves[i];
214  if (x >= percent) {
215  float prevX, prevY;
216  if (i == start) {
217  prevX = 0;
218  prevY = 0;
219  } else {
220  prevX = curves[i - 2];
221  prevY = curves[i - 1];
222  }
223  return prevY + (curves[i + 1] - prevY) * (percent - prevX) / (x - prevX);
224  }
225  }
226  float y = curves[i - 1];
227  return y + (1 - y) * (percent - x) / (1 - x); // Last point is 1,1.
228  }
229  public float GetCurveType (int frameIndex) {
230  return curves[frameIndex * BEZIER_SIZE];
231  }

261  override public void Apply (Skeleton skeleton, float lastTime, float time, ExposedList<Event> firedEvents, float alpha, MixPose pose, MixDirection direction) {
262  Bone bone = skeleton.bones.Items[boneIndex];
263 
264  float[] frames = this.frames;
265  if (time < frames[0]) {
266  switch (pose) {
267  case MixPose.Setup:
268  bone.rotation = bone.data.rotation;
269  return;
270  case MixPose.Current:
271  float rr = bone.data.rotation - bone.rotation;
272  rr -= (16384 - (int)(16384.499999999996 - rr / 360)) * 360;
273  bone.rotation += rr * alpha;
274  return;
275  }
276  return;
277  }
278 
279  if (time >= frames[frames.Length - ENTRIES]) { // Time is after last frame.
280  if (pose == MixPose.Setup) {
281  bone.rotation = bone.data.rotation + frames[frames.Length + PREV_ROTATION] * alpha;
282  } else {
283  float rr = bone.data.rotation + frames[frames.Length + PREV_ROTATION] - bone.rotation;
284  rr -= (16384 - (int)(16384.499999999996 - rr / 360)) * 360; // Wrap within -180 and 180.
285  bone.rotation += rr * alpha;
286  }
287  return;
288  }
289 
290  // Interpolate between the previous frame and the current frame.
291  int frame = Animation.BinarySearch(frames, time, ENTRIES);
292  float prevRotation = frames[frame + PREV_ROTATION];
293  float frameTime = frames[frame];
294  float percent = GetCurvePercent((frame >> 1) - 1, 1 - (time - frameTime) / (frames[frame + PREV_TIME] - frameTime));
295 
296  float r = frames[frame + ROTATION] - prevRotation;
297  r -= (16384 - (int)(16384.499999999996 - r / 360)) * 360;
298  r = prevRotation + r * percent;
299  if (pose == MixPose.Setup) {
300  r -= (16384 - (int)(16384.499999999996 - r / 360)) * 360;
301  bone.rotation = bone.data.rotation + r * alpha;
302  } else {
303  r = bone.data.rotation + r - bone.rotation;
304  r -= (16384 - (int)(16384.499999999996 - r / 360)) * 360;
305  bone.rotation += r * alpha;
306  }
307  }

360  } else {
361  // Interpolate between the previous frame and the current frame.
362  int frame = Animation.BinarySearch(frames, time, ENTRIES);
363  x = frames[frame + PREV_X];
364  y = frames[frame + PREV_Y];

421 
422  x = (x + (frames[frame + X] - x) * percent) * bone.data.scaleX;
423  y = (y + (frames[frame + Y] - y) * percent) * bone.data.scaleY;
424  }
425  if (alpha == 1) {
426  bone.scaleX = x;
427  bone.scaleY = y;
428  } else {
429  float bx, by;
430  if (pose == MixPose.Setup) {
431  bx = bone.data.scaleX;
432  by = bone.data.scaleY;
433  } else {
434  bx = bone.scaleX;
435  by = bone.scaleY;
436  }
437  // Mixing out uses sign of setup or current pose, else use sign of key.
438  if (direction == MixDirection.Out) {
439  x = (x >= 0 ? x : -x) * (bx >= 0 ? 1 : -1);
440  y = (y >= 0 ? y : -y) * (by >= 0 ? 1 : -1);
441  } else {
442  bx = (bx >= 0 ? bx : -bx) * (x >= 0 ? 1 : -1);
443  by = (by >= 0 ? by : -by) * (y >= 0 ? 1 : -1);
444  }
445  bone.scaleX = bx + (x - bx) * alpha;
446  bone.scaleY = by + (y - by) * alpha;
447  }
448  }
