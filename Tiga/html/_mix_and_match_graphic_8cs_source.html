65  void OnValidate () {
66  if (sourceMaterial == null) {
67  var skeletonGraphic = GetComponent<SkeletonGraphic>();
68  if (skeletonGraphic != null)
69  sourceMaterial = skeletonGraphic.SkeletonDataAsset.atlasAssets[0].materials[0];

78  [ContextMenu("Apply")]
79  void Apply () {
80  var skeletonGraphic = GetComponent<SkeletonGraphic>();
81  var skeleton = skeletonGraphic.Skeleton;
82 
83  // STEP 0: PREPARE SKINS
84  // Let's prepare a new skin to be our custom skin with equips/customizations. We get a clone so our original skins are unaffected.
85  customSkin = customSkin ?? new Skin("custom skin"); // This requires that all customizations are done with skin placeholders defined in Spine.
86  //customSkin = customSkin ?? skeleton.UnshareSkin(true, false, skeletonAnimation.AnimationState); // use this if you are not customizing on the default skin and don't plan to remove
87  // Next let's get the skin that contains our source attachments. These are the attachments that
88  var baseSkin = skeleton.Data.FindSkin(baseSkinName);
89 
90  // STEP 1: "EQUIP" ITEMS USING SPRITES
91  // STEP 1.1 Find the original attachment.
92  // Step 1.2 Get a clone of the original attachment.
93  // Step 1.3 Apply the Sprite image to it.
94  // Step 1.4 Add the remapped clone to the new custom skin.
95 
96  // Let's do this for the visor.
97  int visorSlotIndex = skeleton.FindSlotIndex(visorSlot); // You can access GetAttachment and SetAttachment via string, but caching the slotIndex is faster.
98  Attachment baseAttachment = baseSkin.GetAttachment(visorSlotIndex, visorKey); // STEP 1.1
99  Attachment newAttachment = baseAttachment.GetRemappedClone(visorSprite, sourceMaterial); // STEP 1.2 - 1.3
100  customSkin.SetAttachment(visorSlotIndex, visorKey, newAttachment); // STEP 1.4
101 
102  // And now for the gun.
103  int gunSlotIndex = skeleton.FindSlotIndex(gunSlot);
104  Attachment baseGun = baseSkin.GetAttachment(gunSlotIndex, gunKey); // STEP 1.1
105  Attachment newGun = baseGun.GetRemappedClone(gunSprite, sourceMaterial); // STEP 1.2 - 1.3
106  if (newGun != null) customSkin.SetAttachment(gunSlotIndex, gunKey, newGun); // STEP 1.4
107 
108  // customSkin.RemoveAttachment(gunSlotIndex, gunKey); // To remove an item.
109  // customSkin.Clear()
110  // Use skin.Clear() To remove all customizations.
111  // Customizations will fall back to the value in the default skin if it was defined there.
112  // To prevent fallback from happening, make sure the key is not defined in the default skin.
113 
114  // STEP 3: APPLY AND CLEAN UP.
115  // Recommended: REPACK THE CUSTOM SKIN TO MINIMIZE DRAW CALLS
116  // Repacking requires that you set all source textures/sprites/atlases to be Read/Write enabled in the inspector.
117  // Combine all the attachment sources into one skin. Usually this means the default skin and the custom skin.
118  // call Skin.GetRepackedSkin to get a cloned skin with cloned attachments that all use one texture.
119  if (repack) {
120  var repackedSkin = new Skin("repacked skin");
121  repackedSkin.Append(skeleton.Data.DefaultSkin);
122  repackedSkin.Append(customSkin);
123  repackedSkin = repackedSkin.GetRepackedSkin("repacked skin", sourceMaterial, out runtimeMaterial, out runtimeAtlas);
124  skeleton.SetSkin(repackedSkin);
125  } else {
126  skeleton.SetSkin(customSkin);
127  }
128 
129  //skeleton.SetSlotsToSetupPose();
130  skeleton.SetToSetupPose();
131  skeletonGraphic.Update(0);
132  skeletonGraphic.OverrideTexture = runtimeAtlas;
133 
134  Resources.UnloadUnusedAssets();
135  }
136  }
