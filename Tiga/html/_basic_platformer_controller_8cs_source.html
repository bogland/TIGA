36  [RequireComponent(typeof(CharacterController))]
37  public class BasicPlatformerController : MonoBehaviour {

39  [Header("Controls")]
40  public string XAxis = "Horizontal";
41  public string YAxis = "Vertical";
42  public string JumpButton = "Jump";
43 
44  [Header("Moving")]
45  public float walkSpeed = 1.5f;
46  public float runSpeed = 7f;
47  public float gravityScale = 6.6f;
48 
49  [Header("Jumping")]
50  public float jumpSpeed = 25;
51  public float minimumJumpDuration = 0.5f;
52  public float jumpInterruptFactor = 0.5f;
53  public float forceCrouchVelocity = 25;
54  public float forceCrouchDuration = 0.5f;
55 
56  [Header("Visuals")]
57  public SkeletonAnimation skeletonAnimation;
58 
59  [Header("Animation")]
60  public TransitionDictionaryExample transitions;
61  public AnimationReferenceAsset walk;
62  public AnimationReferenceAsset run;
63  public AnimationReferenceAsset idle;
64  public AnimationReferenceAsset jump;
65  public AnimationReferenceAsset fall;
66  public AnimationReferenceAsset crouch;
67  public AnimationReferenceAsset runFromFall;
68 
69  [Header("Effects")]
70  public AudioSource jumpAudioSource;
71  public AudioSource hardfallAudioSource;
72  public ParticleSystem landParticles;
73  public HandleEventWithAudioExample footstepHandler;
74 
75  CharacterController controller;
76  Vector2 input = default(Vector2);
77  Vector3 velocity = default(Vector3);
78  float minimumJumpEndTime = 0;
79  float forceCrouchEndTime;
80  bool wasGrounded = false;
81 
82  AnimationReferenceAsset targetAnimation;
83  AnimationReferenceAsset previousTargetAnimation;

85  void Awake () {
86  controller = GetComponent<CharacterController>();
87  }

89  void Update () {
90  float dt = Time.deltaTime;
91  bool isGrounded = controller.isGrounded;
92  bool landed = !wasGrounded && isGrounded;
93 
94  // Dummy input.
95  input.x = Input.GetAxis(XAxis);
96  input.y = Input.GetAxis(YAxis);
97  bool inputJumpStop = Input.GetButtonUp(JumpButton);
98  bool inputJumpStart = Input.GetButtonDown(JumpButton);
99  bool doCrouch = (isGrounded && input.y < -0.5f) || (forceCrouchEndTime > Time.time);
100  bool doJumpInterrupt = false;
101  bool doJump = false;
102  bool hardLand = false;
103 
104  if (landed) {
105  if (-velocity.y > forceCrouchVelocity) {
106  hardLand = true;
107  doCrouch = true;
108  forceCrouchEndTime = Time.time + forceCrouchDuration;
109  }
110  }
111 
112  if (!doCrouch) {
113  if (isGrounded) {
114  if (inputJumpStart) {
115  doJump = true;
116  }
117  } else {
118  doJumpInterrupt = inputJumpStop && Time.time < minimumJumpEndTime;
119  }
120  }
121 
122  // Dummy physics and controller using UnityEngine.CharacterController.
123  Vector3 gravityDeltaVelocity = Physics.gravity * gravityScale * dt;
124 
125 
126  if (doJump) {
127  velocity.y = jumpSpeed;
128  minimumJumpEndTime = Time.time + minimumJumpDuration;
129  } else if (doJumpInterrupt) {
130  if (velocity.y > 0)
131  velocity.y *= jumpInterruptFactor;
132  }
133 
134  velocity.x = 0;
135  if (!doCrouch) {
136  if (input.x != 0) {
137  velocity.x = Mathf.Abs(input.x) > 0.6f ? runSpeed : walkSpeed;
138  velocity.x *= Mathf.Sign(input.x);
139  }
140  }
141 
142 
143  if (!isGrounded) {
144  if (wasGrounded) {
145  if (velocity.y < 0)
146  velocity.y = 0;
147  } else {
148  velocity += gravityDeltaVelocity;
149  }
150  }
151  controller.Move(velocity * dt);
152 
153  // Animation
154  // Determine target animation.
155  if (isGrounded) {
156  if (doCrouch) {
157  targetAnimation = crouch;
158  } else {
159  if (input.x == 0)
160  targetAnimation = idle;
161  else
162  targetAnimation = Mathf.Abs(input.x) > 0.6f ? run : walk;
163  }
164  } else {
165  targetAnimation = velocity.y > 0 ? jump : fall;
166  }
167 
168  // Handle change in target animation.
169  if (previousTargetAnimation != targetAnimation) {
170  Animation transition = null;
171  if (transitions != null && previousTargetAnimation != null) {
172  transition = transitions.GetTransition(previousTargetAnimation, targetAnimation);
173  }
174 
175  if (transition != null) {
176  skeletonAnimation.AnimationState.SetAnimation(0, transition, false).MixDuration = 0.05f;
177  skeletonAnimation.AnimationState.AddAnimation(0, targetAnimation, true, 0f);
178  } else {
179  skeletonAnimation.AnimationState.SetAnimation(0, targetAnimation, true);
180  }
181  }
182  previousTargetAnimation = targetAnimation;
183 
184  // Face intended direction.
185  if (input.x != 0)
186  skeletonAnimation.Skeleton.FlipX = input.x < 0;
187 
188 
189  // Effects
190  if (doJump) {
191  jumpAudioSource.Stop();
192  jumpAudioSource.Play();
193  }
194 
195  if (landed) {
196  if (hardLand) {
197  hardfallAudioSource.Play();
198  } else {
199  footstepHandler.Play();
200  }
201 
202  landParticles.Emit((int)(velocity.y / -9f) + 2);
203  }
204 
205  wasGrounded = isGrounded;
206  }
207 
208  }
209 }
