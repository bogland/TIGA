251  public static void GenerateSkeletonRendererInstruction (SkeletonRendererInstruction instructionOutput, Skeleton skeleton, Dictionary<Slot, Material> customSlotMaterials, List<Slot> separatorSlots, bool generateMeshOverride, bool immutableTriangles = false) {
252 // if (skeleton == null) throw new ArgumentNullException("skeleton");
253 // if (instructionOutput == null) throw new ArgumentNullException("instructionOutput");
254 
255  ExposedList<Slot> drawOrder = skeleton.drawOrder;
256  int drawOrderCount = drawOrder.Count;
257 
258  // Clear last state of attachments and submeshes
259  instructionOutput.Clear(); // submeshInstructions.Clear(); attachments.Clear();
260  var workingSubmeshInstructions = instructionOutput.submeshInstructions;
261  #if SPINE_TRIANGLECHECK
262  instructionOutput.attachments.Resize(drawOrderCount);
263  var workingAttachmentsItems = instructionOutput.attachments.Items;
264  int totalRawVertexCount = 0;
265  bool skeletonHasClipping = false;
266  #endif
267 
268  var current = new SubmeshInstruction {
269  skeleton = skeleton,
270  preActiveClippingSlotSource = -1
271  };
272 
273  #if !SPINE_TK2D
274  bool isCustomSlotMaterialsPopulated = customSlotMaterials != null && customSlotMaterials.Count > 0;
275  #endif
276 
277  int separatorCount = separatorSlots == null ? 0 : separatorSlots.Count;
278  bool hasSeparators = separatorCount > 0;
279 
280  int clippingAttachmentSource = -1;
281  int lastPreActiveClipping = -1; // The index of the last slot that had an active ClippingAttachment.
282  SlotData clippingEndSlot = null;
283  int submeshIndex = 0;
284  var drawOrderItems = drawOrder.Items;
285  for (int i = 0; i < drawOrderCount; i++) {
286  Slot slot = drawOrderItems[i];
287  Attachment attachment = slot.attachment;
288  #if SPINE_TRIANGLECHECK
289  workingAttachmentsItems[i] = attachment;
290  int attachmentVertexCount = 0, attachmentTriangleCount = 0;
291  #endif
292 
293  object rendererObject = null; // An AtlasRegion in plain Spine-Unity. Spine-TK2D hooks into TK2D's system. eventual source of Material object.
294  bool noRender = false; // Using this allows empty slots as separators, and keeps separated parts more stable despite slots being reordered
295 
296  var regionAttachment = attachment as RegionAttachment;
297  if (regionAttachment != null) {
298  rendererObject = regionAttachment.RendererObject;
299  #if SPINE_TRIANGLECHECK
300  attachmentVertexCount = 4;
301  attachmentTriangleCount = 6;
302  #endif
303  } else {
304  var meshAttachment = attachment as MeshAttachment;
305  if (meshAttachment != null) {
306  rendererObject = meshAttachment.RendererObject;
307  #if SPINE_TRIANGLECHECK
308  attachmentVertexCount = meshAttachment.worldVerticesLength >> 1;
309  attachmentTriangleCount = meshAttachment.triangles.Length;
310  #endif
311  } else {
312  #if SPINE_TRIANGLECHECK
313  var clippingAttachment = attachment as ClippingAttachment;
314  if (clippingAttachment != null) {
315  clippingEndSlot = clippingAttachment.endSlot;
316  clippingAttachmentSource = i;
317  current.hasClipping = true;
318  skeletonHasClipping = true;
319  }
320  #endif
321  noRender = true;
322  }
323  }
324 
325  if (clippingEndSlot != null && slot.data == clippingEndSlot && i != clippingAttachmentSource) {
326  clippingEndSlot = null;
327  clippingAttachmentSource = -1;
328  }
329 
330  // Create a new SubmeshInstruction when material changes. (or when forced to separate by a submeshSeparator)
331  // Slot with a separator/new material will become the starting slot of the next new instruction.
332  if (hasSeparators) { //current.forceSeparate = hasSeparators && separatorSlots.Contains(slot);
333  current.forceSeparate = false;
334  for (int s = 0; s < separatorCount; s++) {
335  if (Slot.ReferenceEquals(slot, separatorSlots[s])) {
336  current.forceSeparate = true;
337  break;
338  }
339  }
340  }
341 
342  if (noRender) {
343  if (current.forceSeparate && generateMeshOverride) { // && current.rawVertexCount > 0) {
344  { // Add
345  current.endSlot = i;
346  current.preActiveClippingSlotSource = lastPreActiveClipping;
347 
348  workingSubmeshInstructions.Resize(submeshIndex + 1);
349  workingSubmeshInstructions.Items[submeshIndex] = current;
350 
351  submeshIndex++;
352  }
353 
354  current.startSlot = i;
355  lastPreActiveClipping = clippingAttachmentSource;
356  #if SPINE_TRIANGLECHECK
357  current.rawTriangleCount = 0;
358  current.rawVertexCount = 0;
359  current.rawFirstVertexIndex = totalRawVertexCount;
360  current.hasClipping = clippingAttachmentSource >= 0;
361  #endif
362  }
363  } else {
364  #if !SPINE_TK2D
365  Material material;
366  if (isCustomSlotMaterialsPopulated) {
367  if (!customSlotMaterials.TryGetValue(slot, out material))
368  material = (Material)((AtlasRegion)rendererObject).page.rendererObject;
369  } else {
370  material = (Material)((AtlasRegion)rendererObject).page.rendererObject;
371  }
372  #else
373  Material material = (rendererObject is Material) ? (Material)rendererObject : (Material)((AtlasRegion)rendererObject).page.rendererObject;
374  #endif
375 
376  if (current.forceSeparate || (current.rawVertexCount > 0 && !System.Object.ReferenceEquals(current.material, material))) { // Material changed. Add the previous submesh.
377  { // Add
378  current.endSlot = i;
379  current.preActiveClippingSlotSource = lastPreActiveClipping;
380 
381  workingSubmeshInstructions.Resize(submeshIndex + 1);
382  workingSubmeshInstructions.Items[submeshIndex] = current;
383  submeshIndex++;
384  }
385  current.startSlot = i;
386  lastPreActiveClipping = clippingAttachmentSource;
387  #if SPINE_TRIANGLECHECK
388  current.rawTriangleCount = 0;
389  current.rawVertexCount = 0;
390  current.rawFirstVertexIndex = totalRawVertexCount;
391  current.hasClipping = clippingAttachmentSource >= 0;
392  #endif
393  }
394 
395  // Update state for the next Attachment.
396  current.material = material;
397  #if SPINE_TRIANGLECHECK
398  current.rawTriangleCount += attachmentTriangleCount;
399  current.rawVertexCount += attachmentVertexCount;
400  current.rawFirstVertexIndex = totalRawVertexCount;
401  totalRawVertexCount += attachmentVertexCount;
402  #endif
403  }
404  }
405 
406  if (current.rawVertexCount > 0) {
407  { // Add last or only submesh.
408  current.endSlot = drawOrderCount;
409  current.preActiveClippingSlotSource = lastPreActiveClipping;
410  current.forceSeparate = false;
411 
412  workingSubmeshInstructions.Resize(submeshIndex + 1);
413  workingSubmeshInstructions.Items[submeshIndex] = current;
414  //submeshIndex++;
415  }
416  }
417 
418  #if SPINE_TRIANGLECHECK
419  instructionOutput.hasActiveClipping = skeletonHasClipping;
420  instructionOutput.rawVertexCount = totalRawVertexCount;
421  #endif
422  instructionOutput.immutableTriangles = immutableTriangles;
423  }

459  public void AddSubmesh (SubmeshInstruction instruction, bool updateTriangles = true) {
460  var settings = this.settings;
461 
462  int newCount = submeshIndex + 1;
463  if (submeshes.Items.Length < newCount)
464  submeshes.Resize(newCount);
465  submeshes.Count = newCount;
466  var submesh = submeshes.Items[submeshIndex];
467  if (submesh == null)
468  submeshes.Items[submeshIndex] = submesh = new ExposedList<int>();
469 
470  submesh.Clear(false);
471 
472  var skeleton = instruction.skeleton;
473  var drawOrderItems = skeleton.drawOrder.Items;
474 
475  Color32 color = default(Color32);
476  float skeletonA = skeleton.a * 255, skeletonR = skeleton.r, skeletonG = skeleton.g, skeletonB = skeleton.b;
477  Vector2 meshBoundsMin = this.meshBoundsMin, meshBoundsMax = this.meshBoundsMax;
478 
479  // Settings
480  float zSpacing = settings.zSpacing;
481  bool pmaVertexColors = settings.pmaVertexColors;
482  bool tintBlack = settings.tintBlack;
483  #if SPINE_TRIANGLECHECK
484  bool useClipping = settings.useClipping && instruction.hasClipping;
485  #else
486  bool useClipping = settings.useClipping;
487  #endif
488 
489  if (useClipping) {
490  if (instruction.preActiveClippingSlotSource >= 0) {
491  var slot = drawOrderItems[instruction.preActiveClippingSlotSource];
492  clipper.ClipStart(slot, slot.attachment as ClippingAttachment);
493  }
494  }
495 
496  for (int slotIndex = instruction.startSlot; slotIndex < instruction.endSlot; slotIndex++) {
497  var slot = drawOrderItems[slotIndex];
498  var attachment = slot.attachment;
499  float z = zSpacing * slotIndex;
500 
501  var workingVerts = this.tempVerts;
502  float[] uvs;
503  int[] attachmentTriangleIndices;
504  int attachmentVertexCount;
505  int attachmentIndexCount;
506 
507  Color c = default(Color);
508 
509  // Identify and prepare values.
510  var region = attachment as RegionAttachment;
511  if (region != null) {
512  region.ComputeWorldVertices(slot.bone, workingVerts, 0);
513  uvs = region.uvs;
514  attachmentTriangleIndices = regionTriangles;
515  c.r = region.r; c.g = region.g; c.b = region.b; c.a = region.a;
516  attachmentVertexCount = 4;
517  attachmentIndexCount = 6;
518  } else {
519  var mesh = attachment as MeshAttachment;
520  if (mesh != null) {
521  int meshVerticesLength = mesh.worldVerticesLength;
522  if (workingVerts.Length < meshVerticesLength) {
523  workingVerts = new float[meshVerticesLength];
524  this.tempVerts = workingVerts;
525  }
526  mesh.ComputeWorldVertices(slot, 0, meshVerticesLength, workingVerts, 0); //meshAttachment.ComputeWorldVertices(slot, tempVerts);
527  uvs = mesh.uvs;
528  attachmentTriangleIndices = mesh.triangles;
529  c.r = mesh.r; c.g = mesh.g; c.b = mesh.b; c.a = mesh.a;
530  attachmentVertexCount = meshVerticesLength >> 1; // meshVertexCount / 2;
531  attachmentIndexCount = mesh.triangles.Length;
532  } else {
533  if (useClipping) {
534  var clippingAttachment = attachment as ClippingAttachment;
535  if (clippingAttachment != null) {
536  clipper.ClipStart(slot, clippingAttachment);
537  continue;
538  }
539  }
540 
541  // If not any renderable attachment.
542  clipper.ClipEnd(slot);
543  continue;
544  }
545  }
546 
547  if (pmaVertexColors) {
548  color.a = (byte)(skeletonA * slot.a * c.a);
549  color.r = (byte)(skeletonR * slot.r * c.r * color.a);
550  color.g = (byte)(skeletonG * slot.g * c.g * color.a);
551  color.b = (byte)(skeletonB * slot.b * c.b * color.a);
552  if (slot.data.blendMode == BlendMode.Additive) color.a = 0;
553  } else {
554  color.a = (byte)(skeletonA * slot.a * c.a);
555  color.r = (byte)(skeletonR * slot.r * c.r * 255);
556  color.g = (byte)(skeletonG * slot.g * c.g * 255);
557  color.b = (byte)(skeletonB * slot.b * c.b * 255);
558  }
559 
560  if (useClipping && clipper.IsClipping) {
561  clipper.ClipTriangles(workingVerts, attachmentVertexCount << 1, attachmentTriangleIndices, attachmentIndexCount, uvs);
562  workingVerts = clipper.clippedVertices.Items;
563  attachmentVertexCount = clipper.clippedVertices.Count >> 1;
564  attachmentTriangleIndices = clipper.clippedTriangles.Items;
565  attachmentIndexCount = clipper.clippedTriangles.Count;
566  uvs = clipper.clippedUVs.Items;
567  }
568 
569  // Actually add slot/attachment data into buffers.
570  if (attachmentVertexCount != 0 && attachmentIndexCount != 0) {
571  if (tintBlack)
572  AddAttachmentTintBlack(slot.r2, slot.g2, slot.b2, attachmentVertexCount);
573 
574  //AddAttachment(workingVerts, uvs, color, attachmentTriangleIndices, attachmentVertexCount, attachmentIndexCount, ref meshBoundsMin, ref meshBoundsMax, z);
575  int ovc = vertexBuffer.Count;
576  // Add data to vertex buffers
577  {
578  int newVertexCount = ovc + attachmentVertexCount;
579  if (newVertexCount > vertexBuffer.Items.Length) { // Manual ExposedList.Resize()
580  Array.Resize(ref vertexBuffer.Items, newVertexCount);
581  Array.Resize(ref uvBuffer.Items, newVertexCount);
582  Array.Resize(ref colorBuffer.Items, newVertexCount);
583  }
584  vertexBuffer.Count = uvBuffer.Count = colorBuffer.Count = newVertexCount;
585  }
586 
587  var vbi = vertexBuffer.Items;
588  var ubi = uvBuffer.Items;
589  var cbi = colorBuffer.Items;
590  if (ovc == 0) {
591  for (int i = 0; i < attachmentVertexCount; i++) {
592  int vi = ovc + i;
593  int i2 = i << 1; // i * 2
594  float x = workingVerts[i2];
595  float y = workingVerts[i2 + 1];
596 
597  vbi[vi].x = x;
598  vbi[vi].y = y;
599  vbi[vi].z = z;
600  ubi[vi].x = uvs[i2];
601  ubi[vi].y = uvs[i2 + 1];
602  cbi[vi] = color;
603 
604  // Calculate bounds.
605  if (x < meshBoundsMin.x) meshBoundsMin.x = x;
606  if (x > meshBoundsMax.x) meshBoundsMax.x = x;
607  if (y < meshBoundsMin.y) meshBoundsMin.y = y;
608  if (y > meshBoundsMax.y) meshBoundsMax.y = y;
609  }
610  } else {
611  for (int i = 0; i < attachmentVertexCount; i++) {
612  int vi = ovc + i;
613  int i2 = i << 1; // i * 2
614  float x = workingVerts[i2];
615  float y = workingVerts[i2 + 1];
616 
617  vbi[vi].x = x;
618  vbi[vi].y = y;
619  vbi[vi].z = z;
620  ubi[vi].x = uvs[i2];
621  ubi[vi].y = uvs[i2 + 1];
622  cbi[vi] = color;
623 
624  // Calculate bounds.
625  if (x < meshBoundsMin.x) meshBoundsMin.x = x;
626  else if (x > meshBoundsMax.x) meshBoundsMax.x = x;
627  if (y < meshBoundsMin.y) meshBoundsMin.y = y;
628  else if (y > meshBoundsMax.y) meshBoundsMax.y = y;
629  }
630  }
631 
632 
633  // Add data to triangle buffer
634  if (updateTriangles) {
635  int oldTriangleCount = submesh.Count;
636  { //submesh.Resize(oldTriangleCount + attachmentIndexCount);
637  int newTriangleCount = oldTriangleCount + attachmentIndexCount;
638  if (newTriangleCount > submesh.Items.Length) Array.Resize(ref submesh.Items, newTriangleCount);
639  submesh.Count = newTriangleCount;
640  }
641  var submeshItems = submesh.Items;
642  for (int i = 0; i < attachmentIndexCount; i++)
643  submeshItems[oldTriangleCount + i] = attachmentTriangleIndices[i] + ovc;
644  }
645  }
646

1138  int i1 = triangles[t + 0];

1140  int i3 = triangles[t + 2];
1141

1143  Vector3 v2 = vertices[i2];
1144  Vector3 v3 = vertices[i3];

1147  Vector2 w2 = uvs[i2];
1148  Vector2 w3 = uvs[i3];
1149

1151  float x2 = v3.x - v1.x;
1152  float y1 = v2.y - v1.y;
1153  float y2 = v3.y - v1.y;
1154

1156  float s2 = w3.x - w1.x;
1157  float t1 = w2.y - w1.y;
1158  float t2 = w3.y - w1.y;
1159 
1160  float div = s1 * t2 - s2 * t1;
1161  float r = (div == 0f) ? 0f : 1f / div;
1162 
1163  sdir.x = (t2 * x1 - t1 * x2) * r;
1164  sdir.y = (t2 * y1 - t1 * y2) * r;
1165  tempTanBuffer[i1] = tempTanBuffer[i2] = tempTanBuffer[i3] = sdir;
1166 
1167  tdir.x = (s1 * x2 - s2 * x1) * r;
1168  tdir.y = (s1 * y2 - s2 * y1) * r;
1169  tempTanBuffer[vertexCount + i1] = tempTanBuffer[vertexCount + i2] = tempTanBuffer[vertexCount + i3] = tdir;

1177  internal static void SolveTangents2DBuffer (Vector4[] tangents, Vector2[] tempTanBuffer, int vertexCount) {
1178  Vector4 tangent;
1179  tangent.z = 0;
1180  for (int i = 0; i < vertexCount; ++i) {
1181  Vector2 t = tempTanBuffer[i];
1182 
1183  // t.Normalize() (aggressively inlined). Even better if offloaded to GPU via vertex shader.
1184  float magnitude = Mathf.Sqrt(t.x * t.x + t.y * t.y);
1185  if (magnitude > 1E-05) {
1186  float reciprocalMagnitude = 1f/magnitude;
1187  t.x *= reciprocalMagnitude;
1188  t.y *= reciprocalMagnitude;
1189  }
1190 
1191  Vector2 t2 = tempTanBuffer[vertexCount + i];
1192  tangent.x = t.x;
1193  tangent.y = t.y;
1194  //tangent.z = 0;
1195  tangent.w = (t.y * t2.x > t.x * t2.y) ? 1 : -1; // 2D direction calculation. Used for binormals.
1196  tangents[i] = tangent;
