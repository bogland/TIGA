86  void OnSceneGUI (SceneView sceneView) {

88  return;

90  var transform = skeletonRenderer.transform;

523  static float ViewRound (float x) {
524  const float Factor = 100f;
525  const float Divisor = 1f/Factor;
526  return Mathf.Round(x * Factor) * Divisor;
527  }

529  static Vector2 RoundVector2 (float x, float y) {
530  const float Factor = 100f;
531  const float Divisor = 1f/Factor;
532  return new Vector2(Mathf.Round(x * Factor) * Divisor, Mathf.Round(y * Factor) * Divisor);
533  }

541  void UpdateAttachments () {
542  //skeleton = skeletonRenderer.skeleton;
543  Skin defaultSkin = skeleton.Data.DefaultSkin;
544  Skin skin = skeleton.Skin ?? defaultSkin;
545  bool notDefaultSkin = skin != defaultSkin;
546 
547  attachmentTable.Clear();
548  for (int i = skeleton.Slots.Count - 1; i >= 0; i--) {
549  var attachments = new List<Attachment>();
