95  public void Initialize () {
96  bone = null;
97  valid = skeletonRenderer != null && skeletonRenderer.valid;
98  if (!valid) return;
99 
100  skeletonTransform = skeletonRenderer.transform;
101  skeletonRenderer.OnRebuild -= HandleRebuildRenderer;
102  skeletonRenderer.OnRebuild += HandleRebuildRenderer;
103  skeletonTransformIsParent = Transform.ReferenceEquals(skeletonTransform, transform.parent);
104 
105  if (!string.IsNullOrEmpty(boneName))
106  bone = skeletonRenderer.skeleton.FindBone(boneName);

119  public void LateUpdate () {
120  if (!valid) {
121  Initialize();
122  return;
123  }
124 
125  #if UNITY_EDITOR
126  if (!Application.isPlaying)
127  skeletonTransformIsParent = Transform.ReferenceEquals(skeletonTransform, transform.parent);
128  #endif
129 
130  if (bone == null) {
131  if (string.IsNullOrEmpty(boneName)) return;
132  bone = skeletonRenderer.skeleton.FindBone(boneName);
133  if (!SetBone(boneName)) return;
134  }
135 
136  Transform thisTransform = this.transform;
137  if (skeletonTransformIsParent) {
138  // Recommended setup: Use local transform properties if Spine GameObject is the immediate parent
139  thisTransform.localPosition = new Vector3(bone.worldX, bone.worldY, followZPosition ? 0f : thisTransform.localPosition.z);
140  if (followBoneRotation) {
141  float halfRotation = Mathf.Atan2(bone.c, bone.a) * 0.5f;
142  if (followLocalScale && bone.scaleX < 0) // Negate rotation from negative scaleX. Don't use negative determinant. local scaleY doesn't factor into used rotation.
143  halfRotation += Mathf.PI * 0.5f;
144 
145  var q = default(Quaternion);
146  q.z = Mathf.Sin(halfRotation);
147  q.w = Mathf.Cos(halfRotation);
148  thisTransform.localRotation = q;
149  }
150  } else {
151  // For special cases: Use transform world properties if transform relationship is complicated
152  Vector3 targetWorldPosition = skeletonTransform.TransformPoint(new Vector3(bone.worldX, bone.worldY, 0f));
153  if (!followZPosition) targetWorldPosition.z = thisTransform.position.z;
154 
155  float boneWorldRotation = bone.WorldRotationX;
156 
157  Transform transformParent = thisTransform.parent;
158  if (transformParent != null) {
159  Matrix4x4 m = transformParent.localToWorldMatrix;
160  if (m.m00 * m.m11 - m.m01 * m.m10 < 0) // Determinant2D is negative
161  boneWorldRotation = -boneWorldRotation;
162  }
163 
164  if (followBoneRotation) {
165  Vector3 worldRotation = skeletonTransform.rotation.eulerAngles;
166  if (followLocalScale && bone.scaleX < 0) boneWorldRotation += 180f;
167  #if UNITY_5_6_OR_NEWER
168  thisTransform.SetPositionAndRotation(targetWorldPosition, Quaternion.Euler(worldRotation.x, worldRotation.y, worldRotation.z + boneWorldRotation));
169  #else
170  thisTransform.position = targetWorldPosition;
171  thisTransform.rotation = Quaternion.Euler(worldRotation.x, worldRotation.y, worldRotation.z + bone.WorldRotationX);
172  #endif
173  } else {
174  thisTransform.position = targetWorldPosition;
175  }
176  }
177 
178  Vector3 localScale = followLocalScale ? new Vector3(bone.scaleX, bone.scaleY, 1f) : new Vector3(1f, 1f, 1f);
179  if (followSkeletonFlip) localScale.y *= bone.skeleton.flipX ^ bone.skeleton.flipY ? -1f : 1f;
180  thisTransform.localScale = localScale;
181  }
182  }
