47  public RegionAttachment NewRegionAttachment (Skin skin, string name, string path) {
48  AtlasRegion region = FindRegion(path);
49  if (region == null) throw new ArgumentException(string.Format("Region not found in atlas: {0} (region attachment: {1})", path, name));
50  RegionAttachment attachment = new RegionAttachment(name);
51  attachment.RendererObject = region;
52  attachment.SetUVs(region.u, region.v, region.u2, region.v2, region.rotate);
53  attachment.regionOffsetX = region.offsetX;
54  attachment.regionOffsetY = region.offsetY;
55  attachment.regionWidth = region.width;
56  attachment.regionHeight = region.height;
57  attachment.regionOriginalWidth = region.originalWidth;
58  attachment.regionOriginalHeight = region.originalHeight;
59  return attachment;
60  }
