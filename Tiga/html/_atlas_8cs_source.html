111  private void Load (TextReader reader, string imagesDir, TextureLoader textureLoader) {
112  if (textureLoader == null) throw new ArgumentNullException("textureLoader", "textureLoader cannot be null.");
113  this.textureLoader = textureLoader;
114 
115  string[] tuple = new string[4];
116  AtlasPage page = null;
117  while (true) {
118  string line = reader.ReadLine();
119  if (line == null) break;
120  if (line.Trim().Length == 0)
121  page = null;
122  else if (page == null) {
123  page = new AtlasPage();
124  page.name = line;
125 
126  if (ReadTuple(reader, tuple) == 2) { // size is only optional for an atlas packed with an old TexturePacker.
127  page.width = int.Parse(tuple[0], CultureInfo.InvariantCulture);
128  page.height = int.Parse(tuple[1], CultureInfo.InvariantCulture);
129  ReadTuple(reader, tuple);
130  }
131  page.format = (Format)Enum.Parse(typeof(Format), tuple[0], false);
132 
133  ReadTuple(reader, tuple);
134  page.minFilter = (TextureFilter)Enum.Parse(typeof(TextureFilter), tuple[0], false);
135  page.magFilter = (TextureFilter)Enum.Parse(typeof(TextureFilter), tuple[1], false);
136 
137  string direction = ReadValue(reader);
138  page.uWrap = TextureWrap.ClampToEdge;
139  page.vWrap = TextureWrap.ClampToEdge;
140  if (direction == "x")
141  page.uWrap = TextureWrap.Repeat;
142  else if (direction == "y")
143  page.vWrap = TextureWrap.Repeat;
144  else if (direction == "xy")
145  page.uWrap = page.vWrap = TextureWrap.Repeat;
146 
147  textureLoader.Load(page, Path.Combine(imagesDir, line));
148 
149  pages.Add(page);
150 
151  } else {
152  AtlasRegion region = new AtlasRegion();
153  region.name = line;
154  region.page = page;
155 
156  region.rotate = Boolean.Parse(ReadValue(reader));
157 
158  ReadTuple(reader, tuple);
159  int x = int.Parse(tuple[0], CultureInfo.InvariantCulture);
160  int y = int.Parse(tuple[1], CultureInfo.InvariantCulture);
161 
162  ReadTuple(reader, tuple);
163  int width = int.Parse(tuple[0], CultureInfo.InvariantCulture);
164  int height = int.Parse(tuple[1], CultureInfo.InvariantCulture);
165 
166  region.u = x / (float)page.width;
167  region.v = y / (float)page.height;
168  if (region.rotate) {
169  region.u2 = (x + height) / (float)page.width;
170  region.v2 = (y + width) / (float)page.height;
171  } else {
172  region.u2 = (x + width) / (float)page.width;
173  region.v2 = (y + height) / (float)page.height;
174  }
175  region.x = x;
176  region.y = y;
177  region.width = Math.Abs(width);
178  region.height = Math.Abs(height);
179 
180  if (ReadTuple(reader, tuple) == 4) { // split is optional
181  region.splits = new [] {int.Parse(tuple[0], CultureInfo.InvariantCulture),
182  int.Parse(tuple[1], CultureInfo.InvariantCulture),
183  int.Parse(tuple[2], CultureInfo.InvariantCulture),
184  int.Parse(tuple[3], CultureInfo.InvariantCulture)};
185 
186  if (ReadTuple(reader, tuple) == 4) { // pad is optional, but only present with splits
187  region.pads = new [] {int.Parse(tuple[0], CultureInfo.InvariantCulture),
188  int.Parse(tuple[1], CultureInfo.InvariantCulture),
189  int.Parse(tuple[2], CultureInfo.InvariantCulture),
190  int.Parse(tuple[3], CultureInfo.InvariantCulture)};
191 
192  ReadTuple(reader, tuple);
193  }
194  }
195 
196  region.originalWidth = int.Parse(tuple[0], CultureInfo.InvariantCulture);
197  region.originalHeight = int.Parse(tuple[1], CultureInfo.InvariantCulture);
198 
199  ReadTuple(reader, tuple);
200  region.offsetX = int.Parse(tuple[0], CultureInfo.InvariantCulture);
201  region.offsetY = int.Parse(tuple[1], CultureInfo.InvariantCulture);
202 
203  region.index = int.Parse(ReadValue(reader), CultureInfo.InvariantCulture);
204 
205  regions.Add(region);
206  }
207  }

210  static string ReadValue (TextReader reader) {
211  string line = reader.ReadLine();
212  int colon = line.IndexOf(':');
213  if (colon == -1) throw new Exception("Invalid line: " + line);
214  return line.Substring(colon + 1).Trim();
215  }
216 
218  static int ReadTuple (TextReader reader, string[] tuple) {
219  string line = reader.ReadLine();
220  int colon = line.IndexOf(':');
221  if (colon == -1) throw new Exception("Invalid line: " + line);
222  int i = 0, lastMatch = colon + 1;
223  for (; i < 3; i++) {
224  int comma = line.IndexOf(',', lastMatch);
225  if (comma == -1) break;
226  tuple[i] = line.Substring(lastMatch, comma - lastMatch).Trim();
227  lastMatch = comma + 1;
228  }
229  tuple[i] = line.Substring(lastMatch).Trim();
230  return i + 1;
231  }
