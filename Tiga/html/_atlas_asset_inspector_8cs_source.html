298  static public void UpdateSpriteSlices (Texture texture, Atlas atlas) {
299  string texturePath = AssetDatabase.GetAssetPath(texture.GetInstanceID());
300  var t = (TextureImporter)TextureImporter.GetAtPath(texturePath);
301  t.spriteImportMode = SpriteImportMode.Multiple;
302  var spriteSheet = t.spritesheet;
303  var sprites = new List<SpriteMetaData>(spriteSheet);
304 
305  var regions = AtlasAssetInspector.GetRegions(atlas);
306  char[] FilenameDelimiter = {'.'};
307  int updatedCount = 0;
308  int addedCount = 0;
309 
310  foreach (var r in regions) {
311  string pageName = r.page.name.Split(FilenameDelimiter, StringSplitOptions.RemoveEmptyEntries)[0];
312  string textureName = texture.name;
313  bool pageMatch = string.Equals(pageName, textureName, StringComparison.Ordinal);
314 
315 // if (pageMatch) {
316 // int pw = r.page.width;
317 // int ph = r.page.height;
318 // bool mismatchSize = pw != texture.width || pw > t.maxTextureSize || ph != texture.height || ph > t.maxTextureSize;
319 // if (mismatchSize)
320 // Debug.LogWarningFormat("Size mismatch found.\nExpected atlas size is {0}x{1}. Texture Import Max Size of texture '{2}'({4}x{5}) is currently set to {3}.", pw, ph, texture.name, t.maxTextureSize, texture.width, texture.height);
321 // }
322 
323  int spriteIndex = pageMatch ? sprites.FindIndex(
324  (s) => string.Equals(s.name, r.name, StringComparison.Ordinal)
325  ) : -1;
326  bool spriteNameMatchExists = spriteIndex >= 0;
327 
328  if (pageMatch) {
329  Rect spriteRect = new Rect();
330 
331  if (r.rotate) {
332  spriteRect.width = r.height;
333  spriteRect.height = r.width;
334  } else {
335  spriteRect.width = r.width;
336  spriteRect.height = r.height;
337  }
338  spriteRect.x = r.x;
339  spriteRect.y = r.page.height - spriteRect.height - r.y;
340 
341  if (spriteNameMatchExists) {
342  var s = sprites[spriteIndex];
343  s.rect = spriteRect;
344  sprites[spriteIndex] = s;
345  updatedCount++;
346  } else {
347  sprites.Add(new SpriteMetaData {
348  name = r.name,
349  pivot = new Vector2(0.5f, 0.5f),
350  rect = spriteRect
351  });
352  addedCount++;
353  }
354  }
355 
356  }
357 
358  t.spritesheet = sprites.ToArray();
359  EditorUtility.SetDirty(t);
360  AssetDatabase.ImportAsset(texturePath, ImportAssetOptions.ForceUpdate);
361  EditorGUIUtility.PingObject(texture);
362  Debug.Log(string.Format("Applied sprite slices to {2}. {0} added. {1} updated.", addedCount, updatedCount, texture.name));
363  }
364  }
