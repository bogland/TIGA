221  public void AddPartsRenderer (int count) {
222  var componentRenderers = component.partsRenderers;
223  bool emptyFound = componentRenderers.Contains(null);
224  if (emptyFound) {
225  bool userClearEntries = EditorUtility.DisplayDialog("Empty entries found", "Null entries found. Do you want to remove null entries before adding the new renderer? ", "Clear Empty Entries", "Don't Clear");
226  if (userClearEntries) componentRenderers.RemoveAll(x => x == null);
227  }
228 
229  Undo.RegisterCompleteObjectUndo(component, "Add Parts Renderers");
230  for (int i = 0; i < count; i++) {
231  int index = componentRenderers.Count;
232  var smr = SkeletonPartsRenderer.NewPartsRendererGameObject(component.transform, index.ToString());
233  Undo.RegisterCreatedObjectUndo(smr.gameObject, "New Parts Renderer GameObject.");
234  componentRenderers.Add(smr);
235 
236  // increment renderer sorting order.
237  if (index == 0) continue;
238  var prev = componentRenderers[index - 1]; if (prev == null) continue;
239 
240  var prevMeshRenderer = prev.GetComponent<MeshRenderer>();
241  var currentMeshRenderer = smr.GetComponent<MeshRenderer>();
242  if (prevMeshRenderer == null || currentMeshRenderer == null) continue;
243 
244  int prevSortingLayer = prevMeshRenderer.sortingLayerID;
245  int prevSortingOrder = prevMeshRenderer.sortingOrder;
246  currentMeshRenderer.sortingLayerID = prevSortingLayer;
247  currentMeshRenderer.sortingOrder = prevSortingOrder + SkeletonRenderSeparator.DefaultSortingOrderIncrement;
248  }

253  public void DetectOrphanedPartsRenderers (SkeletonRenderSeparator component) {
254  var children = component.GetComponentsInChildren<SkeletonPartsRenderer>();
255 
256  var orphans = new System.Collections.Generic.List<SkeletonPartsRenderer>();
257  foreach (var r in children) {
258  if (!component.partsRenderers.Contains(r))
259  orphans.Add(r);
260  }
261 
262  if (orphans.Count > 0) {
263  if (EditorUtility.DisplayDialog("Destroy Submesh Renderers", "Unassigned renderers were found. Do you want to delete them? (These may belong to another Render Separator in the same hierarchy. If you don't have another Render Separator component in the children of this GameObject, it's likely safe to delete. Warning: This operation cannot be undone.)", "Delete", "Cancel")) {
264  foreach (var o in orphans) {
265  Undo.DestroyObjectImmediate(o.gameObject);
266  }
267  }
268  }
