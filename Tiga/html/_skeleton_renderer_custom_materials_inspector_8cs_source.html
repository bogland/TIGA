78  public override void OnInspectorGUI () {
79  component = (SkeletonRendererCustomMaterials)target;
80  var skeletonRenderer = component.skeletonRenderer;
81 
82  // Draw the default inspector
83  DrawDefaultInspector();
84 
85  if (serializedObject.isEditingMultipleObjects)
86  return;
87 
88  if (componentCustomMaterialOverrides == null) {
89  Type cm = typeof(SkeletonRendererCustomMaterials);
90  componentCustomMaterialOverrides = cm.GetField("customMaterialOverrides", PrivateInstance).GetValue(component) as List<SkeletonRendererCustomMaterials.AtlasMaterialOverride>;
91  componentCustomSlotMaterials = cm.GetField("customSlotMaterials", PrivateInstance).GetValue(component) as List<SkeletonRendererCustomMaterials.SlotMaterialOverride>;
92  if (componentCustomMaterialOverrides == null) {
93  Debug.Log("Reflection failed.");
94  return;
95  }
96  }
97 
98  // Fill with current values at start
99  if (_customMaterialOverridesPrev == null || _customSlotMaterialsPrev == null) {
100  _customMaterialOverridesPrev = CopyList(componentCustomMaterialOverrides);
101  _customSlotMaterialsPrev = CopyList(componentCustomSlotMaterials);
102  }
103 
104  // Compare new values with saved. If change is detected:
105  // store new values, restore old values, remove overrides, restore new values, restore overrides.
106 
107  // 1. Store new values
108  var customMaterialOverridesNew = CopyList(componentCustomMaterialOverrides);
109  var customSlotMaterialsNew = CopyList(componentCustomSlotMaterials);
110 
111  // Detect changes
112  if (!_customMaterialOverridesPrev.SequenceEqual(customMaterialOverridesNew) ||
113  !_customSlotMaterialsPrev.SequenceEqual(customSlotMaterialsNew)) {
114  // 2. Restore old values
115  componentCustomMaterialOverrides.Clear();
116  componentCustomSlotMaterials.Clear();
117  componentCustomMaterialOverrides.AddRange(_customMaterialOverridesPrev);
118  componentCustomSlotMaterials.AddRange(_customSlotMaterialsPrev);
119 
120  // 3. Remove overrides
121  RemoveCustomMaterials();
122 
123  // 4. Restore new values
124  componentCustomMaterialOverrides.Clear();
125  componentCustomSlotMaterials.Clear();
126  componentCustomMaterialOverrides.AddRange(customMaterialOverridesNew);
127  componentCustomSlotMaterials.AddRange(customSlotMaterialsNew);
128 
129  // 5. Restore overrides
130  SetCustomMaterials();
131 
132  if (skeletonRenderer != null)
133  skeletonRenderer.LateUpdate();
134  }
135 
136  _customMaterialOverridesPrev = CopyList(componentCustomMaterialOverrides);
137  _customSlotMaterialsPrev = CopyList(componentCustomSlotMaterials);
138 
139  if (SpineInspectorUtility.LargeCenteredButton(SpineInspectorUtility.TempContent("Clear and Reapply Changes", tooltip: "Removes all non-serialized overrides in the SkeletonRenderer and reapplies the overrides on this component."))) {
140  if (skeletonRenderer != null) {
141  #if SPINE_OPTIONAL_MATERIALOVERRIDE
142  skeletonRenderer.CustomMaterialOverride.Clear();
143  #endif
144  skeletonRenderer.CustomSlotMaterials.Clear();
145  RemoveCustomMaterials();
146  SetCustomMaterials();
147  skeletonRenderer.LateUpdate();
148  }
149  }
150  }
