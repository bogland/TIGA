55  #if UNITY_EDITOR
56  protected override void OnValidate () {
57  // This handles Scene View preview.
58  base.OnValidate ();
59  if (this.IsValid) {
60  if (skeletonDataAsset == null) {
61  Clear();
62  startingAnimation = "";
63  } else if (skeletonDataAsset.GetSkeletonData(true) != skeleton.data) {
64  Clear();
65  Initialize(true);
66  startingAnimation = "";
67  if (skeletonDataAsset.atlasAssets.Length > 1 || skeletonDataAsset.atlasAssets[0].materials.Length > 1)
68  Debug.LogError("Unity UI does not support multiple textures per Renderer. Your skeleton will not be rendered correctly. Recommend using SkeletonAnimation instead. This requires the use of a Screen space camera canvas.");
69  } else {
70  if (freeze) return;
71 
72  if (!string.IsNullOrEmpty(initialSkinName)) {
73  var skin = skeleton.data.FindSkin(initialSkinName);
74  if (skin != null) {
75  if (skin == skeleton.data.defaultSkin)
76  skeleton.SetSkin((Skin)null);
77  else
78  skeleton.SetSkin(skin);
79  }
80 
81  }
82 
83  // Only provide visual feedback to inspector changes in Unity Editor Edit mode.
84  if (!Application.isPlaying) {
85  skeleton.flipX = this.initialFlipX;
86  skeleton.flipY = this.initialFlipY;
87 
88  skeleton.SetToSetupPose();
89  if (!string.IsNullOrEmpty(startingAnimation))
90  skeleton.PoseWithAnimation(startingAnimation, 0f, false);
91  }
92 
93  }
94  } else {
95  if (skeletonDataAsset != null)
96  Initialize(true);
97  }
98  }
99 
100  protected override void Reset () {
101  base.Reset();
102  if (material == null || material.shader != Shader.Find("Spine/SkeletonGraphic (Premultiply Alpha)"))
103  Debug.LogWarning("SkeletonGraphic works best with the SkeletonGraphic material.");
104  }

219  public void Initialize (bool overwrite) {
220  if (this.IsValid && !overwrite) return;
221 
222  // Make sure none of the stuff is null
223  if (this.skeletonDataAsset == null) return;
224  var skeletonData = this.skeletonDataAsset.GetSkeletonData(false);
225  if (skeletonData == null) return;
226 
227  if (skeletonDataAsset.atlasAssets.Length <= 0 || skeletonDataAsset.atlasAssets[0].materials.Length <= 0) return;
228 
229  this.state = new Spine.AnimationState(skeletonDataAsset.GetAnimationStateData());
230  if (state == null) {
231  Clear();
232  return;
233  }
234 
235  this.skeleton = new Skeleton(skeletonData) {
236  flipX = this.initialFlipX,
237  flipY = this.initialFlipY
238  };
239 
240  meshBuffers = new DoubleBuffered<MeshRendererBuffers.SmartMesh>();
241  canvasRenderer.SetTexture(this.mainTexture); // Needed for overwriting initializations.
242 
243  // Set the initial Skin and Animation
244  if (!string.IsNullOrEmpty(initialSkinName))
245  skeleton.SetSkin(initialSkinName);
246 
247  #if UNITY_EDITOR
248  if (!string.IsNullOrEmpty(startingAnimation)) {
249  if (Application.isPlaying) {
250  state.SetAnimation(0, startingAnimation, startingLoop);
251  } else {
252  // Assume SkeletonAnimation is valid for skeletonData and skeleton. Checked above.
253  var animationObject = skeletonDataAsset.GetSkeletonData(false).FindAnimation(startingAnimation);
254  if (animationObject != null)
255  animationObject.PoseSkeleton(skeleton, 0);
256  }
257  Update(0);
258  }
259  #else
260  if (!string.IsNullOrEmpty(startingAnimation)) {
261  state.SetAnimation(0, startingAnimation, startingLoop);
262  Update(0);
263  }
264  #endif
265  }
266 
267  public void UpdateMesh () {
268  if (!this.IsValid) return;
269 
270  skeleton.SetColor(this.color);
271  var smartMesh = meshBuffers.GetNext();
272  var currentInstructions = this.currentInstructions;
273 
274  MeshGenerator.GenerateSingleSubmeshInstruction(currentInstructions, skeleton, this.material);
275  bool updateTriangles = SkeletonRendererInstruction.GeometryNotEqual(currentInstructions, smartMesh.instructionUsed);
276 
277  meshGenerator.Begin();
278  if (currentInstructions.hasActiveClipping) {
279  meshGenerator.AddSubmesh(currentInstructions.submeshInstructions.Items[0], updateTriangles);
280  } else {
281  meshGenerator.BuildMeshWithArrays(currentInstructions, updateTriangles);
282  }
283 
284  if (canvas != null) meshGenerator.ScaleVertexData(canvas.referencePixelsPerUnit);
285  if (OnPostProcessVertices != null) OnPostProcessVertices.Invoke(this.meshGenerator.Buffers);
286 
287  var mesh = smartMesh.mesh;
288  meshGenerator.FillVertexData(mesh);
289  if (updateTriangles) meshGenerator.FillTrianglesSingle(mesh);
290  meshGenerator.FillLateVertexData(mesh);
291 
292  canvasRenderer.SetMesh(mesh);
293  smartMesh.instructionUsed.Set(currentInstructions);
294 
295  //this.UpdateMaterial(); // TODO: This allocates memory.
296  }
297  #endregion
298  }
