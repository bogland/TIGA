38  [RequireComponent(typeof(SkeletonRenderer))]
39  public class SkeletonGhost : MonoBehaviour {
40  // Internal Settings
41  const HideFlags GhostHideFlags = HideFlags.HideInHierarchy;
42  const string GhostingShaderName = "Spine/Special/SkeletonGhost";
43 
44  public bool ghostingEnabled = true;
45  public float spawnRate = 0.05f;
46  public Color32 color = new Color32(0xFF, 0xFF, 0xFF, 0x00); // default for additive.
47  [Tooltip("Remember to set color alpha to 0 if Additive is true")]
48  public bool additive = true;
49  public int maximumGhosts = 10;
50  public float fadeSpeed = 10;
51  public Shader ghostShader;
52  [Tooltip("0 is Color and Alpha, 1 is Alpha only.")]
53  [Range(0, 1)]
54  public float textureFade = 1;
55 
56  [Header("Sorting")]
57  public bool sortWithDistanceOnly;
58  public float zOffset = 0f;
59 
60  float nextSpawnTime;
61  SkeletonGhostRenderer[] pool;
62  int poolIndex = 0;
63  SkeletonRenderer skeletonRenderer;
64  MeshRenderer meshRenderer;
65  MeshFilter meshFilter;
66 
67  readonly Dictionary<Material, Material> materialTable = new Dictionary<Material, Material>();
68 
69  void Start () {
70  if (ghostShader == null)
71  ghostShader = Shader.Find(GhostingShaderName);
72 
73  skeletonRenderer = GetComponent<SkeletonRenderer>();
74  meshFilter = GetComponent<MeshFilter>();
75  meshRenderer = GetComponent<MeshRenderer>();
76  nextSpawnTime = Time.time + spawnRate;
77  pool = new SkeletonGhostRenderer[maximumGhosts];
78  for (int i = 0; i < maximumGhosts; i++) {
79  GameObject go = new GameObject(gameObject.name + " Ghost", typeof(SkeletonGhostRenderer));
80  pool[i] = go.GetComponent<SkeletonGhostRenderer>();
81  go.SetActive(false);
82  go.hideFlags = GhostHideFlags;
83  }
84 
85  var skeletonAnimation = skeletonRenderer as Spine.Unity.IAnimationStateComponent;
86  if (skeletonAnimation != null) skeletonAnimation.AnimationState.Event += OnEvent;
87  }
88 
89  //SkeletonAnimation
90  /*
91  * Int Value: 0 sets ghostingEnabled to false, 1 sets ghostingEnabled to true
92  * Float Value: Values greater than 0 set the spawnRate equal the float value
93  * String Value: Pass RGBA hex color values in to set the color property. IE: "A0FF8BFF"
94  */
95  void OnEvent (Spine.TrackEntry trackEntry, Spine.Event e) {
96  if (e.Data.Name.Equals("Ghosting", System.StringComparison.Ordinal)) {
97  ghostingEnabled = e.Int > 0;
98  if (e.Float > 0)
99  spawnRate = e.Float;
100 
101  if (!string.IsNullOrEmpty(e.stringValue))
102  this.color = HexToColor(e.String);
103  }
104  }
105 
106  //SkeletonAnimator
107  //SkeletonAnimator or Mecanim based animations only support toggling ghostingEnabled. Be sure not to set anything other than the Int param in Spine or String will take priority.
108  void Ghosting (float val) {
109  ghostingEnabled = val > 0;
110  }
111 
112  void Update () {
113  if (!ghostingEnabled)
114  return;
115 
116  if (Time.time >= nextSpawnTime) {
117  GameObject go = pool[poolIndex].gameObject;
118 
119  Material[] materials = meshRenderer.sharedMaterials;
120  for (int i = 0; i < materials.Length; i++) {
121  var originalMat = materials[i];
122  Material ghostMat;
123  if (!materialTable.ContainsKey(originalMat)) {
124  ghostMat = new Material(originalMat);
125  ghostMat.shader = ghostShader;
126  ghostMat.color = Color.white;
127  if (ghostMat.HasProperty("_TextureFade"))
128  ghostMat.SetFloat("_TextureFade", textureFade);
129  materialTable.Add(originalMat, ghostMat);
130  } else {
131  ghostMat = materialTable[originalMat];
132  }
133 
134  materials[i] = ghostMat;
135  }
136 
137  var goTransform = go.transform;
138  goTransform.parent = transform;
139 
140  pool[poolIndex].Initialize(meshFilter.sharedMesh, materials, color, additive, fadeSpeed, meshRenderer.sortingLayerID, (sortWithDistanceOnly) ? meshRenderer.sortingOrder : meshRenderer.sortingOrder - 1);
141 
142  goTransform.localPosition = new Vector3(0f, 0f, zOffset);
143  goTransform.localRotation = Quaternion.identity;
144  goTransform.localScale = Vector3.one;
145 
146  goTransform.parent = null;
147 
148  poolIndex++;
149 
150  if (poolIndex == pool.Length)
151  poolIndex = 0;
152 
153  nextSpawnTime = Time.time + spawnRate;
154  }
155  }
156 
157  void OnDestroy () {
158  if (pool != null) {
159  for (int i = 0; i < maximumGhosts; i++)
160  if (pool[i] != null) pool[i].Cleanup();
161  }
162 
163  foreach (var mat in materialTable.Values)
164  Destroy(mat);
165  }
166 
167  //based on UnifyWiki http://wiki.unity3d.com/index.php?title=HexConverter
168  static Color32 HexToColor (string hex) {
169  const System.Globalization.NumberStyles HexStyle = System.Globalization.NumberStyles.HexNumber;
170 
171  if (hex.Length < 6)
172  return Color.magenta;
173 
174  hex = hex.Replace("#", "");
175  byte r = byte.Parse(hex.Substring(0, 2), HexStyle);
176  byte g = byte.Parse(hex.Substring(2, 2), HexStyle);
177  byte b = byte.Parse(hex.Substring(4, 2), HexStyle);
178  byte a = 0xFF;
179  if (hex.Length == 8)
180  a = byte.Parse(hex.Substring(6, 2), HexStyle);
181 
182  return new Color32(r, g, b, a);
183  }
184  }
