44  // NOTE: This function is called at runtime and edit time. Keep that in mind when setting the values of properties.
45  public override void ProcessFrame (Playable playable, FrameData info, object playerData) {
46  var spineComponent = playerData as SkeletonAnimation;
47  if (spineComponent == null) return;
48 
49  var skeleton = spineComponent.Skeleton;
50  var state = spineComponent.AnimationState;
51 
52  if (!Application.isPlaying) {
53  #if SPINE_EDITMODEPOSE
54  PreviewEditModePose(playable, spineComponent);
55  #endif
56  return;
57  }
58 
59  int inputCount = playable.GetInputCount();
60 
61  // Ensure correct buffer size.
62  if (this.lastInputWeights == null || this.lastInputWeights.Length < inputCount) {
63  this.lastInputWeights = new float[inputCount];
64 
65  for (int i = 0; i < inputCount; i++)
66  this.lastInputWeights[i] = default(float);
67  }
68  var lastInputWeights = this.lastInputWeights;
69 
70  // Check all clips. If a clip that was weight 0 turned into weight 1, call SetAnimation.
71  for (int i = 0; i < inputCount; i++) {
72  float lastInputWeight = lastInputWeights[i];
73  float inputWeight = playable.GetInputWeight(i);
74  bool trackStarted = inputWeight > lastInputWeight;
75  lastInputWeights[i] = inputWeight;
76 
77  if (trackStarted) {
78  ScriptPlayable<SpineAnimationStateBehaviour> inputPlayable = (ScriptPlayable<SpineAnimationStateBehaviour>)playable.GetInput(i);
79  SpineAnimationStateBehaviour clipData = inputPlayable.GetBehaviour();
80 
81  if (clipData.animationReference == null) {
82  float mixDuration = clipData.customDuration ? clipData.mixDuration : state.Data.DefaultMix;
83  state.SetEmptyAnimation(0, mixDuration);
84  } else {
85  if (clipData.animationReference.Animation != null) {
86  Spine.TrackEntry trackEntry = state.SetAnimation(0, clipData.animationReference.Animation, clipData.loop);
87 
88  //trackEntry.TrackTime = (float)inputPlayable.GetTime(); // More accurate time-start?
89  trackEntry.EventThreshold = clipData.eventThreshold;
90  trackEntry.DrawOrderThreshold = clipData.drawOrderThreshold;
91  trackEntry.AttachmentThreshold = clipData.attachmentThreshold;
92 
93  if (clipData.customDuration)
94  trackEntry.MixDuration = clipData.mixDuration;
95  }
96  //else Debug.LogWarningFormat("Animation named '{0}' not found", clipData.animationName);
97  }
98 
99  // Ensure that the first frame ends with an updated mesh.
100  spineComponent.Update(0);
101  spineComponent.LateUpdate();
102  }
103  }
104  }

106  #if SPINE_EDITMODEPOSE
107  public void PreviewEditModePose (Playable playable, SkeletonAnimation spineComponent) {
108  if (Application.isPlaying) return;
109  if (spineComponent == null) return;
110 
111  int inputCount = playable.GetInputCount();
112  int lastOneWeight = -1;
113 
114  for (int i = 0; i < inputCount; i++) {
115  float inputWeight = playable.GetInputWeight(i);
116  if (inputWeight >= 1) lastOneWeight = i;
117  }
118 
119  if (lastOneWeight != -1) {
120  ScriptPlayable<SpineAnimationStateBehaviour> inputPlayableClip = (ScriptPlayable<SpineAnimationStateBehaviour>)playable.GetInput(lastOneWeight);
121  SpineAnimationStateBehaviour clipData = inputPlayableClip.GetBehaviour();
122 
123  var skeleton = spineComponent.Skeleton;
124 
125  bool skeletonDataMismatch = clipData.animationReference != null && spineComponent.SkeletonDataAsset.GetSkeletonData(true) != clipData.animationReference.SkeletonDataAsset.GetSkeletonData(true);
126  if (skeletonDataMismatch) {
127  Debug.LogWarningFormat("SpineAnimationStateMixerBehaviour tried to apply an animation for the wrong skeleton. Expected {0}. Was {1}", spineComponent.SkeletonDataAsset, clipData.animationReference.SkeletonDataAsset);
128  }
129 
130  // Getting the from-animation here because it's required to get the mix information from AnimationStateData.
131  Animation fromAnimation = null;
132  float fromClipTime = 0;
133  bool fromClipLoop = false;
134  if (lastOneWeight != 0 && inputCount > 1) {
135  var fromClip = (ScriptPlayable<SpineAnimationStateBehaviour>)playable.GetInput(lastOneWeight - 1);
136  var fromClipData = fromClip.GetBehaviour();
137  fromAnimation = fromClipData.animationReference.Animation;
138  fromClipTime = (float)fromClip.GetTime();
139  fromClipLoop = fromClipData.loop;
140  }
141 
142  Animation toAnimation = clipData.animationReference.Animation;
143  float toClipTime = (float)inputPlayableClip.GetTime();
144  float mixDuration = clipData.mixDuration;
145 
146  if (!clipData.customDuration && fromAnimation != null) {
147  mixDuration = spineComponent.AnimationState.Data.GetMix(fromAnimation, toAnimation);
148  }
149 
150  // Approximate what AnimationState might do at runtime.
151  if (fromAnimation != null && mixDuration > 0 && toClipTime < mixDuration) {
152  skeleton.SetToSetupPose();
153  float fauxFromAlpha = (1f - toClipTime/mixDuration);
154  fauxFromAlpha = fauxFromAlpha > 0.5f ? 1f : fauxFromAlpha * 2f; // fake value, but reduce dip.
155  fromAnimation.Apply(skeleton, 0, fromClipTime, fromClipLoop, null, fauxFromAlpha, MixPose.Setup, MixDirection.Out); //fromAnimation.PoseSkeleton(skeleton, fromClipTime, fromClipLoop);
156  toAnimation.Apply(skeleton, 0, toClipTime, clipData.loop, null, toClipTime/mixDuration, MixPose.Current, MixDirection.In);
157  } else {
158  skeleton.SetToSetupPose();
159  toAnimation.PoseSkeleton(skeleton, toClipTime, clipData.loop);
160  }
161 
162  }
163  // Do nothing outside of the first clip and the last clip.
