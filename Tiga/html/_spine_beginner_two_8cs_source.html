79  IEnumerator DoDemoRoutine () {
80  while (true) {
81  // SetAnimation is the basic way to set an animation.
82  // SetAnimation sets the animation and starts playing it from the beginning.
83  // Common Mistake: If you keep calling it in Update, it will keep showing the first pose of the animation, do don't do that.
84 
85  spineAnimationState.SetAnimation(0, walkAnimationName, true);
86  yield return new WaitForSeconds(runWalkDuration);
87 
88  spineAnimationState.SetAnimation(0, runAnimationName, true);
89  yield return new WaitForSeconds(runWalkDuration);
90 
91  // AddAnimation queues up an animation to play after the previous one ends.
92  spineAnimationState.SetAnimation(0, runToIdleAnimationName, false);
93  spineAnimationState.AddAnimation(0, idleAnimationName, true, 0);
94  yield return new WaitForSeconds(1f);
95 
96  skeleton.FlipX = true; // skeleton allows you to flip the skeleton.
97  spineAnimationState.SetAnimation(0, idleTurnAnimationName, false);
98  spineAnimationState.AddAnimation(0, idleAnimationName, true, 0);
99  yield return new WaitForSeconds(0.5f);
100  skeleton.FlipX = false;
101  spineAnimationState.SetAnimation(0, idleTurnAnimationName, false);
102  spineAnimationState.AddAnimation(0, idleAnimationName, true, 0);
103  yield return new WaitForSeconds(0.5f);
104 
105  }
