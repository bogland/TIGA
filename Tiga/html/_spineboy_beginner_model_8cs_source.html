57  public void TryShoot () {
58  float currentTime = Time.time;
59 
60  if (currentTime - lastShootTime > shootInterval) {
61  lastShootTime = currentTime;
62  if (ShootEvent != null) ShootEvent(); // Fire the "ShootEvent" event.
63  }
64  }

66  public void TryMove (float speed) {
67  currentSpeed = speed; // show the "speed" in the Inspector.
68 
69  if (speed != 0) {
70  bool speedIsNegative = (speed < 0f);
71  facingLeft = speedIsNegative; // Change facing direction whenever speed is not 0.
72  }
73 
74  if (state != SpineBeginnerBodyState.Jumping) {
75  state = (speed == 0) ? SpineBeginnerBodyState.Idle : SpineBeginnerBodyState.Running;
76  }
77 
78  }
79  #endregion
80 
81  IEnumerator JumpRoutine () {
82  if (state == SpineBeginnerBodyState.Jumping) yield break; // Don't jump when already jumping.
83 
84  state = SpineBeginnerBodyState.Jumping;
85 
86  // Fake jumping.
87  {
88  var pos = transform.localPosition;
89  const float jumpTime = 1.2f;
90  const float half = jumpTime * 0.5f;
91  const float jumpPower = 20f;
92  for (float t = 0; t < half; t += Time.deltaTime) {
93  float d = jumpPower * (half - t);
94  transform.Translate((d * Time.deltaTime) * Vector3.up);
95  yield return null;
96  }
97  for (float t = 0; t < half; t += Time.deltaTime) {
98  float d = jumpPower * t;
99  transform.Translate((d * Time.deltaTime) * Vector3.down);
100  yield return null;
101  }
102  transform.localPosition = pos;
103  }
104 
105  state = SpineBeginnerBodyState.Idle;
106  }
107 
108  }
