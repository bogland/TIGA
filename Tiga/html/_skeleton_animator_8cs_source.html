155  if (layerMixModes.Length < animator.layerCount)
156  System.Array.Resize<MixMode>(ref layerMixModes, animator.layerCount);
157 
158  InitClipInfosForLayers();
159  for (int layer = 0, n = animator.layerCount; layer < n; layer++) {
160  GetStateUpdatesFromAnimator(layer);
161  }
162 
163  //skeleton.Update(Time.deltaTime); // Doesn't actually do anything, currently. (Spine 3.6).
164 
165  // Clear Previous
166  if (autoReset) {
167  var previousAnimations = this.previousAnimations;
168  for (int i = 0, n = previousAnimations.Count; i < n; i++)
169  previousAnimations[i].SetKeyedItemsToSetupPose(skeleton);
170 
171  previousAnimations.Clear();
172  for (int layer = 0, n = animator.layerCount; layer < n; layer++) {
173  float layerWeight = (layer == 0) ? 1 : animator.GetLayerWeight(layer); // Animator.GetLayerWeight always returns 0 on the first layer. Should be interpreted as 1.
174  if (layerWeight <= 0) continue;
175 
176  AnimatorStateInfo nextStateInfo = animator.GetNextAnimatorStateInfo(layer);
177 
178  bool hasNext = nextStateInfo.fullPathHash != 0;
179 
180  int clipInfoCount, nextClipInfoCount, interruptingClipInfoCount;
181  IList<AnimatorClipInfo> clipInfo, nextClipInfo, interruptingClipInfo;
182  bool isInterruptionActive, shallInterpolateWeightTo1;
183  GetAnimatorClipInfos(layer, out isInterruptionActive, out clipInfoCount, out nextClipInfoCount, out interruptingClipInfoCount,
184  out clipInfo, out nextClipInfo, out interruptingClipInfo, out shallInterpolateWeightTo1);
185 
186  for (int c = 0; c < clipInfoCount; c++) {
187  var info = clipInfo[c];
188  float weight = info.weight * layerWeight; if (weight == 0) continue;
189  previousAnimations.Add(GetAnimation(info.clip));
190  }
191 
192  if (hasNext) {
193  for (int c = 0; c < nextClipInfoCount; c++) {
194  var info = nextClipInfo[c];
195  float weight = info.weight * layerWeight; if (weight == 0) continue;
196  previousAnimations.Add(GetAnimation(info.clip));
197  }
198  }
199 
200  if (isInterruptionActive) {
201  for (int c = 0; c < interruptingClipInfoCount; c++)
202  {
203  var info = interruptingClipInfo[c];
204  float clipWeight = shallInterpolateWeightTo1 ? (info.weight + 1.0f) * 0.5f : info.weight;
205  float weight = clipWeight * layerWeight; if (weight == 0) continue;
206  previousAnimations.Add(GetAnimation(info.clip));
207  }
208  }
209  }
210  }
211 
212  // Apply
213  for (int layer = 0, n = animator.layerCount; layer < n; layer++) {
214  float layerWeight = (layer == 0) ? 1 : animator.GetLayerWeight(layer); // Animator.GetLayerWeight always returns 0 on the first layer. Should be interpreted as 1.
215 
216  bool isInterruptionActive;
217  AnimatorStateInfo stateInfo;
218  AnimatorStateInfo nextStateInfo;
219  AnimatorStateInfo interruptingStateInfo;
220  float interruptingClipTimeAddition;
221  GetAnimatorStateInfos(layer, out isInterruptionActive, out stateInfo, out nextStateInfo, out interruptingStateInfo, out interruptingClipTimeAddition);
222 
223  bool hasNext = nextStateInfo.fullPathHash != 0;
224 
225  int clipInfoCount, nextClipInfoCount, interruptingClipInfoCount;
226  IList<AnimatorClipInfo> clipInfo, nextClipInfo, interruptingClipInfo;
227  bool shallInterpolateWeightTo1;
228  GetAnimatorClipInfos(layer, out isInterruptionActive, out clipInfoCount, out nextClipInfoCount, out interruptingClipInfoCount,
229  out clipInfo, out nextClipInfo, out interruptingClipInfo, out shallInterpolateWeightTo1);
230 
231  MixMode mode = layerMixModes[layer];
232  if (mode == MixMode.AlwaysMix) {
233  // Always use Mix instead of Applying the first non-zero weighted clip.
234  for (int c = 0; c < clipInfoCount; c++) {
235  var info = clipInfo[c]; float weight = info.weight * layerWeight; if (weight == 0) continue;
236  GetAnimation(info.clip).Apply(skeleton, 0, AnimationTime(stateInfo.normalizedTime, info.clip.length, stateInfo.loop, stateInfo.speed < 0), stateInfo.loop, null, weight, MixPose.Current, MixDirection.In);
237  }
238  if (hasNext) {
239  for (int c = 0; c < nextClipInfoCount; c++) {
240  var info = nextClipInfo[c]; float weight = info.weight * layerWeight; if (weight == 0) continue;
241  GetAnimation(info.clip).Apply(skeleton, 0, AnimationTime(nextStateInfo.normalizedTime , info.clip.length, nextStateInfo.speed < 0), nextStateInfo.loop, null, weight, MixPose.Current, MixDirection.In);
242  }
243  }
244  if (isInterruptionActive) {
245  for (int c = 0; c < interruptingClipInfoCount; c++)
246  {
247  var info = interruptingClipInfo[c];
248  float clipWeight = shallInterpolateWeightTo1 ? (info.weight + 1.0f) * 0.5f : info.weight;
249  float weight = clipWeight * layerWeight; if (weight == 0) continue;
250  GetAnimation(info.clip).Apply(skeleton, 0, AnimationTime(interruptingStateInfo.normalizedTime + interruptingClipTimeAddition, info.clip.length, interruptingStateInfo.speed < 0),
251  interruptingStateInfo.loop, null, weight, MixPose.Current, MixDirection.In);
252  }
253  }
254  } else { // case MixNext || SpineStyle
255  // Apply first non-zero weighted clip
256  int c = 0;
257  for (; c < clipInfoCount; c++) {
258  var info = clipInfo[c]; float weight = info.weight * layerWeight; if (weight == 0) continue;
259  GetAnimation(info.clip).Apply(skeleton, 0, AnimationTime(stateInfo.normalizedTime, info.clip.length, stateInfo.loop, stateInfo.speed < 0), stateInfo.loop, null, 1f, MixPose.Current, MixDirection.In);
260  break;
261  }
262  // Mix the rest
263  for (; c < clipInfoCount; c++) {
264  var info = clipInfo[c]; float weight = info.weight * layerWeight; if (weight == 0) continue;
265  GetAnimation(info.clip).Apply(skeleton, 0, AnimationTime(stateInfo.normalizedTime, info.clip.length, stateInfo.loop, stateInfo.speed < 0), stateInfo.loop, null, weight, MixPose.Current, MixDirection.In);
266  }
267 
268  c = 0;
269  if (hasNext) {
270  // Apply next clip directly instead of mixing (ie: no crossfade, ignores mecanim transition weights)
271  if (mode == MixMode.SpineStyle) {
272  for (; c < nextClipInfoCount; c++) {
273  var info = nextClipInfo[c]; float weight = info.weight * layerWeight; if (weight == 0) continue;
274  GetAnimation(info.clip).Apply(skeleton, 0, AnimationTime(nextStateInfo.normalizedTime , info.clip.length, nextStateInfo.speed < 0), nextStateInfo.loop, null, 1f, MixPose.Current, MixDirection.In);
275  break;
276  }
277  }
278  // Mix the rest
279  for (; c < nextClipInfoCount; c++) {
280  var info = nextClipInfo[c]; float weight = info.weight * layerWeight; if (weight == 0) continue;
281  GetAnimation(info.clip).Apply(skeleton, 0, AnimationTime(nextStateInfo.normalizedTime , info.clip.length, nextStateInfo.speed < 0), nextStateInfo.loop, null, weight, MixPose.Current, MixDirection.In);
282  }
283  }
284 
285  c = 0;
286  if (isInterruptionActive) {
287  // Apply next clip directly instead of mixing (ie: no crossfade, ignores mecanim transition weights)
288  if (mode == MixMode.SpineStyle) {
289  for (; c < interruptingClipInfoCount; c++) {
290  var info = interruptingClipInfo[c]; float clipWeight = shallInterpolateWeightTo1 ? (info.weight + 1.0f) * 0.5f : info.weight;
291  float weight = clipWeight * layerWeight; if (weight == 0) continue;
292  GetAnimation(info.clip).Apply(skeleton, 0, AnimationTime(interruptingStateInfo.normalizedTime + interruptingClipTimeAddition, info.clip.length, interruptingStateInfo.speed < 0), interruptingStateInfo.loop, null, 1f, MixPose.Current, MixDirection.In);
293  break;
294  }
295  }
296  // Mix the rest
297  for (; c < interruptingClipInfoCount; c++) {
298  var info = interruptingClipInfo[c]; float clipWeight = shallInterpolateWeightTo1 ? (info.weight + 1.0f) * 0.5f : info.weight;
299  float weight = clipWeight * layerWeight; if (weight == 0) continue;
300  GetAnimation(info.clip).Apply(skeleton, 0, AnimationTime(interruptingStateInfo.normalizedTime + interruptingClipTimeAddition, info.clip.length, interruptingStateInfo.speed < 0), interruptingStateInfo.loop, null, weight, MixPose.Current, MixDirection.In);
301  }
302  }
303  }

307  static float AnimationTime (float normalizedTime, float clipLength, bool loop, bool reversed) {
308  if (reversed)
309  normalizedTime = (1-normalizedTime + (int)normalizedTime) + (int)normalizedTime;
310  float time = normalizedTime * clipLength;
311  if (loop) return time;
312  const float EndSnapEpsilon = 1f/30f; // Workaround for end-duration keys not being applied.
313  return (clipLength - time < EndSnapEpsilon) ? clipLength : time; // return a time snapped to clipLength;
314  }

319

347  void GetStateUpdatesFromAnimator(int layer) {
348 
349  var layerInfos = layerClipInfos[layer];
350  int clipInfoCount = animator.GetCurrentAnimatorClipInfoCount(layer);
351  int nextClipInfoCount = animator.GetNextAnimatorClipInfoCount(layer);
352 
353  var clipInfos = layerInfos.clipInfos;
354  var nextClipInfos = layerInfos.nextClipInfos;
355  var interruptingClipInfos = layerInfos.interruptingClipInfos;
356 
357  layerInfos.isInterruptionActive = (clipInfoCount == 0 && nextClipInfoCount == 0);
358 
359  // Note: during interruption, GetCurrentAnimatorClipInfoCount and GetNextAnimatorClipInfoCount
360  // are returning 0 in calls above. Therefore we keep previous clipInfos and nextClipInfos
361  // until the interruption is over.
362  if (layerInfos.isInterruptionActive) {
363 
364  // Note: The last frame of a transition interruption
365  // will have fullPathHash set to 0, therefore we have to use previous
366  // frame's infos about interruption clips and correct some values
367  // accordingly (normalizedTime and weight).
368  var interruptingStateInfo = animator.GetNextAnimatorStateInfo(layer);
369  layerInfos.isLastFrameOfInterruption = interruptingStateInfo.fullPathHash == 0;
370  if (!layerInfos.isLastFrameOfInterruption) {
371  layerInfos.interruptingClipInfoCount = interruptingClipInfos.Count;
372 
373  animator.GetNextAnimatorClipInfo(layer, interruptingClipInfos);
374  float oldTime = layerInfos.interruptingStateInfo.normalizedTime;
375  float newTime = interruptingStateInfo.normalizedTime;
376  layerInfos.interruptingClipTimeAddition = newTime - oldTime;
377  layerInfos.interruptingStateInfo = interruptingStateInfo;
378  }
379  }
380  else {
381  layerInfos.clipInfoCount = clipInfoCount;
382  layerInfos.nextClipInfoCount = nextClipInfoCount;
383  layerInfos.interruptingClipInfoCount = 0;
384  layerInfos.isLastFrameOfInterruption = false;
385 
386  if (clipInfos.Capacity < clipInfoCount) clipInfos.Capacity = clipInfoCount;
387  if (nextClipInfos.Capacity < nextClipInfoCount) nextClipInfos.Capacity = nextClipInfoCount;
388 
389  animator.GetCurrentAnimatorClipInfo(layer, clipInfos);
390  animator.GetNextAnimatorClipInfo(layer, nextClipInfos);
391 
392  layerInfos.stateInfo = animator.GetCurrentAnimatorStateInfo(layer);
393  layerInfos.nextStateInfo = animator.GetNextAnimatorStateInfo(layer);
394  }
