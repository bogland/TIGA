517  mesh.uv = uvs;

592  internal static Mesh ExtractWeightedMeshAttachment (string name, MeshAttachment attachment, int slotIndex, SkeletonData skeletonData, List<Transform> boneList, Mesh mesh = null) {
593  if (!attachment.IsWeighted())
594  throw new System.ArgumentException("Mesh is not weighted.", "attachment");
595 
596  Skeleton skeleton = new Skeleton(skeletonData);
597  skeleton.UpdateWorldTransform();
598 
599  float[] floatVerts = new float[attachment.WorldVerticesLength];
600  attachment.ComputeWorldVertices(skeleton.Slots.Items[slotIndex], floatVerts);
601 
602  Vector2[] uvs = ExtractUV(attachment.UVs);
603  Vector3[] verts = ExtractVerts(floatVerts);
604 
605  int[] triangles = attachment.Triangles;
606  Color color = new Color(attachment.R, attachment.G, attachment.B, attachment.A);
607 
608  mesh = mesh ?? new Mesh();
609 
610  mesh.triangles = new int[0];
611 
612  mesh.vertices = verts;
613  mesh.uv = uvs;
614  mesh.triangles = triangles;
615  Color[] colors = new Color[verts.Length];
616  for (int i = 0; i < verts.Length; i++)
617  colors[i] = color;
618 
619  mesh.colors = colors;
620  mesh.name = name;
621  mesh.RecalculateNormals();
622  mesh.RecalculateBounds();
623 
624  // Handle weights and binding
625  var weightTable = new Dictionary<int, BoneWeightContainer>();
626  var warningBuilder = new System.Text.StringBuilder();
627 
628  int[] bones = attachment.Bones;
629  float[] weights = attachment.Vertices;
630  for (int w = 0, v = 0, b = 0, n = bones.Length; v < n; w += 2) {
631 
632  int nn = bones[v++] + v;
633  for (; v < nn; v++, b += 3) {
634  Transform boneTransform = boneList[bones[v]];
635  int vIndex = w / 2;
636  BoneWeightContainer container;
637  if (weightTable.ContainsKey(vIndex))
638  container = weightTable[vIndex];
639  else {
640  container = new BoneWeightContainer();
641  weightTable.Add(vIndex, container);
642  }
643 
644  float weight = weights[b + 2];
645  container.Add(boneTransform, weight);
646  }
647  }
648 
649  BoneWeight[] boneWeights = new BoneWeight[weightTable.Count];
650 
651  for (int i = 0; i < weightTable.Count; i++) {
652  BoneWeight bw = new BoneWeight();
653  var container = weightTable[i];
654 
655  var pairs = container.pairs.OrderByDescending(pair => pair.weight).ToList();
656 
657  for (int b = 0; b < pairs.Count; b++) {
658  if (b > 3) {
659  if (warningBuilder.Length == 0)
660  warningBuilder.Insert(0, "[Weighted Mesh: " + name + "]\r\nUnity only supports 4 weight influences per vertex! The 4 strongest influences will be used.\r\n");
661 
662  warningBuilder.AppendFormat("{0} ignored on vertex {1}!\r\n", pairs[b].bone.name, i);
663  continue;
664  }
665 
666  int boneIndex = boneList.IndexOf(pairs[b].bone);
667  float weight = pairs[b].weight;
668 
669  switch (b) {
670  case 0:
671  bw.boneIndex0 = boneIndex;
672  bw.weight0 = weight;
673  break;
674  case 1:
675  bw.boneIndex1 = boneIndex;
676  bw.weight1 = weight;
677  break;
678  case 2:
679  bw.boneIndex2 = boneIndex;
680  bw.weight2 = weight;
681  break;
682  case 3:
683  bw.boneIndex3 = boneIndex;
684  bw.weight3 = weight;
685  break;
686  }
687  }
688 
689  boneWeights[i] = bw;
690  }
691 
692  Matrix4x4[] bindPoses = new Matrix4x4[boneList.Count];
693  for (int i = 0; i < boneList.Count; i++) {
694  bindPoses[i] = boneList[i].worldToLocalMatrix;
695  }
696 
697  mesh.boneWeights = boneWeights;
698  mesh.bindposes = bindPoses;
699
