220  public static AtlasRegion ToAtlasRegion (this Texture2D t, Shader shader, float scale = DefaultScale, Material materialPropertySource = null) {
221  var material = new Material(shader);
222  if (materialPropertySource != null) {
223  material.CopyPropertiesFromMaterial(materialPropertySource);
224  material.shaderKeywords = materialPropertySource.shaderKeywords;
225  }
226 
227  material.mainTexture = t;
228  var page = material.ToSpineAtlasPage();
229 
230  float width = t.width;
231  float height = t.height;
232 
233  var region = new AtlasRegion();
234  region.name = t.name;
235  region.index = -1;
236  region.rotate = false;
237 
238  // World space units
239  Vector2 boundsMin = Vector2.zero, boundsMax = new Vector2(width, height) * scale;
240 
241  // Texture space/pixel units
242  region.width = (int)width;
243  region.originalWidth = (int)width;
244  region.height = (int)height;
245  region.originalHeight = (int)height;
246  region.offsetX = width * (0.5f - InverseLerp(boundsMin.x, boundsMax.x, 0));
247  region.offsetY = height * (0.5f - InverseLerp(boundsMin.y, boundsMax.y, 0));
248 
249  // Use the full area of the texture.
250  region.u = 0;
251  region.v = 1;
252  region.u2 = 1;
253  region.v2 = 0;
254  region.x = 0;
255  region.y = 0;
256 
257  region.page = page;
258 
259  return region;
260  }

353  internal static AtlasRegion ToAtlasRegion (this Sprite s, bool isolatedTexture = false) {
354  var region = new AtlasRegion();
355  region.name = s.name;
356  region.index = -1;
357  region.rotate = s.packed && s.packingRotation != SpritePackingRotation.None;
358 
359  // World space units
360  Bounds bounds = s.bounds;
361  Vector2 boundsMin = bounds.min, boundsMax = bounds.max;
362 
363  // Texture space/pixel units
364  Rect spineRect = s.rect.SpineUnityFlipRect(s.texture.height);
365  region.width = (int)spineRect.width;
366  region.originalWidth = (int)spineRect.width;
367  region.height = (int)spineRect.height;
368  region.originalHeight = (int)spineRect.height;
369  region.offsetX = spineRect.width * (0.5f - InverseLerp(boundsMin.x, boundsMax.x, 0));
370  region.offsetY = spineRect.height * (0.5f - InverseLerp(boundsMin.y, boundsMax.y, 0));
371 
372  if (isolatedTexture) {
373  region.u = 0;
374  region.v = 1;
375  region.u2 = 1;
376  region.v2 = 0;
377  region.x = 0;
378  region.y = 0;
379  } else {
380  Texture2D tex = s.texture;
381  Rect uvRect = TextureRectToUVRect(s.textureRect, tex.width, tex.height);
382  region.u = uvRect.xMin;
383  region.v = uvRect.yMax;
384  region.u2 = uvRect.xMax;
385  region.v2 = uvRect.yMin;
386  region.x = (int)spineRect.x;
387  region.y = (int)spineRect.y;
388  }
389 
390  return region;
391  }

394  public static void GetRepackedAttachments (List<Attachment> sourceAttachments, List<Attachment> outputAttachments, Material materialPropertySource, out Material outputMaterial, out Texture2D outputTexture, int maxAtlasSize = 1024, int padding = 2, TextureFormat textureFormat = SpineTextureFormat, bool mipmaps = UseMipMaps, string newAssetName = "Repacked Attachments", bool clearCache = false, bool useOriginalNonrenderables = true) {
401  if (sourceAttachments == null) throw new System.ArgumentNullException("sourceAttachments");
402  if (outputAttachments == null) throw new System.ArgumentNullException("outputAttachments");
403 
404  // Use these to detect and use shared regions.
405  var existingRegions = new Dictionary<AtlasRegion, int>();
406  var regionIndexes = new List<int>();
407  var texturesToPack = new List<Texture2D>();
408  var originalRegions = new List<AtlasRegion>();
409 
410  outputAttachments.Clear();
411  outputAttachments.AddRange(sourceAttachments);
412 
413  int newRegionIndex = 0;
414  for (int i = 0, n = sourceAttachments.Count; i < n; i++) {
415  var originalAttachment = sourceAttachments[i];
416 
417  if (IsRenderable(originalAttachment)) {
418  var newAttachment = originalAttachment.GetClone(true);
419  var region = newAttachment.GetRegion();
420  int existingIndex;
421  if (existingRegions.TryGetValue(region, out existingIndex)) {
422  regionIndexes.Add(existingIndex); // Store the region index for the eventual new attachment.
423  } else {
424  originalRegions.Add(region);
425  texturesToPack.Add(region.ToTexture()); // Add the texture to the PackTextures argument
426  existingRegions.Add(region, newRegionIndex); // Add the region to the dictionary of known regions
427  regionIndexes.Add(newRegionIndex); // Store the region index for the eventual new attachment.
428  newRegionIndex++;
429  }
430 
431  outputAttachments[i] = newAttachment;
432  } else {
433  outputAttachments[i] = useOriginalNonrenderables ? originalAttachment : originalAttachment.GetClone(true);
434  regionIndexes.Add(NonrenderingRegion); // Output attachments pairs with regionIndexes list 1:1. Pad with a sentinel if the attachment doesn't have a region.
435  }
436  }
437 
438  // Fill a new texture with the collected attachment textures.
439  var newTexture = new Texture2D(maxAtlasSize, maxAtlasSize, textureFormat, mipmaps);
440  newTexture.mipMapBias = AtlasUtilities.DefaultMipmapBias;
441  newTexture.anisoLevel = texturesToPack[0].anisoLevel;
442  newTexture.name = newAssetName;
443  var rects = newTexture.PackTextures(texturesToPack.ToArray(), padding, maxAtlasSize);
444 
445  // Rehydrate the repacked textures as a Material, Spine atlas and Spine.AtlasAttachments
446  Shader shader = materialPropertySource == null ? Shader.Find("Spine/Skeleton") : materialPropertySource.shader;
447  var newMaterial = new Material(shader);
448  if (materialPropertySource != null) {
449  newMaterial.CopyPropertiesFromMaterial(materialPropertySource);
450  newMaterial.shaderKeywords = materialPropertySource.shaderKeywords;
451  }
452 
453  newMaterial.name = newAssetName;
454  newMaterial.mainTexture = newTexture;
455  var page = newMaterial.ToSpineAtlasPage();
456  page.name = newAssetName;
457 
458  var repackedRegions = new List<AtlasRegion>();
459  for (int i = 0, n = originalRegions.Count; i < n; i++) {
460  var oldRegion = originalRegions[i];
461  var newRegion = UVRectToAtlasRegion(rects[i], oldRegion.name, page, oldRegion.offsetX, oldRegion.offsetY, oldRegion.rotate);
462  repackedRegions.Add(newRegion);
463  }
464 
465  // Map the cloned attachments to the repacked atlas.
466  for (int i = 0, n = outputAttachments.Count; i < n; i++) {
467  var a = outputAttachments[i];
468  if (IsRenderable(a))
469  a.SetRegion(repackedRegions[regionIndexes[i]]);
470  }
471 
472  // Clean up.
473  if (clearCache)
474  AtlasUtilities.ClearCache();
475 
476  outputTexture = newTexture;
477  outputMaterial = newMaterial;
478  }
