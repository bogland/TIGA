36  // SkeletonUtilityKinematicShadow allows hinge chains to inherit a velocity interpreted from changes in parent transform position or from unrelated rigidbodies.
37  // Note: Uncheck "useRootTransformIfNull
38  public class SkeletonUtilityKinematicShadow : MonoBehaviour {
39  #region Inspector
40  [Tooltip("If checked, the hinge chain can inherit your root transform's velocity or position/rotation changes.")]
41  public bool detachedShadow = false;
42  public Transform parent;
43  public bool hideShadow = true;
44  public PhysicsSystem physicsSystem = PhysicsSystem.Physics3D;
45  #endregion

47  GameObject shadowRoot;
48  readonly List<TransformPair> shadowTable = new List<TransformPair>();
49  struct TransformPair {
50  public Transform dest, src;
51  }
52 
53  public enum PhysicsSystem {
54  Physics2D,
55  Physics3D
56  };

58  void Start () {
59  // Duplicate this gameObject as the "shadow" with a different parent.
60  shadowRoot = Instantiate<GameObject>(this.gameObject);
61  Destroy(shadowRoot.GetComponent<SkeletonUtilityKinematicShadow>());
62 
63  // Prepare shadow gameObject's properties.
64  var shadowRootTransform = shadowRoot.transform;
65  shadowRootTransform.position = transform.position;
66  shadowRootTransform.rotation = transform.rotation;
67 
68  Vector3 scaleRef = transform.TransformPoint(Vector3.right);
69  float scale = Vector3.Distance(transform.position, scaleRef);
70  shadowRootTransform.localScale = Vector3.one;
71 
72  if (!detachedShadow) {
73  // Do not change to null coalescing operator (??). Unity overloads null checks for UnityEngine.Objects but not the ?? operator.
74  if (parent == null)
75  shadowRootTransform.parent = transform.root;
76  else
77  shadowRootTransform.parent = parent;
78  }
79 
80  if (hideShadow)
81  shadowRoot.hideFlags = HideFlags.HideInHierarchy;
82 
83  var shadowJoints = shadowRoot.GetComponentsInChildren<Joint>();
84  foreach (Joint j in shadowJoints)
85  j.connectedAnchor *= scale;
86 
87  // Build list of bone pairs (matches shadow transforms with bone transforms)
88  var bones = GetComponentsInChildren<SkeletonUtilityBone>();
89  var shadowBones = shadowRoot.GetComponentsInChildren<SkeletonUtilityBone>();
90  foreach (var b in bones) {
91  if (b.gameObject == this.gameObject)
92  continue;
93 
94  System.Type checkType = (physicsSystem == PhysicsSystem.Physics2D) ? typeof(Rigidbody2D) : typeof(Rigidbody);
95  foreach (var sb in shadowBones) {
96  if (sb.GetComponent(checkType) != null && sb.boneName == b.boneName) {
97  shadowTable.Add(new TransformPair {
98  dest = b.transform,
99  src = sb.transform
100  });
101  break;
102  }
103  }
104 
105  }
106 
107  // Destroy conflicting and unneeded components
108  DestroyComponents(shadowBones);
109 
110  DestroyComponents(GetComponentsInChildren<Joint>());
111  DestroyComponents(GetComponentsInChildren<Rigidbody>());
112  DestroyComponents(GetComponentsInChildren<Collider>());
113  }
114 
115  static void DestroyComponents (Component[] components) {
116  for (int i = 0, n = components.Length; i < n; i++)
117  Destroy(components[i]);
118  }
119 
120  void FixedUpdate () {
121  if (physicsSystem == PhysicsSystem.Physics2D) {
122  var shadowRootRigidbody = shadowRoot.GetComponent<Rigidbody2D>();
123  shadowRootRigidbody.MovePosition(transform.position);
124  shadowRootRigidbody.MoveRotation(transform.rotation.eulerAngles.z);
125  } else {
126  var shadowRootRigidbody = shadowRoot.GetComponent<Rigidbody>();
127  shadowRootRigidbody.MovePosition(transform.position);
128  shadowRootRigidbody.MoveRotation(transform.rotation);
129  }
130 
131  for (int i = 0, n = shadowTable.Count; i < n; i++) {
132  var pair = shadowTable[i];
133  pair.dest.localPosition = pair.src.localPosition;
134  pair.dest.localRotation = pair.src.localRotation;
135  }
136  }
137  }
138 }
