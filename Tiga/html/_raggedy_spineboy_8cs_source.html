54  }

66  void Launch () {
67  RemoveRigidbody();
68  ragdoll.Apply();
69  ragdoll.RootRigidbody.velocity = new Vector2(Random.Range(-launchVelocity.x, launchVelocity.x), launchVelocity.y);
70  StartCoroutine(WaitUntilStopped());
71  }
72 
73  IEnumerator Restore () {
74  Vector3 estimatedPos = ragdoll.EstimatedSkeletonPosition;
75  Vector3 rbPosition = ragdoll.RootRigidbody.position;
76 
77  Vector3 skeletonPoint = estimatedPos;
78  RaycastHit2D hit = Physics2D.Raycast((Vector2)rbPosition, (Vector2)(estimatedPos - rbPosition), Vector3.Distance(estimatedPos, rbPosition), groundMask);
79  if (hit.collider != null)
80  skeletonPoint = hit.point;
81 
82  ragdoll.RootRigidbody.isKinematic = true;
83  ragdoll.SetSkeletonPosition(skeletonPoint);
84 
85  yield return ragdoll.SmoothMix(0, restoreDuration);
86  ragdoll.Remove();
87 
88  AddRigidbody();
89  }
90 
91  IEnumerator WaitUntilStopped () {
92  yield return new WaitForSeconds(0.5f);
93 
94  float t = 0;
95  while (t < 0.5f) {
96  t = (ragdoll.RootRigidbody.velocity.magnitude > 0.09f) ? 0 : t + Time.deltaTime;
97  yield return null;
98  }
99 
100  StartCoroutine(Restore());
101  }
102  }
