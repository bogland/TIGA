53  void OnEnable () {
54  skeletonUtility = (SkeletonUtility)target;
55  skeletonRenderer = skeletonUtility.GetComponent<SkeletonRenderer>();
56  skeleton = skeletonRenderer.Skeleton;
57 
58  if (skeleton == null) {
59  skeletonRenderer.Initialize(false);
60  skeletonRenderer.LateUpdate();
61  skeleton = skeletonRenderer.skeleton;
62  }
63 
64  if (!skeletonRenderer.valid) return;
65 
66  isPrefab |= PrefabUtility.GetPrefabType(this.target) == PrefabType.Prefab;
67  }

69  public override void OnInspectorGUI () {
70  if (isPrefab) {
71  GUILayout.Label(new GUIContent("Cannot edit Prefabs", Icons.warning));
72  return;

74

77  return;

79 
80  skeletonUtility.boneRoot = (Transform)EditorGUILayout.ObjectField("Bone Root", skeletonUtility.boneRoot, typeof(Transform), true);
81 
82  bool hasRootBone = skeletonUtility.boneRoot != null;
83  using (new EditorGUI.DisabledGroupScope(hasRootBone)) {
84  if (SpineInspectorUtility.LargeCenteredButton(SpawnHierarchyButtonLabel))
85  SpawnHierarchyContextMenu();
86  }
87 
88  if (hasRootBone) {
89  if (SpineInspectorUtility.CenteredButton(new GUIContent("Remove Hierarchy"))) {
90  Undo.RegisterCompleteObjectUndo(skeletonUtility, "Remove Hierarchy");
91  Undo.DestroyObjectImmediate(skeletonUtility.boneRoot.gameObject);
92  skeletonUtility.boneRoot = null;
93  }

109  public static void AttachIcon (SkeletonUtilityBone utilityBone) {
110  Skeleton skeleton = utilityBone.skeletonUtility.skeletonRenderer.skeleton;
111  Texture2D icon = utilityBone.bone.Data.Length == 0 ? Icons.nullBone : Icons.boneNib;
112 
113  foreach (IkConstraint c in skeleton.IkConstraints)
114  if (c.Target == utilityBone.bone) {
115  icon = Icons.constraintNib;
116  break;
117  }
118 
119  typeof(EditorGUIUtility).InvokeMember("SetIconForObject", BindingFlags.InvokeMethod | BindingFlags.Static | BindingFlags.NonPublic, null, null, new object[2] {
120  utilityBone.gameObject,
121  icon
122  });
123  }
124 
125  static void AttachIconsToChildren (Transform root) {
