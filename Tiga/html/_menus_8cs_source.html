48  static void CreateAsset<T> (String name) where T : ScriptableObject {
49  var dir = "Assets/";
50  var selected = Selection.activeObject;
51  if (selected != null) {
52  var assetDir = AssetDatabase.GetAssetPath(selected.GetInstanceID());
53  if (assetDir.Length > 0 && Directory.Exists(assetDir))
54  dir = assetDir + "/";
55  }
56  ScriptableObject asset = ScriptableObject.CreateInstance<T>();
57  AssetDatabase.CreateAsset(asset, dir + name + ".asset");
58  AssetDatabase.SaveAssets();
59  EditorUtility.FocusProjectWindow();
60  Selection.activeObject = asset;
61  }

73  static void CreateSpineGameObject<T> (string name) where T : MonoBehaviour {
74  var parentGameObject = Selection.activeObject as GameObject;
75  var parentTransform = parentGameObject == null ? null : parentGameObject.transform;
76 
77  var gameObject = new GameObject(name, typeof(T));
78  gameObject.transform.SetParent(parentTransform, false);
79  EditorUtility.FocusProjectWindow();
80  Selection.activeObject = gameObject;
81  EditorGUIUtility.PingObject(Selection.activeObject);
82  }
