181  public virtual void Initialize (bool overwrite) {
182  if (valid && !overwrite)
183  return;
184 
185  // Clear
186  {
187  if (meshFilter != null)
188  meshFilter.sharedMesh = null;
189 
190  meshRenderer = GetComponent<MeshRenderer>();
191  if (meshRenderer != null) meshRenderer.sharedMaterial = null;
192 
193  currentInstructions.Clear();
194  rendererBuffers.Clear();
195  meshGenerator.Begin();
196  skeleton = null;
197  valid = false;
198  }
199 
200  if (!skeletonDataAsset) {
201  if (logErrors) Debug.LogError("Missing SkeletonData asset.", this);
202  return;
203  }
204 
205  SkeletonData skeletonData = skeletonDataAsset.GetSkeletonData(false);
206  if (skeletonData == null) return;
207  valid = true;
208 
209  meshFilter = GetComponent<MeshFilter>();
210  meshRenderer = GetComponent<MeshRenderer>();
211  rendererBuffers.Initialize();
212 
213  skeleton = new Skeleton(skeletonData) {
214  flipX = initialFlipX,
215  flipY = initialFlipY
216  };
217 
218  if (!string.IsNullOrEmpty(initialSkinName) && !string.Equals(initialSkinName, "default", System.StringComparison.Ordinal))
219  skeleton.SetSkin(initialSkinName);
220 
221  separatorSlots.Clear();
222  for (int i = 0; i < separatorSlotNames.Length; i++)
223  separatorSlots.Add(skeleton.FindSlot(separatorSlotNames[i]));
224 
225  LateUpdate(); // Generate mesh for the first frame it exists.
226 
227  if (OnRebuild != null)
228  OnRebuild(this);
229  }

233  public virtual void LateUpdate () {
234  if (!valid) return;
235 
236  #if SPINE_OPTIONAL_RENDEROVERRIDE
237  bool doMeshOverride = generateMeshOverride != null;
238  if ((!meshRenderer.enabled) && !doMeshOverride) return;
239  #else
240  const bool doMeshOverride = false;
241  if (!meshRenderer.enabled) return;
242  #endif
243  var currentInstructions = this.currentInstructions;
244  var workingSubmeshInstructions = currentInstructions.submeshInstructions;
245  var currentSmartMesh = rendererBuffers.GetNextMesh(); // Double-buffer for performance.
246 
247  bool updateTriangles;
248 
249  if (this.singleSubmesh) {
250  // STEP 1. Determine a SmartMesh.Instruction. Split up instructions into submeshes. =============================================
251  MeshGenerator.GenerateSingleSubmeshInstruction(currentInstructions, skeleton, skeletonDataAsset.atlasAssets[0].materials[0]);
252 
253  // STEP 1.9. Post-process workingInstructions. ==================================================================================
254  #if SPINE_OPTIONAL_MATERIALOVERRIDE
255  if (customMaterialOverride.Count > 0) // isCustomMaterialOverridePopulated
256  MeshGenerator.TryReplaceMaterials(workingSubmeshInstructions, customMaterialOverride);
257  #endif
258 
259  // STEP 2. Update vertex buffer based on verts from the attachments. ===========================================================
260  meshGenerator.settings = new MeshGenerator.Settings {
261  pmaVertexColors = this.pmaVertexColors,
262  zSpacing = this.zSpacing,
263  useClipping = this.useClipping,
264  tintBlack = this.tintBlack,
265  calculateTangents = this.calculateTangents,
266  addNormals = this.addNormals
267  };
268  meshGenerator.Begin();
269  updateTriangles = SkeletonRendererInstruction.GeometryNotEqual(currentInstructions, currentSmartMesh.instructionUsed);
270  if (currentInstructions.hasActiveClipping) {
271  meshGenerator.AddSubmesh(workingSubmeshInstructions.Items[0], updateTriangles);
272  } else {
273  meshGenerator.BuildMeshWithArrays(currentInstructions, updateTriangles);
274  }
275 
276  } else {
277  // STEP 1. Determine a SmartMesh.Instruction. Split up instructions into submeshes. =============================================
278  MeshGenerator.GenerateSkeletonRendererInstruction(currentInstructions, skeleton, customSlotMaterials, separatorSlots, doMeshOverride, this.immutableTriangles);
279 
280  // STEP 1.9. Post-process workingInstructions. ==================================================================================
281  #if SPINE_OPTIONAL_MATERIALOVERRIDE
282  if (customMaterialOverride.Count > 0) // isCustomMaterialOverridePopulated
283  MeshGenerator.TryReplaceMaterials(workingSubmeshInstructions, customMaterialOverride);
284  #endif
285 
286  #if SPINE_OPTIONAL_RENDEROVERRIDE
287  if (doMeshOverride) {
288  this.generateMeshOverride(currentInstructions);
289  if (disableRenderingOnOverride) return;
290  }
291  #endif
292 
293  updateTriangles = SkeletonRendererInstruction.GeometryNotEqual(currentInstructions, currentSmartMesh.instructionUsed);
294 
295  // STEP 2. Update vertex buffer based on verts from the attachments. ===========================================================
296  meshGenerator.settings = new MeshGenerator.Settings {
297  pmaVertexColors = this.pmaVertexColors,
298  zSpacing = this.zSpacing,
299  useClipping = this.useClipping,
300  tintBlack = this.tintBlack,
301  calculateTangents = this.calculateTangents,
302  addNormals = this.addNormals
303  };
304  meshGenerator.Begin();
305  if (currentInstructions.hasActiveClipping)
306  meshGenerator.BuildMesh(currentInstructions, updateTriangles);
307  else
308  meshGenerator.BuildMeshWithArrays(currentInstructions, updateTriangles);
309  }
310 
311  if (OnPostProcessVertices != null) OnPostProcessVertices.Invoke(this.meshGenerator.Buffers);
312 
313  // STEP 3. Move the mesh data into a UnityEngine.Mesh ===========================================================================
314  var currentMesh = currentSmartMesh.mesh;
315  meshGenerator.FillVertexData(currentMesh);
316  rendererBuffers.UpdateSharedMaterials(workingSubmeshInstructions);
317  if (updateTriangles) { // Check if the triangles should also be updated.
318  meshGenerator.FillTriangles(currentMesh);
319  meshRenderer.sharedMaterials = rendererBuffers.GetUpdatedSharedMaterialsArray();
320  } else if (rendererBuffers.MaterialsChangedInLastUpdate()) {
321  meshRenderer.sharedMaterials = rendererBuffers.GetUpdatedSharedMaterialsArray();
322  }
323 
324  meshGenerator.FillLateVertexData(currentMesh);
325 
326  // STEP 4. The UnityEngine.Mesh is ready. Set it as the MeshFilter's mesh. Store the instructions used for that mesh. ===========
327  meshFilter.sharedMesh = currentMesh;
328  currentSmartMesh.instructionUsed.Set(currentInstructions);
329  }
330  }
