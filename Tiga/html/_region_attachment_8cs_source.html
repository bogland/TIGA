79  public void UpdateOffset () {
80  float width = this.width;
81  float height = this.height;
82  float localX2 = width * 0.5f;
83  float localY2 = height * 0.5f;
84  float localX = -localX2;
85  float localY = -localY2;
86  if (regionOriginalWidth != 0) { // if (region != null)
87  localX += regionOffsetX / regionOriginalWidth * width;
88  localY += regionOffsetY / regionOriginalHeight * height;
89  localX2 -= (regionOriginalWidth - regionOffsetX - regionWidth) / regionOriginalWidth * width;
90  localY2 -= (regionOriginalHeight - regionOffsetY - regionHeight) / regionOriginalHeight * height;
91  }
92  float scaleX = this.scaleX;
93  float scaleY = this.scaleY;
94  localX *= scaleX;
95  localY *= scaleY;
96  localX2 *= scaleX;
97  localY2 *= scaleY;
98  float rotation = this.rotation;
99  float cos = MathUtils.CosDeg(rotation);
100  float sin = MathUtils.SinDeg(rotation);
101  float x = this.x;
102  float y = this.y;
103  float localXCos = localX * cos + x;
104  float localXSin = localX * sin;
105  float localYCos = localY * cos + y;
106  float localYSin = localY * sin;
107  float localX2Cos = localX2 * cos + x;
108  float localX2Sin = localX2 * sin;
109  float localY2Cos = localY2 * cos + y;
110  float localY2Sin = localY2 * sin;
111  float[] offset = this.offset;
112  offset[BLX] = localXCos - localYSin;
113  offset[BLY] = localYCos + localXSin;
114  offset[ULX] = localXCos - localY2Sin;
115  offset[ULY] = localY2Cos + localXSin;
116  offset[URX] = localX2Cos - localY2Sin;
117  offset[URY] = localY2Cos + localX2Sin;
118  offset[BRX] = localX2Cos - localYSin;
119  offset[BRY] = localYCos + localX2Sin;
120  }

151  public void ComputeWorldVertices (Bone bone, float[] worldVertices, int offset, int stride = 2) {
152  float[] vertexOffset = this.offset;
153  float bwx = bone.worldX, bwy = bone.worldY;
154  float a = bone.a, b = bone.b, c = bone.c, d = bone.d;
155  float offsetX, offsetY;
156 
157  // Vertex order is different from RegionAttachment.java
158  offsetX = vertexOffset[BRX]; // 0
159  offsetY = vertexOffset[BRY]; // 1
160  worldVertices[offset] = offsetX * a + offsetY * b + bwx; // bl
161  worldVertices[offset + 1] = offsetX * c + offsetY * d + bwy;
162  offset += stride;
163 
164  offsetX = vertexOffset[BLX]; // 2
165  offsetY = vertexOffset[BLY]; // 3
