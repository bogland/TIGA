51  void Start () {
52  skeletonComponent = skeletonComponent ?? GetComponent<ISkeletonComponent>();
53  if (skeletonComponent == null) return;
54  animationStateComponent = animationStateComponent ?? skeletonComponent as IAnimationStateComponent;
55  if (animationStateComponent == null) return;
56  var skeleton = skeletonComponent.Skeleton;
57  if (skeleton == null) return;
58 
59 
60  var skeletonData = skeleton.Data;
61  var state = animationStateComponent.AnimationState;
62  foreach (var ep in events) {
63  var eventData = skeletonData.FindEvent(ep.spineEvent);
64  ep.eventDelegate = ep.eventDelegate ?? delegate(TrackEntry trackEntry, Event e) { if (e.Data == eventData) ep.unityHandler.Invoke(); };
65  state.Event += ep.eventDelegate;
66  }
67  }
68 
69  void OnDestroy () {
70  animationStateComponent = animationStateComponent ?? GetComponent<IAnimationStateComponent>();
71  if (animationStateComponent == null) return;
72 
73  var state = animationStateComponent.AnimationState;
74  foreach (var ep in events) {
75  if (ep.eventDelegate != null) state.Event -= ep.eventDelegate;
76  ep.eventDelegate = null;
77  }
78  }
