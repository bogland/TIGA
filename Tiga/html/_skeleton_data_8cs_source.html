94  public BoneData FindBone (string boneName) {
95  if (boneName == null) throw new ArgumentNullException("boneName", "boneName cannot be null.");
96  var bones = this.bones;
97  var bonesItems = bones.Items;
98  for (int i = 0, n = bones.Count; i < n; i++) {
99  BoneData bone = bonesItems[i];
100  if (bone.name == boneName) return bone;
101  }
102  return null;
103  }
