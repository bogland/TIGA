230  void CreateAnimationReferenceAssets () {
231  const string AssetFolderName = "ReferenceAssets";
232  string parentFolder = System.IO.Path.GetDirectoryName(AssetDatabase.GetAssetPath(targetSkeletonDataAsset));
233  string dataPath = parentFolder + "/" + AssetFolderName;
234  if (!AssetDatabase.IsValidFolder(dataPath)) {
235  AssetDatabase.CreateFolder(parentFolder, AssetFolderName);
236  }
237 
238  FieldInfo nameField = typeof(AnimationReferenceAsset).GetField("animationName", BindingFlags.NonPublic | BindingFlags.Instance);
239  FieldInfo skeletonDataAssetField = typeof(AnimationReferenceAsset).GetField("skeletonDataAsset", BindingFlags.NonPublic | BindingFlags.Instance);
240  foreach (var animation in targetSkeletonData.Animations) {
241  string assetPath = string.Format("{0}/{1}.asset", dataPath, SpineEditorUtilities.GetPathSafeName(animation.Name));
242  AnimationReferenceAsset existingAsset = AssetDatabase.LoadAssetAtPath<AnimationReferenceAsset>(assetPath);
243  if (existingAsset == null) {
244  AnimationReferenceAsset newAsset = ScriptableObject.CreateInstance<AnimationReferenceAsset>();
245  skeletonDataAssetField.SetValue(newAsset, targetSkeletonDataAsset);
246  nameField.SetValue(newAsset, animation.Name);
247  AssetDatabase.CreateAsset(newAsset, assetPath);
248  }
249  }
250 
251  var folderObject = AssetDatabase.LoadAssetAtPath(dataPath, typeof(UnityEngine.Object));
252  if (folderObject != null) {
253  Selection.activeObject = folderObject;
254  EditorGUIUtility.PingObject(folderObject);
255  }

534  void PopulateWarnings () {
535  warnings.Clear();
536 
537  if (skeletonJSON.objectReferenceValue == null) {
538  warnings.Add("Missing Skeleton JSON");
539  } else {
540  var fieldValue = (TextAsset)skeletonJSON.objectReferenceValue;
541  if (!SpineEditorUtilities.SkeletonDataFileValidator.IsSpineData(fieldValue)) {
542  warnings.Add("Skeleton data file is not a valid JSON or binary file.");
543  } else {
544  #if SPINE_TK2D
545  bool searchForSpineAtlasAssets = true;
546  bool isSpriteCollectionNull = spriteCollection.objectReferenceValue == null;
547  if (!isSpriteCollectionNull) searchForSpineAtlasAssets = false;
548  #else
549  // Analysis disable once ConvertToConstant.Local
550  bool searchForSpineAtlasAssets = true;
551  #endif
552 
553  if (searchForSpineAtlasAssets) {
554  bool detectedNullAtlasEntry = false;
555  var atlasList = new List<Atlas>();
556  var actualAtlasAssets = targetSkeletonDataAsset.atlasAssets;
557 
558  for (int i = 0; i < actualAtlasAssets.Length; i++) {
559  if (targetSkeletonDataAsset.atlasAssets[i] == null) {
560  detectedNullAtlasEntry = true;
561  break;
562  } else {
563  atlasList.Add(actualAtlasAssets[i].GetAtlas());
564  }
565  }
566 
567  if (detectedNullAtlasEntry) {
568  warnings.Add("AtlasAsset elements should not be null.");
569  } else {
570  List<string> missingPaths = null;
571  if (atlasAssets.arraySize > 0) {
572  missingPaths = SpineEditorUtilities.GetRequiredAtlasRegions(AssetDatabase.GetAssetPath(skeletonJSON.objectReferenceValue));
573 
574  foreach (var atlas in atlasList) {
575  for (int i = 0; i < missingPaths.Count; i++) {
576  if (atlas.FindRegion(missingPaths[i]) != null) {
577  missingPaths.RemoveAt(i);
578  i--;
579  }

970  currentTrack.TimeScale = (currentTrack.TimeScale == 0) ? 1f : 0f; // pause/play
971  } else {
972  currentTrack.TimeScale = 1f;

974  }

979  currentAnimationEventTimes.Clear();

984  currentAnimationEvents.Add(eventTimeline.Events[i]);
985  currentAnimationEventTimes.Add(eventTimeline.Frames[i]);
986  }
987  }
988  }
