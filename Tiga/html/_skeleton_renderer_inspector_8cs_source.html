115  public static void ReapplySeparatorSlotNames (SkeletonRenderer skeletonRenderer) {
116  if (!skeletonRenderer.valid) return;
117 
118  var separatorSlots = skeletonRenderer.separatorSlots;
119  var separatorSlotNames = skeletonRenderer.separatorSlotNames;
120  var skeleton = skeletonRenderer.skeleton;
121 
122  separatorSlots.Clear();
123  for (int i = 0, n = separatorSlotNames.Length; i < n; i++) {
124  var slot = skeleton.FindSlot(separatorSlotNames[i]);
125  if (slot != null) {
126  separatorSlots.Add(slot);
127  } else {
128  Debug.LogWarning(separatorSlotNames[i] + " is not a slot in " + skeletonRenderer.skeletonDataAsset.skeletonJSON.name);
129  }
130  }

336  public static void SeparatorsField (SerializedProperty separatorSlotNames) {
337  bool multi = separatorSlotNames.serializedObject.isEditingMultipleObjects;
338  bool hasTerminalSlot = false;
339  if (!multi) {
340  var sr = separatorSlotNames.serializedObject.targetObject as ISkeletonComponent;
341  var skeleton = sr.Skeleton;
342  int lastSlot = skeleton.Slots.Count - 1;
343  if (skeleton != null) {
344  for (int i = 0, n = separatorSlotNames.arraySize; i < n; i++) {
345  int index = skeleton.FindSlotIndex(separatorSlotNames.GetArrayElementAtIndex(i).stringValue);
346  if (index == 0 || index == lastSlot) {
347  hasTerminalSlot = true;
348  break;
349  }
350  }
351  }
352  }
353 
354  string terminalSlotWarning = hasTerminalSlot ? " (!)" : "";
355 
356  using (new EditorGUILayout.VerticalScope(EditorStyles.helpBox)) {
357  const string SeparatorsDescription = "Stored names of slots where the Skeleton's render will be split into different batches. This is used by separate components that split the render into different MeshRenderers or GameObjects.";
358  if (separatorSlotNames.isExpanded) {
359  EditorGUILayout.PropertyField(separatorSlotNames, SpineInspectorUtility.TempContent(separatorSlotNames.displayName + terminalSlotWarning, Icons.slotRoot, SeparatorsDescription), true);
360  GUILayout.BeginHorizontal();
361  GUILayout.FlexibleSpace();
362  if (GUILayout.Button("+", GUILayout.MaxWidth(28f), GUILayout.MaxHeight(15f))) {
363  separatorSlotNames.arraySize++;
364  }
365  GUILayout.EndHorizontal();
366 
367  EditorGUILayout.Space();
368  } else
369  EditorGUILayout.PropertyField(separatorSlotNames, new GUIContent(separatorSlotNames.displayName + string.Format("{0} [{1}]", terminalSlotWarning, separatorSlotNames.arraySize), SeparatorsDescription), true);
370  }
371  }
