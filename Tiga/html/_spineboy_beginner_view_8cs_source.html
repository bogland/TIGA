86  void PlayNewStableAnimation () {
87  var newModelState = model.state;
88  Animation nextAnimation;
89 
90  // Add conditionals to not interrupt transient animations.
91 
92  if (previousViewState == SpineBeginnerBodyState.Jumping && newModelState != SpineBeginnerBodyState.Jumping) {
93  PlayFootstepSound();
94  }
95 
96  if (newModelState == SpineBeginnerBodyState.Jumping) {
97  jumpSource.Play();
98  nextAnimation = jump;
99  } else {
100  if (newModelState == SpineBeginnerBodyState.Running) {
101  nextAnimation = run;
102  } else {
103  nextAnimation = idle;
104  }
105  }
106 
107  skeletonAnimation.AnimationState.SetAnimation(0, nextAnimation, true);
108  }
