123  public void Apply () {
124  isActive = true;
125  mix = 1;
126 
127  StartingBone = skeleton.FindBone(startingBoneName);
128  RecursivelyCreateBoneProxies(StartingBone);
129 
130  RootRigidbody = boneTable[StartingBone].GetComponent<Rigidbody>();
131  RootRigidbody.isKinematic = pinStartBone;
132  RootRigidbody.mass = rootMass;
133  var boneColliders = new List<Collider>();
134  foreach (var pair in boneTable) {
135  var b = pair.Key;
136  var t = pair.Value;
137  Transform parentTransform;
138  boneColliders.Add(t.GetComponent<Collider>());
139  if (b == StartingBone) {
140  ragdollRoot = new GameObject("RagdollRoot").transform;
141  ragdollRoot.SetParent(transform, false);
142  if (b == skeleton.RootBone) { // RagdollRoot is skeleton root.
143  ragdollRoot.localPosition = new Vector3(b.WorldX, b.WorldY, 0);
144  ragdollRoot.localRotation = Quaternion.Euler(0, 0, GetPropagatedRotation(b));
145  } else {
146  ragdollRoot.localPosition = new Vector3(b.Parent.WorldX, b.Parent.WorldY, 0);
147  ragdollRoot.localRotation = Quaternion.Euler(0, 0, GetPropagatedRotation(b.Parent));
148  }
149  parentTransform = ragdollRoot;
150  rootOffset = t.position - transform.position;
151  } else {
152  parentTransform = boneTable[b.Parent];
153  }
154 
155  // Add joint and attach to parent.
156  var rbParent = parentTransform.GetComponent<Rigidbody>();
157  if (rbParent != null) {
158  var joint = t.gameObject.AddComponent<HingeJoint>();
159  joint.connectedBody = rbParent;
160  Vector3 localPos = parentTransform.InverseTransformPoint(t.position);
161  localPos.x *= 1;
162  joint.connectedAnchor = localPos;
163  joint.axis = Vector3.forward;
164 
165  joint.GetComponent<Rigidbody>().mass = joint.connectedBody.mass * massFalloffFactor;
166  joint.limits = new JointLimits {
167  min = -rotationLimit,
168  max = rotationLimit,
169  };
170  joint.useLimits = true;
171  joint.enableCollision = enableJointCollision;
172  }
173  }
174 
175  // Ignore collisions among bones.
176  for (int x = 0; x < boneColliders.Count; x++) {
177  for (int y = 0; y < boneColliders.Count; y++) {
178  if (x == y) continue;
179  Physics.IgnoreCollision(boneColliders[x], boneColliders[y]);
180  }
181  }
182 
183  // Destroy existing override-mode SkeletonUtilityBones.
184  var utilityBones = GetComponentsInChildren<SkeletonUtilityBone>();
185  if (utilityBones.Length > 0) {
186  var destroyedUtilityBoneNames = new List<string>();
187  foreach (var ub in utilityBones) {
188  if (ub.mode == SkeletonUtilityBone.Mode.Override) {
189  destroyedUtilityBoneNames.Add(ub.gameObject.name);
190  Destroy(ub.gameObject);
191  }
192  }
193  if (destroyedUtilityBoneNames.Count > 0) {
194  string msg = "Destroyed Utility Bones: ";
195  for (int i = 0; i < destroyedUtilityBoneNames.Count; i++) {
196  msg += destroyedUtilityBoneNames[i];
197  if (i != destroyedUtilityBoneNames.Count - 1) {
198  msg += ",";
199  }
200  }
201  Debug.LogWarning(msg);
202  }
203  }
204 
205  // Disable skeleton constraints.
206  if (disableIK) {
207  var ikConstraints = skeleton.IkConstraints;
208  for (int i = 0, n = ikConstraints.Count; i < n; i++)
209  ikConstraints.Items[i].mix = 0;
210  }
211 
212  if (disableOtherConstraints) {
213  var transformConstraints = skeleton.transformConstraints;
214  for (int i = 0, n = transformConstraints.Count; i < n; i++) {
215  transformConstraints.Items[i].rotateMix = 0;
216  transformConstraints.Items[i].scaleMix = 0;
217  transformConstraints.Items[i].shearMix = 0;
218  transformConstraints.Items[i].translateMix = 0;
219  }
220 
221  var pathConstraints = skeleton.pathConstraints;
222  for (int i = 0, n = pathConstraints.Count; i < n; i++) {
223  pathConstraints.Items[i].rotateMix = 0;
224  pathConstraints.Items[i].translateMix = 0;
225  }
226  }
227 
228  targetSkeletonComponent.UpdateWorld += UpdateSpineSkeleton;
229  }

232  public Coroutine SmoothMix (float target, float duration) {
233  return StartCoroutine(SmoothMixCoroutine(target, duration));

236  IEnumerator SmoothMixCoroutine (float target, float duration) {
237  float startTime = Time.time;
238  float startMix = mix;
239  while (mix > 0) {
240  skeleton.SetBonesToSetupPose();
241  mix = Mathf.SmoothStep(startMix, target, (Time.time - startTime) / duration);
242  yield return null;
243  }
244  }

280  void RecursivelyCreateBoneProxies (Bone b) {
281  string boneName = b.data.name;
282  if (stopBoneNames.Contains(boneName))
283  return;
284 
285  var boneGameObject = new GameObject(boneName);
286  boneGameObject.layer = colliderLayer;
287  Transform t = boneGameObject.transform;
288  boneTable.Add(b, t);
289 
290  t.parent = transform;
291  t.localPosition = new Vector3(b.WorldX, b.WorldY, 0);
292  t.localRotation = Quaternion.Euler(0, 0, b.WorldRotationX - b.shearX);
293  t.localScale = new Vector3(b.WorldScaleX, b.WorldScaleY, 1);
294 
295  // MITCH: You left "todo: proper ragdoll branching"
296  var colliders = AttachBoundingBoxRagdollColliders(b);
297  if (colliders.Count == 0) {
298  float length = b.Data.Length;
299  if (length == 0) {
300  var ball = boneGameObject.AddComponent<SphereCollider>();
301  ball.radius = thickness * 0.5f;
302  } else {
303  var box = boneGameObject.AddComponent<BoxCollider>();
304  box.size = new Vector3(length, thickness, thickness);
305  box.center = new Vector3(length * 0.5f, 0);
306  }
307  }
308  var rb = boneGameObject.AddComponent<Rigidbody>();
309  rb.constraints = RigidbodyConstraints.FreezePositionZ;
310 
311  foreach (Bone child in b.Children)
312  RecursivelyCreateBoneProxies(child);
313  }

315  void UpdateSpineSkeleton (ISkeletonAnimation skeletonRenderer) {
316  bool flipX = skeleton.flipX;
317  bool flipY = skeleton.flipY;
318  bool flipXOR = flipX ^ flipY;
319  bool flipOR = flipX || flipY;
320 
321  foreach (var pair in boneTable) {
322  var b = pair.Key;
323  var t = pair.Value;
324  bool isStartingBone = b == StartingBone;
325  Transform parentTransform = isStartingBone ? ragdollRoot : boneTable[b.Parent];
326  Vector3 parentTransformWorldPosition = parentTransform.position;
327  Quaternion parentTransformWorldRotation = parentTransform.rotation;
328 
329  parentSpaceHelper.position = parentTransformWorldPosition;
330  parentSpaceHelper.rotation = parentTransformWorldRotation;
331  parentSpaceHelper.localScale = parentTransform.localScale;
332 
333  Vector3 boneWorldPosition = t.position;
334  Vector3 right = parentSpaceHelper.InverseTransformDirection(t.right);
335 
336  Vector3 boneLocalPosition = parentSpaceHelper.InverseTransformPoint(boneWorldPosition);
337  float boneLocalRotation = Mathf.Atan2(right.y, right.x) * Mathf.Rad2Deg;
338 
339  if (flipOR) {
340  if (isStartingBone) {
341  if (flipX) boneLocalPosition.x *= -1f;
342  if (flipY) boneLocalPosition.y *= -1f;
343 
344  boneLocalRotation = boneLocalRotation * (flipXOR ? -1f : 1f);
345  if (flipX) boneLocalRotation += 180;
346  } else {
347  if (flipXOR) {
348  boneLocalRotation *= -1f;
349  boneLocalPosition.y *= -1f; // wtf??
350  }
351  }
352  }
353 
354  b.x = Mathf.Lerp(b.x, boneLocalPosition.x, mix);
355  b.y = Mathf.Lerp(b.y, boneLocalPosition.y, mix);
356  b.rotation = Mathf.Lerp(b.rotation, boneLocalRotation, mix);
357  //b.AppliedRotation = Mathf.Lerp(b.AppliedRotation, boneLocalRotation, mix);
358  }
359  }

361  List<Collider> AttachBoundingBoxRagdollColliders (Bone b) {
362  const string AttachmentNameMarker = "ragdoll";
363  var colliders = new List<Collider>();
364 
365  Transform t = boneTable[b];
366  GameObject go = t.gameObject;
367  var skin = skeleton.Skin ?? skeleton.Data.DefaultSkin;
368 
369  var attachments = new List<Attachment>();
370  foreach (Slot s in skeleton.Slots) {
371  if (s.Bone == b) {
372  skin.FindAttachmentsForSlot(skeleton.Slots.IndexOf(s), attachments);
373  foreach (var a in attachments) {
374  var bbAttachment = a as BoundingBoxAttachment;
375  if (bbAttachment != null) {
376  if (!a.Name.ToLower().Contains(AttachmentNameMarker))
377  continue;
378 
379  var bbCollider = go.AddComponent<BoxCollider>();
380  var bounds = SkeletonUtility.GetBoundingBoxBounds(bbAttachment, thickness);
381  bbCollider.center = bounds.center;
382  bbCollider.size = bounds.size;
383  colliders.Add(bbCollider);
384  }
385  }
386  }
387  }
388 
389  return colliders;
390  }

392  static float GetPropagatedRotation (Bone b) {
393  Bone parent = b.Parent;
394  float a = b.AppliedRotation;
395  while (parent != null) {
396  a += parent.AppliedRotation;
397  parent = parent.parent;
398  }
399  return a;
