65  public void Update (Skeleton skeleton, bool updateAabb) {
66  ExposedList<BoundingBoxAttachment> boundingBoxes = BoundingBoxes;
67  ExposedList<Polygon> polygons = Polygons;
68  ExposedList<Slot> slots = skeleton.slots;
69  int slotCount = slots.Count;
70 
71  boundingBoxes.Clear();
72  for (int i = 0, n = polygons.Count; i < n; i++)
73  polygonPool.Add(polygons.Items[i]);
74  polygons.Clear();
75 
76  for (int i = 0; i < slotCount; i++) {
77  Slot slot = slots.Items[i];
78  BoundingBoxAttachment boundingBox = slot.attachment as BoundingBoxAttachment;
79  if (boundingBox == null) continue;
80  boundingBoxes.Add(boundingBox);
81 
82  Polygon polygon = null;
83  int poolCount = polygonPool.Count;
84  if (poolCount > 0) {
85  polygon = polygonPool.Items[poolCount - 1];
86  polygonPool.RemoveAt(poolCount - 1);
87  } else
88  polygon = new Polygon();
89  polygons.Add(polygon);
90 
91  int count = boundingBox.worldVerticesLength;
92  polygon.Count = count;
93  if (polygon.Vertices.Length < count) polygon.Vertices = new float[count];
94  boundingBox.ComputeWorldVertices(slot, polygon.Vertices);
95  }
96 
97  if (updateAabb) {
98  AabbCompute();
99  } else {
100  minX = int.MinValue;
101  minY = int.MinValue;
102  maxX = int.MaxValue;
103  maxY = int.MaxValue;
104  }
105  }

107  private void AabbCompute () {
108  float minX = int.MaxValue, minY = int.MaxValue, maxX = int.MinValue, maxY = int.MinValue;
109  ExposedList<Polygon> polygons = Polygons;
110  for (int i = 0, n = polygons.Count; i < n; i++) {
111  Polygon polygon = polygons.Items[i];
112  float[] vertices = polygon.Vertices;
113  for (int ii = 0, nn = polygon.Count; ii < nn; ii += 2) {
114  float x = vertices[ii];
115  float y = vertices[ii + 1];
116  minX = Math.Min(minX, x);
117  minY = Math.Min(minY, y);
118  maxX = Math.Max(maxX, x);
119  maxY = Math.Max(maxY, y);
120  }
121  }
122  this.minX = minX;
123  this.minY = minY;
124  this.maxX = maxX;
125  this.maxY = maxY;
126  }

135  public bool AabbIntersectsSegment (float x1, float y1, float x2, float y2) {
136  float minX = this.minX;
137  float minY = this.minY;
138  float maxX = this.maxX;
139  float maxY = this.maxY;
140  if ((x1 <= minX && x2 <= minX) || (y1 <= minY && y2 <= minY) || (x1 >= maxX && x2 >= maxX) || (y1 >= maxY && y2 >= maxY))
141  return false;
142  float m = (y2 - y1) / (x2 - x1);
143  float y = m * (minX - x1) + y1;
144  if (y > minY && y < maxY) return true;
145  y = m * (maxX - x1) + y1;
146  if (y > minY && y < maxY) return true;
147  float x = (minY - y1) / m + x1;
148  if (x > minX && x < maxX) return true;
149  x = (maxY - y1) / m + x1;
150  if (x > minX && x < maxX) return true;
151  return false;
152  }

160  public bool ContainsPoint (Polygon polygon, float x, float y) {
161  float[] vertices = polygon.Vertices;
162  int nn = polygon.Count;
163 
164  int prevIndex = nn - 2;
165  bool inside = false;
166  for (int ii = 0; ii < nn; ii += 2) {
167  float vertexY = vertices[ii + 1];
168  float prevY = vertices[prevIndex + 1];
169  if ((vertexY < y && prevY >= y) || (prevY < y && vertexY >= y)) {
170  float vertexX = vertices[ii];
171  if (vertexX + (y - vertexY) / (prevY - vertexY) * (vertices[prevIndex] - vertexX) < x) inside = !inside;
172  }
173  prevIndex = ii;
174  }
175  return inside;
176  }

197  public bool IntersectsSegment (Polygon polygon, float x1, float y1, float x2, float y2) {
198  float[] vertices = polygon.Vertices;
199  int nn = polygon.Count;
200 
201  float width12 = x1 - x2, height12 = y1 - y2;
202  float det1 = x1 * y2 - y1 * x2;
203  float x3 = vertices[nn - 2], y3 = vertices[nn - 1];
204  for (int ii = 0; ii < nn; ii += 2) {
205  float x4 = vertices[ii], y4 = vertices[ii + 1];
206  float det2 = x3 * y4 - y3 * x4;
207  float width34 = x3 - x4, height34 = y3 - y4;
208  float det3 = width12 * height34 - height12 * width34;
209  float x = (det1 * width34 - width12 * det2) / det3;
210  if (((x >= x3 && x <= x4) || (x >= x4 && x <= x3)) && ((x >= x1 && x <= x2) || (x >= x2 && x <= x1))) {
211  float y = (det1 * height34 - height12 * det2) / det3;
212  if (((y >= y3 && y <= y4) || (y >= y4 && y <= y3)) && ((y >= y1 && y <= y2) || (y >= y2 && y <= y1))) return true;
213  }
214  x3 = x4;
215  y3 = y4;
216  }
217  return false;
218  }
