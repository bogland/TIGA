348  static void SceneViewDragAndDrop (SceneView sceneview) {
349  var current = UnityEngine.Event.current;
350  var references = DragAndDrop.objectReferences;
351  if (current.type == EventType.Layout) return;
352 
353  // Allow drag and drop of one SkeletonDataAsset.
354  if (references.Length == 1) {
355  var skeletonDataAsset = references[0] as SkeletonDataAsset;
356  if (skeletonDataAsset != null) {
357  var mousePos = current.mousePosition;
358 
359  bool invalidSkeletonData = skeletonDataAsset.GetSkeletonData(true) == null;
360  if (invalidSkeletonData) {
361  DragAndDrop.visualMode = DragAndDropVisualMode.Rejected;
362  Handles.BeginGUI();
363  GUI.Label(new Rect(mousePos + new Vector2(20f, 20f), new Vector2(400f, 40f)), new GUIContent(string.Format("{0} is invalid.\nCannot create new Spine GameObject.", skeletonDataAsset.name), SpineEditorUtilities.Icons.warning));
364  Handles.EndGUI();
365  return;
366  } else {
367  DragAndDrop.visualMode = DragAndDropVisualMode.Copy;
368  Handles.BeginGUI();
369  GUI.Label(new Rect(mousePos + new Vector2(20f, 20f), new Vector2(400f, 20f)), new GUIContent(string.Format("Create Spine GameObject ({0})", skeletonDataAsset.skeletonJSON.name), SpineEditorUtilities.Icons.skeletonDataAssetIcon));
370  Handles.EndGUI();
371 
372  if (current.type == EventType.DragPerform) {
373  RectTransform rectTransform = (Selection.activeGameObject == null) ? null : Selection.activeGameObject.GetComponent<RectTransform>();
374  Plane plane = (rectTransform == null) ? new Plane(Vector3.back, Vector3.zero) : new Plane(-rectTransform.forward, rectTransform.position);
375  Vector3 spawnPoint = MousePointToWorldPoint2D(mousePos, sceneview.camera, plane);
376  ShowInstantiateContextMenu(skeletonDataAsset, spawnPoint);
377  DragAndDrop.AcceptDrag();
378  current.Use();
379  }
380  }
381  }
382  }

422  public static void HandleSkeletonComponentDrop (object spawnMenuData) {
423  var data = (SpawnMenuData)spawnMenuData;
424 
425  if (data.skeletonDataAsset.GetSkeletonData(true) == null) {
426  EditorUtility.DisplayDialog("Invalid SkeletonDataAsset", "Unable to create Spine GameObject.\n\nPlease check your SkeletonDataAsset.", "Ok");
427  return;
428  }
429 
430  bool isUI = data.isUI;
431 
432  Component newSkeletonComponent = data.instantiateDelegate.Invoke(data.skeletonDataAsset);
433  GameObject newGameObject = newSkeletonComponent.gameObject;
434  Transform newTransform = newGameObject.transform;
435 
436  var activeGameObject = Selection.activeGameObject;
437  if (isUI && activeGameObject != null)
438  newTransform.SetParent(activeGameObject.transform, false);
439 
440  newTransform.position = isUI ? data.spawnPoint : RoundVector(data.spawnPoint, 2);
441 
442  if (isUI && (activeGameObject == null || activeGameObject.GetComponent<RectTransform>() == null))
443  Debug.Log("Created a UI Skeleton GameObject not under a RectTransform. It may not be visible until you parent it to a canvas.");
444 
445  if (!isUI && activeGameObject != null && activeGameObject.transform.localScale != Vector3.one)
446  Debug.Log("New Spine GameObject was parented to a scaled Transform. It may not be the intended size.");
447 
448  Selection.activeGameObject = newGameObject;
449  //EditorGUIUtility.PingObject(newGameObject); // Doesn't work when setting activeGameObject.
450  Undo.RegisterCreatedObjectUndo(newGameObject, "Create Spine GameObject");
451  }

467  var screenPos = new Vector3(mousePosition.x, camera.pixelHeight - mousePosition.y, 0f);

469  float distance;
470  bool hit = plane.Raycast(ray, out distance);
