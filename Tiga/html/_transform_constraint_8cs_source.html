83  void ApplyAbsoluteWorld () {
84  float rotateMix = this.rotateMix, translateMix = this.translateMix, scaleMix = this.scaleMix, shearMix = this.shearMix;
85  Bone target = this.target;
86  float ta = target.a, tb = target.b, tc = target.c, td = target.d;
87  float degRadReflect = ta * td - tb * tc > 0 ? MathUtils.DegRad : -MathUtils.DegRad;
88  float offsetRotation = data.offsetRotation * degRadReflect, offsetShearY = data.offsetShearY * degRadReflect;
89  var bones = this.bones;
90  for (int i = 0, n = bones.Count; i < n; i++) {
91  Bone bone = bones.Items[i];
92  bool modified = false;
93 
94  if (rotateMix != 0) {
95  float a = bone.a, b = bone.b, c = bone.c, d = bone.d;
96  float r = MathUtils.Atan2(tc, ta) - MathUtils.Atan2(c, a) + offsetRotation;
97  if (r > MathUtils.PI)
98  r -= MathUtils.PI2;
99  else if (r < -MathUtils.PI) r += MathUtils.PI2;
100  r *= rotateMix;
101  float cos = MathUtils.Cos(r), sin = MathUtils.Sin(r);
102  bone.a = cos * a - sin * c;
103  bone.b = cos * b - sin * d;
104  bone.c = sin * a + cos * c;
105  bone.d = sin * b + cos * d;
106  modified = true;
107  }
108 
109  if (translateMix != 0) {
110  float tx, ty; //Vector2 temp = this.temp;
111  target.LocalToWorld(data.offsetX, data.offsetY, out tx, out ty); //target.localToWorld(temp.set(data.offsetX, data.offsetY));
112  bone.worldX += (tx - bone.worldX) * translateMix;
113  bone.worldY += (ty - bone.worldY) * translateMix;
114  modified = true;
115  }
116 
117  if (scaleMix > 0) {
118  float s = (float)Math.Sqrt(bone.a * bone.a + bone.c * bone.c);
119  //float ts = (float)Math.sqrt(ta * ta + tc * tc);
120  if (s > 0.00001f) s = (s + ((float)Math.Sqrt(ta * ta + tc * tc) - s + data.offsetScaleX) * scaleMix) / s;
121  bone.a *= s;
122  bone.c *= s;
123  s = (float)Math.Sqrt(bone.b * bone.b + bone.d * bone.d);
124  //ts = (float)Math.Sqrt(tb * tb + td * td);
125  if (s > 0.00001f) s = (s + ((float)Math.Sqrt(tb * tb + td * td) - s + data.offsetScaleY) * scaleMix) / s;
126  bone.b *= s;
127  bone.d *= s;
128  modified = true;
129  }
130 
131  if (shearMix > 0) {
132  float b = bone.b, d = bone.d;
133  float by = MathUtils.Atan2(d, b);
134  float r = MathUtils.Atan2(td, tb) - MathUtils.Atan2(tc, ta) - (by - MathUtils.Atan2(bone.c, bone.a));
135  if (r > MathUtils.PI)
136  r -= MathUtils.PI2;
137  else if (r < -MathUtils.PI) r += MathUtils.PI2;
138  r = by + (r + offsetShearY) * shearMix;
139  float s = (float)Math.Sqrt(b * b + d * d);
140  bone.b = MathUtils.Cos(r) * s;
141  bone.d = MathUtils.Sin(r) * s;
142  modified = true;
143  }
144 
145  if (modified) bone.appliedValid = false;
146  }
147  }

149  void ApplyRelativeWorld () {
150  float rotateMix = this.rotateMix, translateMix = this.translateMix, scaleMix = this.scaleMix, shearMix = this.shearMix;
151  Bone target = this.target;
152  float ta = target.a, tb = target.b, tc = target.c, td = target.d;
153  float degRadReflect = ta * td - tb * tc > 0 ? MathUtils.DegRad : -MathUtils.DegRad;

155  var bones = this.bones;
156  for (int i = 0, n = bones.Count; i < n; i++) {
157  Bone bone = bones.Items[i];
158  bool modified = false;
159 
160  if (rotateMix != 0) {
161  float a = bone.a, b = bone.b, c = bone.c, d = bone.d;
162  float r = MathUtils.Atan2(tc, ta) + offsetRotation;
163  if (r > MathUtils.PI)
164  r -= MathUtils.PI2;
165  else if (r < -MathUtils.PI) r += MathUtils.PI2;

169  bone.b = cos * b - sin * d;

174 
175  if (translateMix != 0) {
176  float tx, ty; //Vector2 temp = this.temp;
177  target.LocalToWorld(data.offsetX, data.offsetY, out tx, out ty); //target.localToWorld(temp.set(data.offsetX, data.offsetY));
178  bone.worldX += tx * translateMix;
179  bone.worldY += ty * translateMix;
180  modified = true;
181  }
182 
183  if (scaleMix > 0) {
184  float s = ((float)Math.Sqrt(ta * ta + tc * tc) - 1 + data.offsetScaleX) * scaleMix + 1;
185  bone.a *= s;
186  bone.c *= s;
187  s = ((float)Math.Sqrt(tb * tb + td * td) - 1 + data.offsetScaleY) * scaleMix + 1;
188  bone.b *= s;
189  bone.d *= s;
190  modified = true;
191  }
192 
193  if (shearMix > 0) {
194  float r = MathUtils.Atan2(td, tb) - MathUtils.Atan2(tc, ta);
195  if (r > MathUtils.PI)
196  r -= MathUtils.PI2;
197  else if (r < -MathUtils.PI) r += MathUtils.PI2;
198  float b = bone.b, d = bone.d;
199  r = MathUtils.Atan2(d, b) + (r - MathUtils.PI / 2 + offsetShearY) * shearMix;
200  float s = (float)Math.Sqrt(b * b + d * d);
201  bone.b = MathUtils.Cos(r) * s;
202  bone.d = MathUtils.Sin(r) * s;
203  modified = true;
204  }
205 
206  if (modified) bone.appliedValid = false;
207  }

210  void ApplyAbsoluteLocal () {
211  float rotateMix = this.rotateMix, translateMix = this.translateMix, scaleMix = this.scaleMix, shearMix = this.shearMix;
212  Bone target = this.target;
213  if (!target.appliedValid) target.UpdateAppliedTransform();
214  var bonesItems = this.bones.Items;
215  for (int i = 0, n = this.bones.Count; i < n; i++) {
216  Bone bone = bonesItems[i];
217  if (!bone.appliedValid) bone.UpdateAppliedTransform();
218 
219  float rotation = bone.arotation;
220  if (rotateMix != 0) {
221  float r = target.arotation - rotation + data.offsetRotation;
222  r -= (16384 - (int)(16384.499999999996 - r / 360)) * 360;
223  rotation += r * rotateMix;
224  }
225 
226  float x = bone.ax, y = bone.ay;
227  if (translateMix != 0) {
228  x += (target.ax - x + data.offsetX) * translateMix;
229  y += (target.ay - y + data.offsetY) * translateMix;
230  }
231 
232  float scaleX = bone.ascaleX, scaleY = bone.ascaleY;
233  if (scaleMix > 0) {
234  if (scaleX > 0.00001f) scaleX = (scaleX + (target.ascaleX - scaleX + data.offsetScaleX) * scaleMix) / scaleX;
235  if (scaleY > 0.00001f) scaleY = (scaleY + (target.ascaleY - scaleY + data.offsetScaleY) * scaleMix) / scaleY;
236  }
237 
238  float shearY = bone.ashearY;
239  if (shearMix > 0) {
240  float r = target.ashearY - shearY + data.offsetShearY;
241  r -= (16384 - (int)(16384.499999999996 - r / 360)) * 360;
242  bone.shearY += r * shearMix;
243  }
244 
245  bone.UpdateWorldTransform(x, y, rotation, scaleX, scaleY, bone.ashearX, shearY);
246  }
247  }

249  void ApplyRelativeLocal () {
250  float rotateMix = this.rotateMix, translateMix = this.translateMix, scaleMix = this.scaleMix, shearMix = this.shearMix;
251  Bone target = this.target;
252  if (!target.appliedValid) target.UpdateAppliedTransform();
253  var bonesItems = this.bones.Items;
254  for (int i = 0, n = this.bones.Count; i < n; i++) {
255  Bone bone = bonesItems[i];
256  if (!bone.appliedValid) bone.UpdateAppliedTransform();
257 
258  float rotation = bone.arotation;
259  if (rotateMix != 0) rotation += (target.arotation + data.offsetRotation) * rotateMix;
260 
261  float x = bone.ax, y = bone.ay;
262  if (translateMix != 0) {
263  x += (target.ax + data.offsetX) * translateMix;
264  y += (target.ay + data.offsetY) * translateMix;
265  }
266 
267  float scaleX = bone.ascaleX, scaleY = bone.ascaleY;
268  if (scaleMix > 0) {
269  if (scaleX > 0.00001f) scaleX *= ((target.ascaleX - 1 + data.offsetScaleX) * scaleMix) + 1;
270  if (scaleY > 0.00001f) scaleY *= ((target.ascaleY - 1 + data.offsetScaleY) * scaleMix) + 1;
271  }
272 
273  float shearY = bone.ashearY;
274  if (shearMix > 0) shearY += (target.ashearY + data.offsetShearY) * shearMix;
275 
276  bone.UpdateWorldTransform(x, y, rotation, scaleX, scaleY, bone.ashearX, shearY);
277  }
