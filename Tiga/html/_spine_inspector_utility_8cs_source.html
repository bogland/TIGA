120  #region SerializedProperty Helpers
121  public static SerializedProperty FindBaseOrSiblingProperty (this SerializedProperty property, string propertyName) {
122  if (string.IsNullOrEmpty(propertyName)) return null;
123 
124  SerializedProperty relativeProperty = property.serializedObject.FindProperty(propertyName); // baseProperty
125 
126  // If base property is not found, look for the sibling property.
127  if (relativeProperty == null) {
128  string propertyPath = property.propertyPath;
129  int localPathLength = property.name.Length;
130 
131  string newPropertyPath = propertyPath.Remove(propertyPath.Length - localPathLength, localPathLength) + propertyName;
132  relativeProperty = property.serializedObject.FindProperty(newPropertyPath);
133 
134  // If a direct sibling property was not found, try to find the sibling of the array.
135  if (relativeProperty == null && property.isArray) {
136  int propertyPathLength = propertyPath.Length;
137 
138  int dotCount = 0;
139  const int SiblingOfListDotCount = 3;
140  for (int i = 1; i < propertyPathLength; i++) {
141  if (propertyPath[propertyPathLength - i] == '.') {
142  dotCount++;
143  if (dotCount >= SiblingOfListDotCount) {
144  localPathLength = i - 1;
145  break;
146  }
147  }
148  }
149 
150  newPropertyPath = propertyPath.Remove(propertyPath.Length - localPathLength, localPathLength) + propertyName;
151  relativeProperty = property.serializedObject.FindProperty(newPropertyPath);
152  }
153  }
154 
155  return relativeProperty;
156  }

280  if (so.isEditingMultipleObjects) {
281  int n = so.targetObjects.Length;
282  var first = so.targetObjects[0] as IHasSkeletonDataAsset;
283  for (int i = 1; i < n; i++) {
284  var sr = so.targetObjects[i] as IHasSkeletonDataAsset;

286  return false;
287  }
288  }

292  public static SerializedObject GetRenderersSerializedObject (SerializedObject serializedObject) {
293  if (serializedObject.isEditingMultipleObjects) {
294  var renderers = new List<Object>();
295  foreach (var o in serializedObject.targetObjects) {
296  var component = o as Component;
297  if (component != null) {
298  var renderer = component.GetComponent<Renderer>();
299  if (renderer != null)
300  renderers.Add(renderer);
301  }
302  }
303  return new SerializedObject(renderers.ToArray());
304  } else {
305  var component = serializedObject.targetObject as Component;
306  if (component != null) {
307  var renderer = component.GetComponent<Renderer>();
308  if (renderer != null)
309  return new SerializedObject(renderer);
310  }
311  }
312 
313  return null;
314  }
