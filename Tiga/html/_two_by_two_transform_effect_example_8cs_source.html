37  // This is a sample component for C# vertex effects for Spine rendering components.
38  // Using shaders and materials to control vertex properties is still more performant
39  // than using this API, but in cases where your vertex effect logic cannot be
40  // expressed as shader code, these vertex effects can be useful.
41  public class TwoByTwoTransformEffectExample : MonoBehaviour {
42 
43  public Vector2 xAxis = new Vector2(1, 0);
44  public Vector2 yAxis = new Vector2(0, 1);
45 
46  SkeletonRenderer skeletonRenderer;
47 
48  void OnEnable () {
49  skeletonRenderer = GetComponent<SkeletonRenderer>();
50  if (skeletonRenderer == null) return;
51 
52  // Use the OnPostProcessVertices callback to modify the vertices at the correct time.
53  skeletonRenderer.OnPostProcessVertices -= ProcessVertices;
54  skeletonRenderer.OnPostProcessVertices += ProcessVertices;
55 
56  Debug.Log("2x2 Transform Effect Enabled.");
57  }
58 
59  void ProcessVertices (MeshGeneratorBuffers buffers) {
60  if (!this.enabled)
61  return;
62 
63  int vertexCount = buffers.vertexCount; // For efficiency, limit your effect to the actual mesh vertex count using vertexCount
64 
65  // Modify vertex positions by accessing Vector3[] vertexBuffer
66  var vertices = buffers.vertexBuffer;
67  Vector3 transformedPos = default(Vector3);
68  for (int i = 0; i < vertexCount; i++) {
69  Vector3 originalPos = vertices[i];
70  transformedPos.x = (xAxis.x * originalPos.x) + (yAxis.x * originalPos.y);
71  transformedPos.y = (xAxis.y * originalPos.x) + (yAxis.y * originalPos.y);
72  vertices[i] = transformedPos;
73  }
74 
75  }
76 
77  void OnDisable () {
78  if (skeletonRenderer == null) return;
79  skeletonRenderer.OnPostProcessVertices -= ProcessVertices;
80  Debug.Log("2x2 Transform Effect Disabled.");
81  }
82  }
83 
84 }
