215  public static Vector2[] GetLocalVertices (this VertexAttachment va, Slot slot, Vector2[] buffer) {
216  int floatsCount = va.worldVerticesLength;
217  int bufferTargetSize = floatsCount >> 1;
218  buffer = buffer ?? new Vector2[bufferTargetSize];
219  if (buffer.Length < bufferTargetSize) throw new System.ArgumentException(string.Format("Vector2 buffer too small. {0} requires an array of size {1}. Use the attachment's .WorldVerticesLength to get the correct size.", va.Name, floatsCount), "buffer");
220 
221  if (va.bones == null) {
222  var localVerts = va.vertices;
223  for (int i = 0; i < bufferTargetSize; i++) {
224  int j = i * 2;
225  buffer[i] = new Vector2(localVerts[j], localVerts[j+1]);
226  }
227  } else {
228  var floats = new float[floatsCount];
229  va.ComputeWorldVertices(slot, floats);
230 
231  Bone sb = slot.bone;
232  float ia, ib, ic, id, bwx = sb.worldX, bwy = sb.worldY;
233  sb.GetWorldToLocalMatrix(out ia, out ib, out ic, out id);
234 
235  for (int i = 0; i < bufferTargetSize; i++) {
236  int j = i * 2;
237  float x = floats[j] - bwx, y = floats[j+1] - bwy;
238  buffer[i] = new Vector2(x * ia + y * ib, x * ic + y * id);
239  }
240  }
241 
242  return buffer;
243  }

293  public static BoneMatrix CalculateSetupWorld (BoneData boneData) {
294  if (boneData == null)
295  return default(BoneMatrix);
296 
297  // End condition: isRootBone
298  if (boneData.parent == null)
299  return GetInheritedInternal(boneData, default(BoneMatrix));
300 
301  BoneMatrix result = CalculateSetupWorld(boneData.parent);
302  return GetInheritedInternal(boneData, result);
303  }
304 
305  static BoneMatrix GetInheritedInternal (BoneData boneData, BoneMatrix parentMatrix) {
306  var parent = boneData.parent;
307  if (parent == null) return new BoneMatrix(boneData); // isRootBone
308 
309  float pa = parentMatrix.a, pb = parentMatrix.b, pc = parentMatrix.c, pd = parentMatrix.d;
310  BoneMatrix result = default(BoneMatrix);
311  result.x = pa * boneData.x + pb * boneData.y + parentMatrix.x;
312  result.y = pc * boneData.x + pd * boneData.y + parentMatrix.y;
313 
314  switch (boneData.transformMode) {
315  case TransformMode.Normal: {
316  float rotationY = boneData.rotation + 90 + boneData.shearY;
317  float la = MathUtils.CosDeg(boneData.rotation + boneData.shearX) * boneData.scaleX;
318  float lb = MathUtils.CosDeg(rotationY) * boneData.scaleY;
319  float lc = MathUtils.SinDeg(boneData.rotation + boneData.shearX) * boneData.scaleX;
320  float ld = MathUtils.SinDeg(rotationY) * boneData.scaleY;
321  result.a = pa * la + pb * lc;
322  result.b = pa * lb + pb * ld;
323  result.c = pc * la + pd * lc;
324  result.d = pc * lb + pd * ld;
325  break;
326  }
327  case TransformMode.OnlyTranslation: {
328  float rotationY = boneData.rotation + 90 + boneData.shearY;
329  result.a = MathUtils.CosDeg(boneData.rotation + boneData.shearX) * boneData.scaleX;
330  result.b = MathUtils.CosDeg(rotationY) * boneData.scaleY;
331  result.c = MathUtils.SinDeg(boneData.rotation + boneData.shearX) * boneData.scaleX;
332  result.d = MathUtils.SinDeg(rotationY) * boneData.scaleY;
333  break;
334  }
335  case TransformMode.NoRotationOrReflection: {
336  float s = pa * pa + pc * pc, prx;
337  if (s > 0.0001f) {
338  s = Math.Abs(pa * pd - pb * pc) / s;
339  pb = pc * s;
340  pd = pa * s;
341  prx = MathUtils.Atan2(pc, pa) * MathUtils.RadDeg;
342  } else {
343  pa = 0;
344  pc = 0;
345  prx = 90 - MathUtils.Atan2(pd, pb) * MathUtils.RadDeg;
346  }
347  float rx = boneData.rotation + boneData.shearX - prx;
348  float ry = boneData.rotation + boneData.shearY - prx + 90;
349  float la = MathUtils.CosDeg(rx) * boneData.scaleX;
350  float lb = MathUtils.CosDeg(ry) * boneData.scaleY;
351  float lc = MathUtils.SinDeg(rx) * boneData.scaleX;
352  float ld = MathUtils.SinDeg(ry) * boneData.scaleY;
353  result.a = pa * la - pb * lc;
354  result.b = pa * lb - pb * ld;
355  result.c = pc * la + pd * lc;
356  result.d = pc * lb + pd * ld;
357  break;
358  }
359  case TransformMode.NoScale:
360  case TransformMode.NoScaleOrReflection: {
361  float cos = MathUtils.CosDeg(boneData.rotation), sin = MathUtils.SinDeg(boneData.rotation);
362  float za = pa * cos + pb * sin;
363  float zc = pc * cos + pd * sin;
364  float s = (float)Math.Sqrt(za * za + zc * zc);
365  if (s > 0.00001f)
366  s = 1 / s;
367  za *= s;
368  zc *= s;
369  s = (float)Math.Sqrt(za * za + zc * zc);
370  float r = MathUtils.PI / 2 + MathUtils.Atan2(zc, za);
371  float zb = MathUtils.Cos(r) * s;
372  float zd = MathUtils.Sin(r) * s;
373  float la = MathUtils.CosDeg(boneData.shearX) * boneData.scaleX;
374  float lb = MathUtils.CosDeg(90 + boneData.shearY) * boneData.scaleY;
375  float lc = MathUtils.SinDeg(boneData.shearX) * boneData.scaleX;
376  float ld = MathUtils.SinDeg(90 + boneData.shearY) * boneData.scaleY;
377  if (boneData.transformMode != TransformMode.NoScaleOrReflection ? pa * pd - pb * pc < 0 : false) {
378  zb = -zb;
379  zd = -zd;
380  }
381  result.a = za * la + zb * lc;
382  result.b = za * lb + zb * ld;
383  result.c = zc * la + zd * lc;
384  result.d = zc * lb + zd * ld;
385  break;
386  }
387  }
388 
389  return result;
390  }
