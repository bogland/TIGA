64  public override void DoUpdate () {
65  rayOrigin = transform.position + new Vector3(castOffset, castDistance, 0);
66 
67  hitY = float.MinValue;
68  if (use2D) {
69  RaycastHit2D hit;
70 
71  if (useRadius)
72  hit = Physics2D.CircleCast(rayOrigin, castRadius, rayDir, castDistance + groundOffset, groundMask);
73  else
74  hit = Physics2D.Raycast(rayOrigin, rayDir, castDistance + groundOffset, groundMask);
75 
76  if (hit.collider != null) {
77  hitY = hit.point.y + groundOffset;
78  if (Application.isPlaying)
79  hitY = Mathf.MoveTowards(lastHitY, hitY, adjustSpeed * Time.deltaTime);
80  } else {
81  if (Application.isPlaying)
82  hitY = Mathf.MoveTowards(lastHitY, transform.position.y, adjustSpeed * Time.deltaTime);
83  }
84  } else {
85  RaycastHit hit;
86  bool validHit = false;
87 
88  if (useRadius)
89  validHit = Physics.SphereCast(rayOrigin, castRadius, rayDir, out hit, castDistance + groundOffset, groundMask);
90  else
91  validHit = Physics.Raycast(rayOrigin, rayDir, out hit, castDistance + groundOffset, groundMask);
92 
93  if (validHit) {
94  hitY = hit.point.y + groundOffset;
95  if (Application.isPlaying)
96  hitY = Mathf.MoveTowards(lastHitY, hitY, adjustSpeed * Time.deltaTime);
97 
98  } else {
99  if (Application.isPlaying)
100  hitY = Mathf.MoveTowards(lastHitY, transform.position.y, adjustSpeed * Time.deltaTime);
101  }
102  }
103 
104  Vector3 v = transform.position;
105  v.y = Mathf.Clamp(v.y, Mathf.Min(lastHitY, hitY), float.MaxValue);
106  transform.position = v;
107 
108  utilBone.bone.X = transform.localPosition.x;
109  utilBone.bone.Y = transform.localPosition.y;
110 
111  lastHitY = hitY;
112  }
113 
114  void OnDrawGizmos () {
115  Vector3 hitEnd = rayOrigin + (rayDir * Mathf.Min(castDistance, rayOrigin.y - hitY));
116  Vector3 clearEnd = rayOrigin + (rayDir * castDistance);
117  Gizmos.DrawLine(rayOrigin, hitEnd);
118 
119  if (useRadius) {
120  Gizmos.DrawLine(new Vector3(hitEnd.x - castRadius, hitEnd.y - groundOffset, hitEnd.z), new Vector3(hitEnd.x + castRadius, hitEnd.y - groundOffset, hitEnd.z));
121  Gizmos.DrawLine(new Vector3(clearEnd.x - castRadius, clearEnd.y, clearEnd.z), new Vector3(clearEnd.x + castRadius, clearEnd.y, clearEnd.z));
122  }
123 
124  Gizmos.color = Color.red;
125  Gizmos.DrawLine(hitEnd, clearEnd);
126  }
