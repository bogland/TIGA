45  #region Context Menu Item

54  var f = go.AddComponent<BoneFollower>();

56 
57  EditorGUIUtility.PingObject(t);

59  Undo.RegisterCreatedObjectUndo(go, "Add BoneFollower");
60  }

90  public void OnSceneGUI () {
91  var tbf = target as BoneFollower;
92  var skeletonRendererComponent = tbf.skeletonRenderer;
93  if (skeletonRendererComponent == null) return;
94 
95  var transform = skeletonRendererComponent.transform;
96  var skeleton = skeletonRendererComponent.skeleton;
97 
98  if (string.IsNullOrEmpty(boneName.stringValue)) {
99  SpineHandles.DrawBones(transform, skeleton);
100  SpineHandles.DrawBoneNames(transform, skeleton);
101  Handles.Label(tbf.transform.position, "No bone selected", EditorStyles.helpBox);
102  } else {
103  var targetBone = tbf.bone;
104  if (targetBone == null) return;
105  SpineHandles.DrawBoneWireframe(transform, targetBone, SpineHandles.TransformContraintColor);
106  Handles.Label(targetBone.GetWorldPosition(transform), targetBone.Data.Name, SpineHandles.BoneNameStyle);
107  }
108  }

110  override public void OnInspectorGUI () {
111  if (serializedObject.isEditingMultipleObjects) {
112  if (needsReset) {
113  needsReset = false;
114  foreach (var o in targets) {
115  var bf = (BoneFollower)o;
116  bf.Initialize();
117  bf.LateUpdate();
118  }
119  SceneView.RepaintAll();
120  }
121 
122  EditorGUI.BeginChangeCheck();
123  DrawDefaultInspector();
124  needsReset |= EditorGUI.EndChangeCheck();
125  return;
126  }
127 
128  if (needsReset && Event.current.type == EventType.Layout) {
129  targetBoneFollower.Initialize();
130  targetBoneFollower.LateUpdate();
131  needsReset = false;
132  SceneView.RepaintAll();
133  }
134  serializedObject.Update();
135 
136  // Find Renderer
137  if (skeletonRenderer.objectReferenceValue == null) {
138  SkeletonRenderer parentRenderer = targetBoneFollower.GetComponentInParent<SkeletonRenderer>();
139  if (parentRenderer != null && parentRenderer.gameObject != targetBoneFollower.gameObject) {
140  skeletonRenderer.objectReferenceValue = parentRenderer;
141  Debug.Log("Inspector automatically assigned BoneFollower.SkeletonRenderer");
142  }
143  }
144 
145  EditorGUILayout.PropertyField(skeletonRenderer);
146  var skeletonRendererReference = skeletonRenderer.objectReferenceValue as SkeletonRenderer;
147  if (skeletonRendererReference != null) {
148  if (skeletonRendererReference.gameObject == targetBoneFollower.gameObject) {
149  skeletonRenderer.objectReferenceValue = null;
150  EditorUtility.DisplayDialog("Invalid assignment.", "BoneFollower can only follow a skeleton on a separate GameObject.\n\nCreate a new GameObject for your BoneFollower, or choose a SkeletonRenderer from a different GameObject.", "Ok");
151  }
152  }
153 
154  if (!targetBoneFollower.valid) {
155  needsReset = true;
156  }
157 
158  if (targetBoneFollower.valid) {
159  EditorGUI.BeginChangeCheck();
160  EditorGUILayout.PropertyField(boneName);
161  needsReset |= EditorGUI.EndChangeCheck();
162 
163  EditorGUILayout.PropertyField(followBoneRotation);
164  EditorGUILayout.PropertyField(followZPosition);
165  EditorGUILayout.PropertyField(followLocalScale);
166  EditorGUILayout.PropertyField(followSkeletonFlip);
167 
168  BoneFollowerInspector.RecommendRigidbodyButton(targetBoneFollower);
169  } else {
170  var boneFollowerSkeletonRenderer = targetBoneFollower.skeletonRenderer;
171  if (boneFollowerSkeletonRenderer == null) {
172  EditorGUILayout.HelpBox("SkeletonRenderer is unassigned. Please assign a SkeletonRenderer (SkeletonAnimation or SkeletonAnimator).", MessageType.Warning);
173  } else {
174  boneFollowerSkeletonRenderer.Initialize(false);
175 
176  if (boneFollowerSkeletonRenderer.skeletonDataAsset == null)
177  EditorGUILayout.HelpBox("Assigned SkeletonRenderer does not have SkeletonData assigned to it.", MessageType.Warning);
178 
179  if (!boneFollowerSkeletonRenderer.valid)
180  EditorGUILayout.HelpBox("Assigned SkeletonRenderer is invalid. Check target SkeletonRenderer, its SkeletonDataAsset or the console for other errors.", MessageType.Warning);
181  }
182  }
183 
184  var current = Event.current;
185  bool wasUndo = (current.type == EventType.ValidateCommand && current.commandName == "UndoRedoPerformed");
186  if (wasUndo)
187  targetBoneFollower.Initialize();
188 
189  serializedObject.ApplyModifiedProperties();
190  }

192  internal static void RecommendRigidbodyButton (Component component) {
193  bool hasCollider2D = component.GetComponent<Collider2D>() != null || component.GetComponent<BoundingBoxFollower>() != null;
194  bool hasCollider3D = !hasCollider2D && component.GetComponent<Collider>();
195  bool missingRigidBody = (hasCollider2D && component.GetComponent<Rigidbody2D>() == null) || (hasCollider3D && component.GetComponent<Rigidbody>() == null);
196  if (missingRigidBody) {
197  using (new SpineInspectorUtility.BoxScope()) {
198  EditorGUILayout.HelpBox("Collider detected. Unity recommends adding a Rigidbody to the parent Transforms of any colliders that are intended to be dynamically repositioned and rotated.", MessageType.Warning);
199  var rbType = hasCollider2D ? typeof(Rigidbody2D) : typeof(Rigidbody);
200  string rbLabel = string.Format("Add {0}", rbType.Name);
201  var rbContent = SpineInspectorUtility.TempContent(rbLabel, SpineInspectorUtility.UnityIcon(rbType), "Add a rigidbody to this GameObject to be the Physics body parent of the attached collider.");
202  if (SpineInspectorUtility.CenteredButton(rbContent)) component.gameObject.AddComponent(rbType);
203  }
204  }
205  }
206  }
