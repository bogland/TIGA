49  #if UNITY_EDITOR
50  void OnValidate () {
51  var skeletonComponent = GetComponent<ISkeletonComponent>();
52  var skeletonRenderer = skeletonComponent as SkeletonRenderer;
53  bool apma;
54 
55  if (skeletonRenderer != null) {
56  apma = skeletonRenderer.pmaVertexColors;
57  } else {
58  var skeletonGraphic = skeletonComponent as SkeletonGraphic;
59  apma = skeletonGraphic != null && skeletonGraphic.MeshGenerator.settings.pmaVertexColors;
60  }
61 
62  if (apma) {
63  try {
64  sprite.texture.GetPixel(0, 0);
65  } catch (UnityException e) {
66  Debug.LogFormat("Texture of {0} ({1}) is not read/write enabled. SpriteAttacher requires this in order to work with a SkeletonRenderer that renders premultiplied alpha. Please check the texture settings.", sprite.name, sprite.texture.name);
67  UnityEditor.EditorGUIUtility.PingObject(sprite.texture);
68  throw e;
69  }
70  }

78  static Dictionary<Texture, AtlasPage> atlasPageCache;
79  static AtlasPage GetPageFor (Texture texture, Shader shader) {
80  if (atlasPageCache == null) atlasPageCache = new Dictionary<Texture, AtlasPage>();
81  AtlasPage atlasPage;
82  atlasPageCache.TryGetValue(texture, out atlasPage);
83  if (atlasPage == null) {
84  var newMaterial = new Material(shader);
85  atlasPage = newMaterial.ToSpineAtlasPage();
86  atlasPageCache[texture] = atlasPage;
87  }
88  return atlasPage;
89  }

104  public void Initialize (bool overwrite = true) {
105  if (overwrite || attachment == null) {
106  // Get the applyPMA value.
107  var skeletonComponent = GetComponent<ISkeletonComponent>();
108  var skeletonRenderer = skeletonComponent as SkeletonRenderer;
109  if (skeletonRenderer != null)
110  this.applyPMA = skeletonRenderer.pmaVertexColors;
111  else {
112  var skeletonGraphic = skeletonComponent as SkeletonGraphic;
113  if (skeletonGraphic != null)
114  this.applyPMA = skeletonGraphic.MeshGenerator.settings.pmaVertexColors;
115  }
116 
117  // Subscribe to UpdateComplete to override animation keys.
118  if (overrideAnimation) {
119  var animatedSkeleton = skeletonComponent as ISkeletonAnimation;
120  if (animatedSkeleton != null) {
121  animatedSkeleton.UpdateComplete -= AnimationOverrideSpriteAttach;
122  animatedSkeleton.UpdateComplete += AnimationOverrideSpriteAttach;
123  }
124  }
125 
126  spineSlot = spineSlot ?? skeletonComponent.Skeleton.FindSlot(slot);
127  Shader attachmentShader = applyPMA ? Shader.Find(DefaultPMAShader) : Shader.Find(DefaultStraightAlphaShader);
128  attachment = applyPMA ? sprite.ToRegionAttachmentPMAClone(attachmentShader) : sprite.ToRegionAttachment(SpriteAttacher.GetPageFor(sprite.texture, attachmentShader));
129  }
130  }
131 
132  void OnDestroy () {
133  var animatedSkeleton = GetComponent<ISkeletonAnimation>();
134  if (animatedSkeleton != null)
135  animatedSkeleton.UpdateComplete -= AnimationOverrideSpriteAttach;
136  }
137 
139  public void Attach () {
140  if (spineSlot != null)
141  spineSlot.Attachment = attachment;
142  }
143 
144  }
