85  public void RenderParts (ExposedList<SubmeshInstruction> instructions, int startSubmesh, int endSubmesh) {
86  LazyIntialize();
87 
88  // STEP 1: Create instruction
89  var smartMesh = buffers.GetNextMesh();
90  currentInstructions.SetWithSubset(instructions, startSubmesh, endSubmesh);
91  bool updateTriangles = SkeletonRendererInstruction.GeometryNotEqual(currentInstructions, smartMesh.instructionUsed);
92 
93  // STEP 2: Generate mesh buffers.
94  var currentInstructionsSubmeshesItems = currentInstructions.submeshInstructions.Items;
95  meshGenerator.Begin();
96  if (currentInstructions.hasActiveClipping) {
97  for (int i = 0; i < currentInstructions.submeshInstructions.Count; i++)
98  meshGenerator.AddSubmesh(currentInstructionsSubmeshesItems[i], updateTriangles);
99  } else {
100  meshGenerator.BuildMeshWithArrays(currentInstructions, updateTriangles);
101  }
102 
103  buffers.UpdateSharedMaterials(currentInstructions.submeshInstructions);
104 
105  // STEP 3: modify mesh.
106  var mesh = smartMesh.mesh;
107 
108  if (meshGenerator.VertexCount <= 0) { // Clear an empty mesh
109  updateTriangles = false;
110  mesh.Clear();
111  } else {
112  meshGenerator.FillVertexData(mesh);
113  if (updateTriangles) {
114  meshGenerator.FillTriangles(mesh);
115  meshRenderer.sharedMaterials = buffers.GetUpdatedSharedMaterialsArray();
116  } else if (buffers.MaterialsChangedInLastUpdate()) {
117  meshRenderer.sharedMaterials = buffers.GetUpdatedSharedMaterialsArray();
118  }
119  }
120  meshGenerator.FillLateVertexData(mesh);
121 
122  meshFilter.sharedMesh = mesh;
123  smartMesh.instructionUsed.Set(currentInstructions);
124  }
