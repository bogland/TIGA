69  public override void OnGUI (Rect position, SerializedProperty property, GUIContent label) {
70  SerializedProperty = property;
71

80  var objectReferenceValue = dataField.objectReferenceValue;
81  if (objectReferenceValue is SkeletonDataAsset) {

83  } else if (objectReferenceValue is IHasSkeletonDataAsset) {
84  var hasSkeletonDataAsset = (IHasSkeletonDataAsset)objectReferenceValue;
85  if (hasSkeletonDataAsset != null)
86  skeletonDataAsset = hasSkeletonDataAsset.SkeletonDataAsset;
87  } else if (objectReferenceValue != null) {
88  EditorGUI.LabelField(position, "ERROR:", "Invalid reference type");
89  return;
90  }
91 
92  } else {
93  var targetObject = property.serializedObject.targetObject;
94 
95  IHasSkeletonDataAsset hasSkeletonDataAsset = targetObject as IHasSkeletonDataAsset;
96  if (hasSkeletonDataAsset == null) {
97  var component = targetObject as Component;
98  if (component != null)
99  hasSkeletonDataAsset = component.GetComponentInChildren(typeof(IHasSkeletonDataAsset)) as IHasSkeletonDataAsset;
100  }
101 
102  if (hasSkeletonDataAsset != null)
103  skeletonDataAsset = hasSkeletonDataAsset.SkeletonDataAsset;
104  }
105 
106  if (skeletonDataAsset == null) {
107  if (TargetAttribute.fallbackToTextField) {
108  EditorGUI.PropertyField(position, property); //EditorGUI.TextField(position, label, property.stringValue);
109  } else {
110  EditorGUI.LabelField(position, "ERROR:", "Must have reference to a SkeletonDataAsset");
111  }
112 
113  skeletonDataAsset = property.serializedObject.targetObject as SkeletonDataAsset;
114  if (skeletonDataAsset == null) return;
115  }
116 
117  position = EditorGUI.PrefixLabel(position, label);
118 
119  var image = Icon;
120  var propertyStringValue = (property.hasMultipleDifferentValues) ? SpineInspectorUtility.EmDash : property.stringValue;
121  if (GUI.Button(position, string.IsNullOrEmpty(propertyStringValue) ? NoneLabel(image) : SpineInspectorUtility.TempContent(propertyStringValue, image), EditorStyles.popup))
122  Selector(property);
123  }

125  public ISkeletonComponent GetTargetSkeletonComponent (SerializedProperty property) {
126  var dataField = property.FindBaseOrSiblingProperty(TargetAttribute.dataField);
127 
128  if (dataField != null) {
129  var skeletonComponent = dataField.objectReferenceValue as ISkeletonComponent;
130  if (dataField.objectReferenceValue != null && skeletonComponent != null) // note the overloaded UnityEngine.Object == null check. Do not simplify.
131  return skeletonComponent;

134  if (component != null)
135  return component.GetComponentInChildren(typeof(ISkeletonComponent)) as ISkeletonComponent;
136  }
137 
138  return null;

392  protected override void PopulateMenu (GenericMenu menu, SerializedProperty property, SpineAttachment targetAttribute, SkeletonData data) {
393  ISkeletonComponent skeletonComponent = GetTargetSkeletonComponent(property);
394  var validSkins = new List<Skin>();
395 
396  if (skeletonComponent != null && targetAttribute.currentSkinOnly) {
397  Skin currentSkin = null;
398 
399  var skinProperty = property.FindBaseOrSiblingProperty(targetAttribute.skinField);
400  if (skinProperty != null) currentSkin = skeletonComponent.Skeleton.Data.FindSkin(skinProperty.stringValue);
401 
402  currentSkin = currentSkin ?? skeletonComponent.Skeleton.Skin;
403  if (currentSkin != null)
404  validSkins.Add(currentSkin);
405  else
406  validSkins.Add(data.Skins.Items[0]);
407 
408  } else {
409  foreach (Skin skin in data.Skins)
410  if (skin != null) validSkins.Add(skin);
411  }
412 
413  var attachmentNames = new List<string>();
414  var placeholderNames = new List<string>();
415  string prefix = "";
416 
417  if (skeletonComponent != null && targetAttribute.currentSkinOnly)
418  menu.AddDisabledItem(new GUIContent((skeletonComponent as Component).gameObject.name + " (Skeleton)"));
419  else
420  menu.AddDisabledItem(new GUIContent(skeletonDataAsset.name));
421 
422  menu.AddSeparator("");
423  if (TargetAttribute.includeNone) {
424  const string NullAttachmentName = "";
425  menu.AddItem(new GUIContent("Null"), property.stringValue == NullAttachmentName, HandleSelect, new SpineDrawerValuePair(NullAttachmentName, property));
426  menu.AddSeparator("");
427  }
428 
429  Skin defaultSkin = data.Skins.Items[0];
430  var slotProperty = property.FindBaseOrSiblingProperty(TargetAttribute.slotField);
431 
432  string slotMatch = "";
433  if (slotProperty != null) {
434  if (slotProperty.propertyType == SerializedPropertyType.String)
435  slotMatch = slotProperty.stringValue.ToLower();
436  }
437 
438  foreach (Skin skin in validSkins) {
439  string skinPrefix = skin.Name + "/";
440 
441  if (validSkins.Count > 1)
442  prefix = skinPrefix;
443 
444  for (int i = 0; i < data.Slots.Count; i++) {
445  if (slotMatch.Length > 0 && !(data.Slots.Items[i].Name.Equals(slotMatch, StringComparison.OrdinalIgnoreCase)))
446  continue;
447 
448  attachmentNames.Clear();
449  placeholderNames.Clear();
450 
451  skin.FindNamesForSlot(i, attachmentNames);
452  if (skin != defaultSkin) {
453  defaultSkin.FindNamesForSlot(i, attachmentNames);
454  skin.FindNamesForSlot(i, placeholderNames);
455  }
456 
457  for (int a = 0; a < attachmentNames.Count; a++) {
458  string attachmentPath = attachmentNames[a];
459  string menuPath = prefix + data.Slots.Items[i].Name + "/" + attachmentPath;
460  string name = attachmentNames[a];
461 
462  if (targetAttribute.returnAttachmentPath)
463  name = skin.Name + "/" + data.Slots.Items[i].Name + "/" + attachmentPath;
464 
465  if (targetAttribute.placeholdersOnly && !placeholderNames.Contains(attachmentPath)) {
466  menu.AddDisabledItem(new GUIContent(menuPath));
467  } else {
468  menu.AddItem(new GUIContent(menuPath), name == property.stringValue, HandleSelect, new SpineDrawerValuePair(name, property));
