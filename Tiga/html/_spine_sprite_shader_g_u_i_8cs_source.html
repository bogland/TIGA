296  protected virtual bool RenderModes () {
297  bool dataChanged = false;
298 
299  //Lighting Mode
300  {
301  EditorGUI.BeginChangeCheck();
302 
303  eLightMode lightMode = GetMaterialLightMode((Material)_materialEditor.target);
304  EditorGUI.showMixedValue = false;
305  foreach (Material material in _materialEditor.targets) {
306  if (lightMode != GetMaterialLightMode(material)) {
307  EditorGUI.showMixedValue = true;
308  break;
309  }
310  }
311 
312  lightMode = (eLightMode)EditorGUILayout.Popup(_lightingModeText, (int)lightMode, _lightingModeOptions);
313  if (EditorGUI.EndChangeCheck()) {
314  foreach (Material material in _materialEditor.targets) {
315  switch (lightMode) {
316  case eLightMode.VertexLit:
317  if (material.shader.name != kShaderVertexLit)
318  _materialEditor.SetShader(Shader.Find(kShaderVertexLit), false);
319  break;
320  case eLightMode.PixelLit:
321  if (material.shader.name != kShaderPixelLit)
322  _materialEditor.SetShader(Shader.Find(kShaderPixelLit), false);
323  break;
324  case eLightMode.Unlit:
325  if (material.shader.name != kShaderUnlit)
326  _materialEditor.SetShader(Shader.Find(kShaderUnlit), false);
327  break;
328  }
329  }
330 
331  dataChanged = true;
332  }
333  }
334 
335  //Blend Mode
336  {
337  eBlendMode blendMode = GetMaterialBlendMode((Material)_materialEditor.target);
338  EditorGUI.showMixedValue = false;
339  foreach (Material material in _materialEditor.targets) {
340  if (blendMode != GetMaterialBlendMode(material)) {
341  EditorGUI.showMixedValue = true;
342  break;
343  }
344  }
345 
346  EditorGUI.BeginChangeCheck();
347  blendMode = (eBlendMode)EditorGUILayout.Popup(_blendModeText, (int)blendMode, _blendModeOptions);
348  if (EditorGUI.EndChangeCheck()) {
349  foreach (Material mat in _materialEditor.targets) {
350  SetBlendMode(mat, blendMode);
351  }
352 
353  dataChanged = true;
354  }
355  }
356

383

420  protected virtual bool RenderDepthProperties () {
421  bool dataChanged = false;
422 
423  EditorGUI.BeginChangeCheck();
424 
425  bool mixedValue = _writeToDepth.hasMixedValue;
426  EditorGUI.showMixedValue = mixedValue;
427  bool writeTodepth = EditorGUILayout.Toggle(_depthText, _writeToDepth.floatValue != 0.0f);
428 
429  if (EditorGUI.EndChangeCheck()) {
430  SetInt("_ZWrite", writeTodepth ? 1 : 0);
431  _depthAlphaCutoff.floatValue = writeTodepth ? 0.5f : 0.0f;
432  mixedValue = false;
433  dataChanged = true;
434  }
435 
436  if (writeTodepth && !mixedValue && GetMaterialBlendMode((Material)_materialEditor.target) != eBlendMode.Opaque) {
437  EditorGUI.BeginChangeCheck();
438  _materialEditor.RangeProperty(_depthAlphaCutoff, _depthAlphaCutoffText.text);
439  dataChanged |= EditorGUI.EndChangeCheck();
440  }
441 
442  {
443  bool useCustomRenderType = _customRenderQueue.floatValue > 0.0f;
444  EditorGUI.BeginChangeCheck();
445  EditorGUI.showMixedValue = _customRenderQueue.hasMixedValue;
446  useCustomRenderType = EditorGUILayout.Toggle(_customRenderType, useCustomRenderType);
447  if (EditorGUI.EndChangeCheck()) {
448  dataChanged = true;
449 
450  _customRenderQueue.floatValue = useCustomRenderType ? 1.0f : 0.0f;
451 
452  foreach (Material material in _materialEditor.targets) {
453  eBlendMode blendMode = GetMaterialBlendMode(material);
454 
455  switch (blendMode) {
456  case eBlendMode.Opaque:
457  {
458  SetRenderType(material, "Opaque", useCustomRenderType);
459  }
460  break;
461  default:
462  {
463  bool zWrite = material.GetFloat("_ZWrite") > 0.0f;
464  SetRenderType(material, zWrite ? "TransparentCutout" : "Transparent", useCustomRenderType);
465  }
466  break;
467  }
468  }
469  }
470  }
471 
472  EditorGUI.showMixedValue = false;
473 
474  return dataChanged;
475  }

477  protected virtual bool RenderNormalsProperties () {
478  bool dataChanged = false;
479 
480  eNormalsMode normalsMode = GetMaterialNormalsMode((Material)_materialEditor.target);
481  bool mixedNormalsMode = false;
482  foreach (Material material in _materialEditor.targets) {
483  if (normalsMode != GetMaterialNormalsMode(material)) {
484  mixedNormalsMode = true;
485  break;
486  }
487  }
488 
489  EditorGUI.BeginChangeCheck();
490  EditorGUI.showMixedValue = mixedNormalsMode;
491  bool fixedNormals = BoldToggleField(_fixedNormalText, normalsMode != eNormalsMode.MeshNormals);
492 
493  if (EditorGUI.EndChangeCheck()) {
494  normalsMode = fixedNormals ? eNormalsMode.FixedNormalsViewSpace : eNormalsMode.MeshNormals;
495  SetNormalsMode(_materialEditor, normalsMode, false);
496  _fixedNormal.vectorValue = new Vector4(0.0f, 0.0f, normalsMode == eNormalsMode.FixedNormalsViewSpace ? 1.0f : -1.0f, 1.0f);
497  mixedNormalsMode = false;
498  dataChanged = true;
499  }
500 
501  if (fixedNormals) {
502  //Show drop down for normals space
503  EditorGUI.BeginChangeCheck();
504  EditorGUI.showMixedValue = mixedNormalsMode;
505  normalsMode = (eNormalsMode)EditorGUILayout.Popup(_fixedNormalSpaceText, (int)normalsMode, _fixedNormalSpaceOptions);
506  if (EditorGUI.EndChangeCheck()) {
507  SetNormalsMode((Material)_materialEditor.target, normalsMode, GetMaterialFixedNormalsBackfaceRenderingOn((Material)_materialEditor.target));
508 
509  foreach (Material material in _materialEditor.targets) {
510  SetNormalsMode(material, normalsMode, GetMaterialFixedNormalsBackfaceRenderingOn(material));
511  }
512 
513  //Reset fixed normal to default (Vector3.forward for model-space, -Vector3.forward for view-space).
514  _fixedNormal.vectorValue = new Vector4(0.0f, 0.0f, normalsMode == eNormalsMode.FixedNormalsViewSpace ? 1.0f : -1.0f, 1.0f);
515 
516  mixedNormalsMode = false;
517  dataChanged = true;
518  }
519 
520  //Show fixed normal
521  EditorGUI.BeginChangeCheck();
522  EditorGUI.showMixedValue = _fixedNormal.hasMixedValue;
523  Vector3 normal = EditorGUILayout.Vector3Field(_fixedNormalDirectionText, _fixedNormal.vectorValue);
524  if (EditorGUI.EndChangeCheck()) {
525  _fixedNormal.vectorValue = new Vector4(normal.x, normal.y, normal.z, 1.0f);
526  dataChanged = true;
527  }
528 
529  //Show adjust for back face rendering
530  {
531  bool fixBackFaceRendering = GetMaterialFixedNormalsBackfaceRenderingOn((Material)_materialEditor.target);
532  bool mixedBackFaceRendering = false;
533  foreach (Material material in _materialEditor.targets) {
534  if (fixBackFaceRendering != GetMaterialFixedNormalsBackfaceRenderingOn(material)) {
535  mixedBackFaceRendering = true;
536  break;
537  }
538  }
539 
540  EditorGUI.BeginChangeCheck();
541  EditorGUI.showMixedValue = mixedBackFaceRendering;
542  bool backRendering = EditorGUILayout.Toggle(_adjustBackfaceTangentText, fixBackFaceRendering);
543 
544  if (EditorGUI.EndChangeCheck()) {
545  SetNormalsMode(_materialEditor, normalsMode, backRendering);
546  dataChanged = true;
547  }
548  }
549 
550  }
551 
552  EditorGUI.showMixedValue = false;
553 
554  return dataChanged;
555  }

873  static void SetBlendMode (Material material, eBlendMode blendMode) {
874  SetKeyword(material, "_ALPHABLEND_ON", blendMode == eBlendMode.StandardAlpha);
875  SetKeyword(material, "_ALPHAPREMULTIPLY_ON", blendMode == eBlendMode.PreMultipliedAlpha);
876  SetKeyword(material, "_MULTIPLYBLEND", blendMode == eBlendMode.Multiply);
877  SetKeyword(material, "_MULTIPLYBLEND_X2", blendMode == eBlendMode.Multiplyx2);
878  SetKeyword(material, "_ADDITIVEBLEND", blendMode == eBlendMode.Additive);
879  SetKeyword(material, "_ADDITIVEBLEND_SOFT", blendMode == eBlendMode.SoftAdditive);
880 
881  int renderQueue;
882  bool useCustomRenderQueue = material.GetFloat("_CustomRenderQueue") > 0.0f;
883 
884  switch (blendMode) {
885  case eBlendMode.Opaque:
886  {
887  material.SetInt("_SrcBlend", (int)UnityEngine.Rendering.BlendMode.One);
888  material.SetInt("_DstBlend", (int)UnityEngine.Rendering.BlendMode.Zero);
889  SetRenderType(material, "Opaque", useCustomRenderQueue);
890  renderQueue = kSolidQueue;
891  }
892  break;
893  case eBlendMode.Additive:
894  {
895  material.SetInt("_SrcBlend", (int)UnityEngine.Rendering.BlendMode.SrcAlpha);
896  material.SetInt("_DstBlend", (int)UnityEngine.Rendering.BlendMode.One);
897  bool zWrite = material.GetFloat("_ZWrite") > 0.0f;
898  SetRenderType(material, zWrite ? "TransparentCutout" : "Transparent", useCustomRenderQueue);
899  renderQueue = zWrite ? kAlphaTestQueue : kTransparentQueue;
900  }
901  break;
902  case eBlendMode.SoftAdditive:
903  {
904  material.SetInt("_SrcBlend", (int)UnityEngine.Rendering.BlendMode.One);
905  material.SetInt("_DstBlend", (int)UnityEngine.Rendering.BlendMode.OneMinusSrcColor);
906  bool zWrite = material.GetFloat("_ZWrite") > 0.0f;
907  SetRenderType(material, zWrite ? "TransparentCutout" : "Transparent", useCustomRenderQueue);
908  renderQueue = zWrite ? kAlphaTestQueue : kTransparentQueue;
909  }
910  break;
911  case eBlendMode.Multiply:
912  {
913  material.SetInt("_SrcBlend", (int)UnityEngine.Rendering.BlendMode.Zero);
914  material.SetInt("_DstBlend", (int)UnityEngine.Rendering.BlendMode.SrcColor);
915  bool zWrite = material.GetFloat("_ZWrite") > 0.0f;
916  SetRenderType(material, zWrite ? "TransparentCutout" : "Transparent", useCustomRenderQueue);
917  renderQueue = zWrite ? kAlphaTestQueue : kTransparentQueue;
918  }
919  break;
920  case eBlendMode.Multiplyx2:
921  {
922  material.SetInt("_SrcBlend", (int)UnityEngine.Rendering.BlendMode.DstColor);
923  material.SetInt("_DstBlend", (int)UnityEngine.Rendering.BlendMode.SrcColor);
924  bool zWrite = material.GetFloat("_ZWrite") > 0.0f;
925  SetRenderType(material, zWrite ? "TransparentCutout" : "Transparent", useCustomRenderQueue);
926  renderQueue = zWrite ? kAlphaTestQueue : kTransparentQueue;
927  }
928  break;
929  default:
930  {
931  material.SetInt("_SrcBlend", (int)UnityEngine.Rendering.BlendMode.One);
932  material.SetInt("_DstBlend", (int)UnityEngine.Rendering.BlendMode.OneMinusSrcAlpha);
933  bool zWrite = material.GetFloat("_ZWrite") > 0.0f;
934  SetRenderType(material, zWrite ? "TransparentCutout" : "Transparent", useCustomRenderQueue);
935  renderQueue = zWrite ? kAlphaTestQueue : kTransparentQueue;
936  }
937  break;
938  }
939 
940  material.renderQueue = renderQueue + material.GetInt("_RenderQueue");
941  material.SetOverrideTag("IgnoreProjector", blendMode == eBlendMode.Opaque ? "False" : "True");
942  }
