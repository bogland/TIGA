63  public override void OnInspectorGUI () {
64  animationNameProperty = animationNameProperty ?? serializedObject.FindProperty("animationName");
65  string animationName = animationNameProperty.stringValue;
66 
67  Animation animation = null;
68  if (ThisSkeletonDataAsset != null) {
69  var skeletonData = ThisSkeletonDataAsset.GetSkeletonData(true);
70  if (skeletonData != null) {
71  animation = skeletonData.FindAnimation(animationName);
72  }
73  }
74  bool animationNotFound = (animation == null);
75 
76  if (changeNextFrame) {
77  changeNextFrame = false;
78 
79  if (ThisSkeletonDataAsset != lastSkeletonDataAsset) {
80  preview.Clear();
81  preview.Initialize(Repaint, ThisSkeletonDataAsset, LastSkinName);
82 
83  if (animationNotFound) {
84  animationNameProperty.stringValue = "";
85  preview.ClearAnimationSetupPose();
86  }
87  }
88 
89  preview.ClearAnimationSetupPose();
90 
91  if (!string.IsNullOrEmpty(animationNameProperty.stringValue))
92  preview.PlayPauseAnimation(animationNameProperty.stringValue, true);
93  }
94  lastSkeletonDataAsset = ThisSkeletonDataAsset;
95 
96  //EditorGUILayout.HelpBox(AnimationReferenceAssetEditor.InspectorHelpText, MessageType.Info, true);
97  EditorGUILayout.Space();
98  EditorGUI.BeginChangeCheck();
99  DrawDefaultInspector();
100  if (EditorGUI.EndChangeCheck()) {
101  changeNextFrame = true;
102  }
103 
104  // Draw extra info below default inspector.
105  EditorGUILayout.Space();
106  if (ThisSkeletonDataAsset == null) {
107  EditorGUILayout.HelpBox("SkeletonDataAsset is missing.", MessageType.Error);
108  } else if (string.IsNullOrEmpty(animationName)) {
109  EditorGUILayout.HelpBox("No animation selected.", MessageType.Warning);
110  } else if (animationNotFound) {
111  EditorGUILayout.HelpBox(string.Format("Animation named {0} was not found for this Skeleton.", animationNameProperty.stringValue), MessageType.Warning);
112  } else {

114  if (!string.Equals(SpineEditorUtilities.GetPathSafeName(animationName), ThisAnimationReferenceAsset.name, System.StringComparison.OrdinalIgnoreCase))
115  EditorGUILayout.HelpBox("Animation name value does not match this asset's name. Inspectors using this asset may be misleading.", MessageType.None);
116 
117  EditorGUILayout.LabelField(SpineInspectorUtility.TempContent(animationName, SpineEditorUtilities.Icons.animation));
