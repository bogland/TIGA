65  public void SetMix (Animation from, Animation to, float duration) {
66  if (from == null) throw new ArgumentNullException("from", "from cannot be null.");
67  if (to == null) throw new ArgumentNullException("to", "to cannot be null.");
68  AnimationPair key = new AnimationPair(from, to);

77  public float GetMix (Animation from, Animation to) {
78  if (from == null) throw new ArgumentNullException("from", "from cannot be null.");
79  if (to == null) throw new ArgumentNullException("to", "to cannot be null.");
80  AnimationPair key = new AnimationPair(from, to);
81  float duration;
82  if (animationToMixTime.TryGetValue(key, out duration)) return duration;
83  return defaultMix;
84  }
