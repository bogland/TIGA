61  void Apply (SkeletonRenderer skeletonRenderer) {
62  if (!this.enabled) return;
63 
64  atlas = atlasAsset.GetAtlas();
65  if (atlas == null) return;
66  float scale = skeletonRenderer.skeletonDataAsset.scale;
67 
68  foreach (var entry in attachments) {
69  Slot slot = skeletonRenderer.Skeleton.FindSlot(entry.slot);
70  Attachment originalAttachment = slot.Attachment;
71  AtlasRegion region = atlas.FindRegion(entry.region);
72 
73  if (region == null) {
74  slot.Attachment = null;
75  } else if (inheritProperties && originalAttachment != null) {
76  slot.Attachment = originalAttachment.GetRemappedClone(region, true, true, scale);
77  } else {
78  var newRegionAttachment = region.ToRegionAttachment(region.name, scale);
79  slot.Attachment = newRegionAttachment;
80  }
81  }
82  }
