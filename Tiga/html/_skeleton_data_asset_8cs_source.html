173  internal Atlas[] GetAtlasArray () {
174  var returnList = new System.Collections.Generic.List<Atlas>(atlasAssets.Length);
175  for (int i = 0; i < atlasAssets.Length; i++) {
176  var aa = atlasAssets[i];
177  if (aa == null) continue;
178  var a = aa.GetAtlas();
179  if (a == null) continue;
180  returnList.Add(a);
181  }
182  return returnList.ToArray();
