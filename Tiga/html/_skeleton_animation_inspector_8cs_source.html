102  protected void TrySetAnimation (Object o, bool multi) {
103  var skeletonAnimation = o as SkeletonAnimation;
104  if (skeletonAnimation == null) return;
105  if (!skeletonAnimation.valid)
106  return;
107 
108  if (!isInspectingPrefab) {
109  if (wasAnimationNameChanged) {
110  if (!Application.isPlaying) {
111  if (skeletonAnimation.state != null) skeletonAnimation.state.ClearTrack(0);
112  skeletonAnimation.skeleton.SetToSetupPose();
113  }
114 
115  Spine.Animation animationToUse = skeletonAnimation.skeleton.Data.FindAnimation(animationName.stringValue);
116 
117  if (!Application.isPlaying) {
118  if (animationToUse != null) animationToUse.PoseSkeleton(skeletonAnimation.Skeleton, 0f);
119  skeletonAnimation.Update(0);
120  skeletonAnimation.LateUpdate();
121  requireRepaint = true;
122  } else {
123  if (animationToUse != null)
124  skeletonAnimation.state.SetAnimation(0, animationToUse, loop.boolValue);
125  else
126  skeletonAnimation.state.ClearTrack(0);
127  }
128 
129  wasAnimationNameChanged = false;
130  }
131 
132  // Reflect animationName serialized property in the inspector even if SetAnimation API was used.
133  if (!multi && Application.isPlaying) {
134  TrackEntry current = skeletonAnimation.state.GetCurrent(0);
135  if (current != null) {
136  if (skeletonAnimation.AnimationName != animationName.stringValue)
137  animationName.stringValue = current.Animation.Name;
138  }
139  }
140  }
141  }
142  }
