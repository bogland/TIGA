74  IEnumerator Fade () {
75  Color32 c;
76  for (int t = 0; t < 500; t++) {
77  bool breakout = true;
78  for (int i = 0; i < colors.Length; i++) {
79  c = colors[i];

81  breakout = false;

83  colors[i] = Color32.Lerp(c, black, Time.deltaTime * fadeSpeed);
84  }
