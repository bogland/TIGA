78  void UpdateReferences () {
79  skeletonTransform = skeletonRenderer.transform;
80  skeletonRenderer.OnRebuild -= HandleRebuildRenderer;
81  skeletonRenderer.OnRebuild += HandleRebuildRenderer;
82  skeletonTransformIsParent = Transform.ReferenceEquals(skeletonTransform, transform.parent);

84  bone = null;
85  point = null;

98  public void LateUpdate () {
99  #if UNITY_EDITOR
100  if (!Application.isPlaying) skeletonTransformIsParent = Transform.ReferenceEquals(skeletonTransform, transform.parent);
101  #endif
102 
103  if (point == null) {
104  if (string.IsNullOrEmpty(pointAttachmentName)) return;
105  UpdateReferences();
106  if (point == null) return;
107  }
108 
109  Vector2 worldPos;
110  point.ComputeWorldPosition(bone, out worldPos.x, out worldPos.y);
111  float rotation = point.ComputeWorldRotation(bone);
112 
113  Transform thisTransform = this.transform;
114  if (skeletonTransformIsParent) {
115  // Recommended setup: Use local transform properties if Spine GameObject is the immediate parent
116  thisTransform.localPosition = new Vector3(worldPos.x, worldPos.y, followSkeletonZPosition ? 0f : thisTransform.localPosition.z);
117  if (followRotation) {
118  float halfRotation = rotation * 0.5f * Mathf.Deg2Rad;
119 
120  var q = default(Quaternion);
121  q.z = Mathf.Sin(halfRotation);
122  q.w = Mathf.Cos(halfRotation);
123  thisTransform.localRotation = q;
124  }
125  } else {
126  // For special cases: Use transform world properties if transform relationship is complicated
127  Vector3 targetWorldPosition = skeletonTransform.TransformPoint(new Vector3(worldPos.x, worldPos.y, 0f));
128  if (!followSkeletonZPosition)
129  targetWorldPosition.z = thisTransform.position.z;
130 
131  Transform transformParent = thisTransform.parent;
132  if (transformParent != null) {
133  Matrix4x4 m = transformParent.localToWorldMatrix;
134  if (m.m00 * m.m11 - m.m01 * m.m10 < 0) // Determinant2D is negative
135  rotation = -rotation;
136  }
137 
138  if (followRotation) {
139  Vector3 transformWorldRotation = skeletonTransform.rotation.eulerAngles;
140  thisTransform.SetPositionAndRotation(targetWorldPosition, Quaternion.Euler(transformWorldRotation.x, transformWorldRotation.y, transformWorldRotation.z + rotation));
141  } else {
142  thisTransform.position = targetWorldPosition;
143  }
144  }
145 
146  if (followSkeletonFlip) {
147  Vector3 localScale = thisTransform.localScale;
148  localScale.y = Mathf.Abs(localScale.y) * (bone.skeleton.flipX ^ bone.skeleton.flipY ? -1f : 1f);
149  thisTransform.localScale = localScale;
150  }
151  }
152  }
