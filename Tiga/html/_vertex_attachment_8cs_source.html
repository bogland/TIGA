68  public void ComputeWorldVertices (Slot slot, int start, int count, float[] worldVertices, int offset, int stride = 2) {
69  count = offset + (count >> 1) * stride;
70  Skeleton skeleton = slot.bone.skeleton;
71  var deformArray = slot.attachmentVertices;
72  float[] vertices = this.vertices;
73  int[] bones = this.bones;
74  if (bones == null) {
75  if (deformArray.Count > 0) vertices = deformArray.Items;
76  Bone bone = slot.bone;
77  float x = bone.worldX, y = bone.worldY;
78  float a = bone.a, b = bone.b, c = bone.c, d = bone.d;
79  for (int vv = start, w = offset; w < count; vv += 2, w += stride) {
80  float vx = vertices[vv], vy = vertices[vv + 1];
81  worldVertices[w] = vx * a + vy * b + x;
82  worldVertices[w + 1] = vx * c + vy * d + y;
83  }
84  return;
85  }
86  int v = 0, skip = 0;
87  for (int i = 0; i < start; i += 2) {
88  int n = bones[v];
89  v += n + 1;
90  skip += n;
91  }
92  var skeletonBones = skeleton.bones.Items;
93  if (deformArray.Count == 0) {
94  for (int w = offset, b = skip * 3; w < count; w += stride) {
95  float wx = 0, wy = 0;
96  int n = bones[v++];
97  n += v;
98  for (; v < n; v++, b += 3) {
99  Bone bone = skeletonBones[bones[v]];
100  float vx = vertices[b], vy = vertices[b + 1], weight = vertices[b + 2];
101  wx += (vx * bone.a + vy * bone.b + bone.worldX) * weight;
102  wy += (vx * bone.c + vy * bone.d + bone.worldY) * weight;
103  }
104  worldVertices[w] = wx;
105  worldVertices[w + 1] = wy;
106  }
107  } else {
108  float[] deform = deformArray.Items;
109  for (int w = offset, b = skip * 3, f = skip << 1; w < count; w += stride) {
110  float wx = 0, wy = 0;
111  int n = bones[v++];
112  n += v;
113  for (; v < n; v++, b += 3, f += 2) {
114  Bone bone = skeletonBones[bones[v]];
115  float vx = vertices[b] + deform[f], vy = vertices[b + 1] + deform[f + 1], weight = vertices[b + 2];
116  wx += (vx * bone.a + vy * bone.b + bone.worldX) * weight;
117  wy += (vx * bone.c + vy * bone.d + bone.worldY) * weight;
118  }
119  worldVertices[w] = wx;
120  worldVertices[w + 1] = wy;
121  }
122  }
123  }
