76  public void Update (float delta) {
77  delta *= timeScale;
78  var tracksItems = tracks.Items;
79  for (int i = 0, n = tracks.Count; i < n; i++) {
80  TrackEntry current = tracksItems[i];
81  if (current == null) continue;
82 
83  current.animationLast = current.nextAnimationLast;
84  current.trackLast = current.nextTrackLast;
85 
86  float currentDelta = delta * current.timeScale;
87 
88  if (current.delay > 0) {
89  current.delay -= currentDelta;
90  if (current.delay > 0) continue;
91  currentDelta = -current.delay;
92  current.delay = 0;
93  }
94 
95  TrackEntry next = current.next;
96  if (next != null) {
97  // When the next entry's delay is passed, change to the next entry, preserving leftover time.
98  float nextTime = current.trackLast - next.delay;
99  if (nextTime >= 0) {
100  next.delay = 0;
101  next.trackTime = nextTime + (delta * next.timeScale);
102  current.trackTime += currentDelta;
103  SetCurrent(i, next, true);
104  while (next.mixingFrom != null) {
105  next.mixTime += currentDelta;
106  next = next.mixingFrom;
107  }
108  continue;
109  }
110  } else if (current.trackLast >= current.trackEnd && current.mixingFrom == null) {
111  // Clear the track when there is no next entry, the track end time is reached, and there is no mixingFrom.
112  tracksItems[i] = null;
113 
114  queue.End(current);
115  DisposeNext(current);
116  continue;
117  }
118  if (current.mixingFrom != null && UpdateMixingFrom(current, delta)) {
119  // End mixing from entries once all have completed.
120  var from = current.mixingFrom;
121  current.mixingFrom = null;
122  while (from != null) {
123  queue.End(from);
124  from = from.mixingFrom;
125  }
126  }
127 
128  current.trackTime += currentDelta;
129  }
130 
131  queue.Drain();
132  }

135  private bool UpdateMixingFrom (TrackEntry to, float delta) {
136  TrackEntry from = to.mixingFrom;
137  if (from == null) return true;
138 
139  bool finished = UpdateMixingFrom(from, delta);
140 
141  from.animationLast = from.nextAnimationLast;
142  from.trackLast = from.nextTrackLast;
143 
144  // Require mixTime > 0 to ensure the mixing from entry was applied at least once.
145  if (to.mixTime > 0 && (to.mixTime >= to.mixDuration || to.timeScale == 0)) {
146  // Require totalAlpha == 0 to ensure mixing is complete, unless mixDuration == 0 (the transition is a single frame).
147  if (from.totalAlpha == 0 || to.mixDuration == 0) {
148  to.mixingFrom = from.mixingFrom;
149  to.interruptAlpha = from.interruptAlpha;
150  queue.End(from);
151  }
152  return finished;
153  }
154 
155  from.trackTime += delta * from.timeScale;
156  to.mixTime += delta * to.timeScale;
157  return false;
158  }

163  public bool Apply (Skeleton skeleton) {
164  if (skeleton == null) throw new ArgumentNullException("skeleton", "skeleton cannot be null.");
165  if (animationsChanged) AnimationsChanged();
166 
167  var events = this.events;
168 
169  bool applied = false;
170  var tracksItems = tracks.Items;
171  for (int i = 0, m = tracks.Count; i < m; i++) {
172  TrackEntry current = tracksItems[i];
173  if (current == null || current.delay > 0) continue;
174  applied = true;
175  MixPose currentPose = i == 0 ? MixPose.Current : MixPose.CurrentLayered;
176 
177  // Apply mixing from entries first.
178  float mix = current.alpha;
179  if (current.mixingFrom != null)
180  mix *= ApplyMixingFrom(current, skeleton, currentPose);
181  else if (current.trackTime >= current.trackEnd && current.next == null) //
182  mix = 0; // Set to setup pose the last time the entry will be applied.
183 
184  // Apply current entry.
185  float animationLast = current.animationLast, animationTime = current.AnimationTime;
186  int timelineCount = current.animation.timelines.Count;
187  var timelines = current.animation.timelines;
188  var timelinesItems = timelines.Items;
189  if (mix == 1) {
190  for (int ii = 0; ii < timelineCount; ii++)
191  timelinesItems[ii].Apply(skeleton, animationLast, animationTime, events, 1, MixPose.Setup, MixDirection.In);
192  } else {
193  var timelineData = current.timelineData.Items;
194 
195  bool firstFrame = current.timelinesRotation.Count == 0;
196  if (firstFrame) current.timelinesRotation.EnsureCapacity(timelines.Count << 1);
197  var timelinesRotation = current.timelinesRotation.Items;
198 
199  for (int ii = 0; ii < timelineCount; ii++) {
200  Timeline timeline = timelinesItems[ii];
201  MixPose pose = timelineData[ii] >= AnimationState.First ? MixPose.Setup : currentPose;
202  var rotateTimeline = timeline as RotateTimeline;
203  if (rotateTimeline != null)
204  ApplyRotateTimeline(rotateTimeline, skeleton, animationTime, mix, pose, timelinesRotation, ii << 1, firstFrame);
205  else
206  timeline.Apply(skeleton, animationLast, animationTime, events, mix, pose, MixDirection.In);
207  }
208  }
209  QueueEvents(current, animationTime);
210  events.Clear(false);
211  current.nextAnimationLast = animationTime;
212  current.nextTrackLast = current.trackTime;
213  }
214 
215  queue.Drain();
216  return applied;
217  }

219  private float ApplyMixingFrom (TrackEntry to, Skeleton skeleton, MixPose currentPose) {
220  TrackEntry from = to.mixingFrom;
221  if (from.mixingFrom != null) ApplyMixingFrom(from, skeleton, currentPose);
222 
223  float mix;
224  if (to.mixDuration == 0) { // Single frame mix to undo mixingFrom changes.
225  mix = 1;
226  currentPose = MixPose.Setup;
227  } else {
228  mix = to.mixTime / to.mixDuration;
229  if (mix > 1) mix = 1;
230  }
231 
232  var eventBuffer = mix < from.eventThreshold ? this.events : null;
233  bool attachments = mix < from.attachmentThreshold, drawOrder = mix < from.drawOrderThreshold;
234  float animationLast = from.animationLast, animationTime = from.AnimationTime;
235  var timelines = from.animation.timelines;
236  int timelineCount = timelines.Count;
237  var timelinesItems = timelines.Items;
238  var timelineData = from.timelineData.Items;
239  var timelineDipMix = from.timelineDipMix.Items;
240 
241  bool firstFrame = from.timelinesRotation.Count == 0;
242  if (firstFrame) from.timelinesRotation.Resize(timelines.Count << 1); // from.timelinesRotation.setSize
243  var timelinesRotation = from.timelinesRotation.Items;
244 
245  MixPose pose;
246  float alphaDip = from.alpha * to.interruptAlpha, alphaMix = alphaDip * (1 - mix), alpha;
247  from.totalAlpha = 0;
248  for (int i = 0; i < timelineCount; i++) {
249  Timeline timeline = timelinesItems[i];
250  switch (timelineData[i]) {
251  case Subsequent:
252  if (!attachments && timeline is AttachmentTimeline) continue;
253  if (!drawOrder && timeline is DrawOrderTimeline) continue;
254  pose = currentPose;
255  alpha = alphaMix;
256  break;
257  case First:
258  pose = MixPose.Setup;
259  alpha = alphaMix;
260  break;
261  case Dip:
262  pose = MixPose.Setup;
263  alpha = alphaDip;
264  break;
265  default:
266  pose = MixPose.Setup;
267  TrackEntry dipMix = timelineDipMix[i];
268  alpha = alphaDip * Math.Max(0, 1 - dipMix.mixTime / dipMix.mixDuration);
269  break;
270  }
271  from.totalAlpha += alpha;
272  var rotateTimeline = timeline as RotateTimeline;
273  if (rotateTimeline != null) {
274  ApplyRotateTimeline(rotateTimeline, skeleton, animationTime, alpha, pose, timelinesRotation, i << 1, firstFrame);
275  } else {
276  timeline.Apply(skeleton, animationLast, animationTime, eventBuffer, alpha, pose, MixDirection.Out);
277  }
278  }
279 
280  if (to.mixDuration > 0) QueueEvents(from, animationTime);
281  this.events.Clear(false);
282  from.nextAnimationLast = animationTime;
283  from.nextTrackLast = from.trackTime;
284 
285  return mix;
286  }

288  static private void ApplyRotateTimeline (RotateTimeline rotateTimeline, Skeleton skeleton, float time, float alpha, MixPose pose,
289  float[] timelinesRotation, int i, bool firstFrame) {
290 
291  if (firstFrame) timelinesRotation[i] = 0;
292 
293  if (alpha == 1) {
294  rotateTimeline.Apply(skeleton, 0, time, null, 1, pose, MixDirection.In);
295  return;
296  }
297 
298  Bone bone = skeleton.bones.Items[rotateTimeline.boneIndex];
299  float[] frames = rotateTimeline.frames;
300  if (time < frames[0]) {
301  if (pose == MixPose.Setup) bone.rotation = bone.data.rotation;
302  return;
303  }
304 
305  float r2;
306  if (time >= frames[frames.Length - RotateTimeline.ENTRIES]) // Time is after last frame.
307  r2 = bone.data.rotation + frames[frames.Length + RotateTimeline.PREV_ROTATION];
308  else {
309  // Interpolate between the previous frame and the current frame.
310  int frame = Animation.BinarySearch(frames, time, RotateTimeline.ENTRIES);
311  float prevRotation = frames[frame + RotateTimeline.PREV_ROTATION];
312  float frameTime = frames[frame];
313  float percent = rotateTimeline.GetCurvePercent((frame >> 1) - 1,
314  1 - (time - frameTime) / (frames[frame + RotateTimeline.PREV_TIME] - frameTime));
315 
316  r2 = frames[frame + RotateTimeline.ROTATION] - prevRotation;
317  r2 -= (16384 - (int)(16384.499999999996 - r2 / 360)) * 360;
318  r2 = prevRotation + r2 * percent + bone.data.rotation;
319  r2 -= (16384 - (int)(16384.499999999996 - r2 / 360)) * 360;
320  }
321 
322  // Mix between rotations using the direction of the shortest route on the first frame while detecting crosses.
323  float r1 = pose == MixPose.Setup ? bone.data.rotation : bone.rotation;
324  float total, diff = r2 - r1;
325  if (diff == 0) {
326  total = timelinesRotation[i];
327  } else {
328  diff -= (16384 - (int)(16384.499999999996 - diff / 360)) * 360;
329  float lastTotal, lastDiff;
330  if (firstFrame) {
331  lastTotal = 0;
332  lastDiff = diff;
333  } else {
334  lastTotal = timelinesRotation[i]; // Angle and direction of mix, including loops.
335  lastDiff = timelinesRotation[i + 1]; // Difference between bones.
336  }
337  bool current = diff > 0, dir = lastTotal >= 0;
338  // Detect cross at 0 (not 180).
339  if (Math.Sign(lastDiff) != Math.Sign(diff) && Math.Abs(lastDiff) <= 90) {
340  // A cross after a 360 rotation is a loop.
341  if (Math.Abs(lastTotal) > 180) lastTotal += 360 * Math.Sign(lastTotal);
342  dir = current;
343  }
344  total = diff + lastTotal - lastTotal % 360; // Store loops as part of lastTotal.
345  if (dir != current) total += 360 * Math.Sign(lastTotal);
346  timelinesRotation[i] = total;
347  }
348  timelinesRotation[i + 1] = diff;
349  r1 += total * alpha;
350  bone.rotation = r1 - (16384 - (int)(16384.499999999996 - r1 / 360)) * 360;
351  }

353  private void QueueEvents (TrackEntry entry, float animationTime) {
354  float animationStart = entry.animationStart, animationEnd = entry.animationEnd;
355  float duration = animationEnd - animationStart;
356  float trackLastWrapped = entry.trackLast % duration;
357 
358  // Queue events before complete.
359  var events = this.events;
360  var eventsItems = events.Items;
361  int i = 0, n = events.Count;
362  for (; i < n; i++) {
363  var e = eventsItems[i];
364  if (e.time < trackLastWrapped) break;
365  if (e.time > animationEnd) continue; // Discard events outside animation start/end.
366  queue.Event(entry, e);
367  }
368 
369  // Queue complete if completed a loop iteration or the animation.
370  bool complete = false;
371  if (entry.loop)
372  complete = duration == 0 || (trackLastWrapped > entry.trackTime % duration);
373  else
374  complete = animationTime >= animationEnd && entry.animationLast < animationEnd;
375  if (complete) queue.Complete(entry);
376 
377  // Queue events after complete.
378  for (; i < n; i++) {
379  Event e = eventsItems[i];
380  if (e.time < animationStart) continue; // Discard events outside animation start/end.
381  queue.Event(entry, eventsItems[i]);
382  }
383  }

404  public void ClearTrack (int trackIndex) {
405  if (trackIndex >= tracks.Count) return;
406  TrackEntry current = tracks.Items[trackIndex];
407  if (current == null) return;

409  queue.End(current);

411  DisposeNext(current);

413  TrackEntry entry = current;
414  while (true) {
415  TrackEntry from = entry.mixingFrom;

417  queue.End(from);
418  entry.mixingFrom = null;

428  private void SetCurrent (int index, TrackEntry current, bool interrupt) {
429  TrackEntry from = ExpandToIndex(index);
430  tracks.Items[index] = current;
431 
432  if (from != null) {
433  if (interrupt) queue.Interrupt(from);
434  current.mixingFrom = from;
435  current.mixTime = 0;
436 
437  // Store interrupted mix percentage.
438  if (from.mixingFrom != null && from.mixDuration > 0)
439  current.interruptAlpha *= Math.Min(1, from.mixTime / from.mixDuration);
440 
441  from.timelinesRotation.Clear(); // Reset rotation for mixing out, in case entry was mixed in.
442  }
443 
444  queue.Start(current); // triggers AnimationsChanged
445  }

464  bool interrupt = true;

501  public TrackEntry AddAnimation (int trackIndex, Animation animation, bool loop, float delay) {
502  if (animation == null) throw new ArgumentNullException("animation", "animation cannot be null.");
503 
504  TrackEntry last = ExpandToIndex(trackIndex);
505  if (last != null) {
506  while (last.next != null)
507  last = last.next;
508  }
509 
510  TrackEntry entry = NewTrackEntry(trackIndex, animation, loop, last);
511 
512  if (last == null) {
513  SetCurrent(trackIndex, entry, true);
514  queue.Drain();
515  } else {
516  last.next = entry;
517  if (delay <= 0) {
518  float duration = last.animationEnd - last.animationStart;
519  if (duration != 0) {
520  if (last.loop) {
521  delay += duration * (1 + (int)(last.trackTime / duration));
522  } else {
523  delay += duration;
524  }
525  delay -= data.GetMix(last.animation, animation);
526  } else
527  delay = 0;
528  }
529  }
530 
531  entry.delay = delay;
532  return entry;
533  }

696  internal TrackEntry SetTimelineData (TrackEntry to, ExposedList<TrackEntry> mixingToArray, HashSet<int> propertyIDs) {

964  internal void Drain () {
965  if (drainDisabled) return;
966  drainDisabled = true;
967 
968  var entries = this.eventQueueEntries;
969  AnimationState state = this.state;
970 
971  // Don't cache entries.Count so callbacks can queue their own events (eg, call SetAnimation in AnimationState_Complete).
972  for (int i = 0; i < entries.Count; i++) {
973  var queueEntry = entries[i];
974  TrackEntry trackEntry = queueEntry.entry;
975 
976  switch (queueEntry.type) {
977  case EventType.Start:
978  trackEntry.OnStart();
979  state.OnStart(trackEntry);
980  break;
981  case EventType.Interrupt:
982  trackEntry.OnInterrupt();
983  state.OnInterrupt(trackEntry);
984  break;
985  case EventType.End:
986  trackEntry.OnEnd();
987  state.OnEnd(trackEntry);
988  goto case EventType.Dispose; // Fall through. (C#)
989  case EventType.Dispose:
990  trackEntry.OnDispose();
991  state.OnDispose(trackEntry);
992  trackEntryPool.Free(trackEntry); // Pooling
993  break;
994  case EventType.Complete:
995  trackEntry.OnComplete();
996  state.OnComplete(trackEntry);
997  break;
998  case EventType.Event:
999  trackEntry.OnEvent(queueEntry.e);
1000  state.OnEvent(trackEntry, queueEntry.e);
1001  break;
1002  }
1003  }
1004  eventQueueEntries.Clear();
1005 
1006  drainDisabled = false;
1007  }
