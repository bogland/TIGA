85  static public void Apply (Bone bone, float targetX, float targetY, float alpha) {
86  if (!bone.appliedValid) bone.UpdateAppliedTransform();
87  Bone p = bone.parent;
88  float id = 1 / (p.a * p.d - p.b * p.c);
89  float x = targetX - p.worldX, y = targetY - p.worldY;
90  float tx = (x * p.d - y * p.b) * id - bone.ax, ty = (y * p.a - x * p.c) * id - bone.ay;
91  float rotationIK = (float)Math.Atan2(ty, tx) * MathUtils.RadDeg - bone.ashearX - bone.arotation;
92  if (bone.ascaleX < 0) rotationIK += 180;
93  if (rotationIK > 180)
94  rotationIK -= 360;
95  else if (rotationIK < -180) rotationIK += 360;
96  bone.UpdateWorldTransform(bone.ax, bone.ay, bone.arotation + rotationIK * alpha, bone.ascaleX, bone.ascaleY, bone.ashearX,
97  bone.ashearY);
98  }

103  static public void Apply (Bone parent, Bone child, float targetX, float targetY, int bendDir, float alpha) {
104  if (alpha == 0) {
105  child.UpdateWorldTransform ();
106  return;
107  }
108  //float px = parent.x, py = parent.y, psx = parent.scaleX, psy = parent.scaleY, csx = child.scaleX;
109  if (!parent.appliedValid) parent.UpdateAppliedTransform();
110  if (!child.appliedValid) child.UpdateAppliedTransform();
111  float px = parent.ax, py = parent.ay, psx = parent.ascaleX, psy = parent.ascaleY, csx = child.ascaleX;
112  int os1, os2, s2;
113  if (psx < 0) {
114  psx = -psx;
115  os1 = 180;
116  s2 = -1;
117  } else {
118  os1 = 0;
119  s2 = 1;
120  }
121  if (psy < 0) {
122  psy = -psy;
123  s2 = -s2;
124  }
125  if (csx < 0) {
126  csx = -csx;
127  os2 = 180;
128  } else
129  os2 = 0;
130  float cx = child.ax, cy, cwx, cwy, a = parent.a, b = parent.b, c = parent.c, d = parent.d;
131  bool u = Math.Abs(psx - psy) <= 0.0001f;
132  if (!u) {
133  cy = 0;
134  cwx = a * cx + parent.worldX;
135  cwy = c * cx + parent.worldY;
136  } else {
137  cy = child.ay;
138  cwx = a * cx + b * cy + parent.worldX;
139  cwy = c * cx + d * cy + parent.worldY;
140  }
141  Bone pp = parent.parent;
142  a = pp.a;
143  b = pp.b;
144  c = pp.c;
145  d = pp.d;
146  float id = 1 / (a * d - b * c), x = targetX - pp.worldX, y = targetY - pp.worldY;
147  float tx = (x * d - y * b) * id - px, ty = (y * a - x * c) * id - py;
148  x = cwx - pp.worldX;
149  y = cwy - pp.worldY;
150  float dx = (x * d - y * b) * id - px, dy = (y * a - x * c) * id - py;
151  float l1 = (float)Math.Sqrt(dx * dx + dy * dy), l2 = child.data.length * csx, a1, a2;
152  if (u) {
153  l2 *= psx;
154  float cos = (tx * tx + ty * ty - l1 * l1 - l2 * l2) / (2 * l1 * l2);
155  if (cos < -1)
156  cos = -1;
157  else if (cos > 1) cos = 1;
158  a2 = (float)Math.Acos(cos) * bendDir;
159  a = l1 + l2 * cos;
160  b = l2 * (float)Math.Sin(a2);
161  a1 = (float)Math.Atan2(ty * a - tx * b, tx * a + ty * b);
162  } else {
163  a = psx * l2;
164  b = psy * l2;
165  float aa = a * a, bb = b * b, dd = tx * tx + ty * ty, ta = (float)Math.Atan2(ty, tx);
166  c = bb * l1 * l1 + aa * dd - aa * bb;
167  float c1 = -2 * bb * l1, c2 = bb - aa;
168  d = c1 * c1 - 4 * c2 * c;
169  if (d >= 0) {
170  float q = (float)Math.Sqrt(d);
171  if (c1 < 0) q = -q;
172  q = -(c1 + q) / 2;
173  float r0 = q / c2, r1 = c / q;
174  float r = Math.Abs(r0) < Math.Abs(r1) ? r0 : r1;
175  if (r * r <= dd) {
176  y = (float)Math.Sqrt(dd - r * r) * bendDir;
177  a1 = ta - (float)Math.Atan2(y, r);
178  a2 = (float)Math.Atan2(y / psy, (r - l1) / psx);
179  goto break_outer; // break outer;
180  }
181  }
182  float minAngle = MathUtils.PI, minX = l1 - a, minDist = minX * minX, minY = 0;
183  float maxAngle = 0, maxX = l1 + a, maxDist = maxX * maxX, maxY = 0;
184  c = -a * l1 / (aa - bb);
185  if (c >= -1 && c <= 1) {
186  c = (float)Math.Acos(c);
187  x = a * (float)Math.Cos(c) + l1;
188  y = b * (float)Math.Sin(c);
189  d = x * x + y * y;
190  if (d < minDist) {
191  minAngle = c;
192  minDist = d;
193  minX = x;
194  minY = y;
195  }
196  if (d > maxDist) {
197  maxAngle = c;
198  maxDist = d;
199  maxX = x;
200  maxY = y;
201  }
202  }
203  if (dd <= (minDist + maxDist) / 2) {
204  a1 = ta - (float)Math.Atan2(minY * bendDir, minX);
205  a2 = minAngle * bendDir;
206  } else {
207  a1 = ta - (float)Math.Atan2(maxY * bendDir, maxX);
208  a2 = maxAngle * bendDir;
209  }
210  }
211  break_outer:
212  float os = (float)Math.Atan2(cy, cx) * s2;
213  float rotation = parent.arotation;
214  a1 = (a1 - os) * MathUtils.RadDeg + os1 - rotation;
215  if (a1 > 180)
216  a1 -= 360;
217  else if (a1 < -180) a1 += 360;
218  parent.UpdateWorldTransform(px, py, rotation + a1 * alpha, parent.scaleX, parent.ascaleY, 0, 0);
219  rotation = child.arotation;
220  a2 = ((a2 + os) * MathUtils.RadDeg - child.ashearX) * s2 + os2 - rotation;
221  if (a2 > 180)
222  a2 -= 360;
223  else if (a2 < -180) a2 += 360;
224  child.UpdateWorldTransform(cx, cy, rotation + a2 * alpha, child.ascaleX, child.ascaleY, child.ashearX, child.ashearY);
225  }
226  }
