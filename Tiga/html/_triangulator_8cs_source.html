35  private readonly ExposedList<ExposedList<float>> convexPolygons = new ExposedList<ExposedList<float>>();
36  private readonly ExposedList<ExposedList<int>> convexPolygonsIndices = new ExposedList<ExposedList<int>>();

38  private readonly ExposedList<int> indicesArray = new ExposedList<int>();
39  private readonly ExposedList<bool> isConcaveArray = new ExposedList<bool>();
40  private readonly ExposedList<int> triangles = new ExposedList<int>();

42  private readonly Pool<ExposedList<float>> polygonPool = new Pool<ExposedList<float>>();
43  private readonly Pool<ExposedList<int>> polygonIndicesPool = new Pool<ExposedList<int>>();

45  public ExposedList<int> Triangulate (ExposedList<float> verticesArray) {
46  var vertices = verticesArray.Items;
47  int vertexCount = verticesArray.Count >> 1;
48 
49  var indicesArray = this.indicesArray;
50  indicesArray.Clear();
51  int[] indices = indicesArray.Resize(vertexCount).Items;
52  for (int i = 0; i < vertexCount; i++)
53  indices[i] = i;
54 
55  var isConcaveArray = this.isConcaveArray;
56  bool[] isConcave = isConcaveArray.Resize(vertexCount).Items;
57  for (int i = 0, n = vertexCount; i < n; ++i)
58  isConcave[i] = IsConcave(i, vertexCount, vertices, indices);
59 
60  var triangles = this.triangles;
61  triangles.Clear();
62  triangles.EnsureCapacity(Math.Max(0, vertexCount - 2) << 2);
63 
64  while (vertexCount > 3) {
65  // Find ear tip.
66  int previous = vertexCount - 1, i = 0, next = 1;
67 
68  // outer:
69  while (true) {
70  if (!isConcave[i]) {
71  int p1 = indices[previous] << 1, p2 = indices[i] << 1, p3 = indices[next] << 1;
72  float p1x = vertices[p1], p1y = vertices[p1 + 1];
73  float p2x = vertices[p2], p2y = vertices[p2 + 1];
74  float p3x = vertices[p3], p3y = vertices[p3 + 1];
75  for (int ii = (next + 1) % vertexCount; ii != previous; ii = (ii + 1) % vertexCount) {
76  if (!isConcave[ii]) continue;
77  int v = indices[ii] << 1;
78  float vx = vertices[v], vy = vertices[v + 1];
79  if (PositiveArea(p3x, p3y, p1x, p1y, vx, vy)) {
80  if (PositiveArea(p1x, p1y, p2x, p2y, vx, vy)) {
81  if (PositiveArea(p2x, p2y, p3x, p3y, vx, vy)) goto break_outer; // break outer;
82  }
83  }
84  }
85  break;
86  }
87  break_outer:
88 
89  if (next == 0) {
90  do {
91  if (!isConcave[i]) break;
92  i--;
93  } while (i > 0);
94  break;
95  }
96 
97  previous = i;
98  i = next;
99  next = (next + 1) % vertexCount;
100  }
101 
102  // Cut ear tip.
103  triangles.Add(indices[(vertexCount + i - 1) % vertexCount]);
104  triangles.Add(indices[i]);
105  triangles.Add(indices[(i + 1) % vertexCount]);
106  indicesArray.RemoveAt(i);
107  isConcaveArray.RemoveAt(i);
108  vertexCount--;
109 
110  int previousIndex = (vertexCount + i - 1) % vertexCount;
111  int nextIndex = i == vertexCount ? 0 : i;
112  isConcave[previousIndex] = IsConcave(previousIndex, vertexCount, vertices, indices);
113  isConcave[nextIndex] = IsConcave(nextIndex, vertexCount, vertices, indices);
114  }
115 
116  if (vertexCount == 3) {
117  triangles.Add(indices[2]);
118  triangles.Add(indices[0]);
119  triangles.Add(indices[1]);
120  }
121 
122  return triangles;
123  }

125  public ExposedList<ExposedList<float>> Decompose (ExposedList<float> verticesArray, ExposedList<int> triangles) {
126  var vertices = verticesArray.Items;
127  var convexPolygons = this.convexPolygons;
128  for (int i = 0, n = convexPolygons.Count; i < n; i++) {
129  polygonPool.Free(convexPolygons.Items[i]);
130  }
131  convexPolygons.Clear();
132 
133  var convexPolygonsIndices = this.convexPolygonsIndices;
134  for (int i = 0, n = convexPolygonsIndices.Count; i < n; i++) {
135  polygonIndicesPool.Free(convexPolygonsIndices.Items[i]);
136  }
137  convexPolygonsIndices.Clear();
138 
139  var polygonIndices = polygonIndicesPool.Obtain();
140  polygonIndices.Clear();
141 
142  var polygon = polygonPool.Obtain();
143  polygon.Clear();
144 
145  // Merge subsequent triangles if they form a triangle fan.
146  int fanBaseIndex = -1, lastWinding = 0;
147  int[] trianglesItems = triangles.Items;
148  for (int i = 0, n = triangles.Count; i < n; i += 3) {
149  int t1 = trianglesItems[i] << 1, t2 = trianglesItems[i + 1] << 1, t3 = trianglesItems[i + 2] << 1;
150  float x1 = vertices[t1], y1 = vertices[t1 + 1];
151  float x2 = vertices[t2], y2 = vertices[t2 + 1];
152  float x3 = vertices[t3], y3 = vertices[t3 + 1];
153 
154  // If the base of the last triangle is the same as this triangle, check if they form a convex polygon (triangle fan).
155  var merged = false;
156  if (fanBaseIndex == t1) {
157  int o = polygon.Count - 4;
158  float[] p = polygon.Items;
159  int winding1 = Winding(p[o], p[o + 1], p[o + 2], p[o + 3], x3, y3);
160  int winding2 = Winding(x3, y3, p[0], p[1], p[2], p[3]);
161  if (winding1 == lastWinding && winding2 == lastWinding) {
162  polygon.Add(x3);
163  polygon.Add(y3);
164  polygonIndices.Add(t3);
165  merged = true;
166  }
167  }
168 
169  // Otherwise make this triangle the new base.
170  if (!merged) {
171  if (polygon.Count > 0) {
172  convexPolygons.Add(polygon);
173  convexPolygonsIndices.Add(polygonIndices);
174  } else {
175  polygonPool.Free(polygon);
176  polygonIndicesPool.Free(polygonIndices);
177  }
178  polygon = polygonPool.Obtain();
179  polygon.Clear();
180  polygon.Add(x1);
181  polygon.Add(y1);
182  polygon.Add(x2);
183  polygon.Add(y2);
184  polygon.Add(x3);
185  polygon.Add(y3);
186  polygonIndices = polygonIndicesPool.Obtain();
187  polygonIndices.Clear();
188  polygonIndices.Add(t1);
189  polygonIndices.Add(t2);
190  polygonIndices.Add(t3);
191  lastWinding = Winding(x1, y1, x2, y2, x3, y3);
192  fanBaseIndex = t1;
193  }
194  }
195 
196  if (polygon.Count > 0) {
197  convexPolygons.Add(polygon);
198  convexPolygonsIndices.Add(polygonIndices);
199  }
200 
201  // Go through the list of polygons and try to merge the remaining triangles with the found triangle fans.
202  for (int i = 0, n = convexPolygons.Count; i < n; i++) {
203  polygonIndices = convexPolygonsIndices.Items[i];
204  if (polygonIndices.Count == 0) continue;
205  int firstIndex = polygonIndices.Items[0];
206  int lastIndex = polygonIndices.Items[polygonIndices.Count - 1];
207 
208  polygon = convexPolygons.Items[i];
209  int o = polygon.Count - 4;
210  float[] p = polygon.Items;
211  float prevPrevX = p[o], prevPrevY = p[o + 1];
212  float prevX = p[o + 2], prevY = p[o + 3];
213  float firstX = p[0], firstY = p[1];
214  float secondX = p[2], secondY = p[3];
215  int winding = Winding(prevPrevX, prevPrevY, prevX, prevY, firstX, firstY);
216 
217  for (int ii = 0; ii < n; ii++) {
218  if (ii == i) continue;
219  var otherIndices = convexPolygonsIndices.Items[ii];
220  if (otherIndices.Count != 3) continue;
221  int otherFirstIndex = otherIndices.Items[0];
222  int otherSecondIndex = otherIndices.Items[1];
223  int otherLastIndex = otherIndices.Items[2];
224 
225  var otherPoly = convexPolygons.Items[ii];
226  float x3 = otherPoly.Items[otherPoly.Count - 2], y3 = otherPoly.Items[otherPoly.Count - 1];
227 
228  if (otherFirstIndex != firstIndex || otherSecondIndex != lastIndex) continue;
229  int winding1 = Winding(prevPrevX, prevPrevY, prevX, prevY, x3, y3);
230  int winding2 = Winding(x3, y3, firstX, firstY, secondX, secondY);
231  if (winding1 == winding && winding2 == winding) {
232  otherPoly.Clear();
233  otherIndices.Clear();
234  polygon.Add(x3);
235  polygon.Add(y3);
236  polygonIndices.Add(otherLastIndex);
237  prevPrevX = prevX;
238  prevPrevY = prevY;
239  prevX = x3;
240  prevY = y3;
241  ii = 0;
242  }
243  }
244  }
245 
246  // Remove empty polygons that resulted from the merge step above.
247  for (int i = convexPolygons.Count - 1; i >= 0; i--) {
248  polygon = convexPolygons.Items[i];
249  if (polygon.Count == 0) {
250  convexPolygons.RemoveAt(i);
251  polygonPool.Free(polygon);
252  polygonIndices = convexPolygonsIndices.Items[i];
253  convexPolygonsIndices.RemoveAt(i);
254  polygonIndicesPool.Free(polygonIndices);
255  }
256  }
257 
258  return convexPolygons;
259  }

261  static private bool IsConcave (int index, int vertexCount, float[] vertices, int[] indices) {
262  int previous = indices[(vertexCount + index - 1) % vertexCount] << 1;
263  int current = indices[index] << 1;
264  int next = indices[(index + 1) % vertexCount] << 1;
265  return !PositiveArea(vertices[previous], vertices[previous + 1], vertices[current], vertices[current + 1], vertices[next],
266  vertices[next + 1]);
267  }
268 
269  static private bool PositiveArea (float p1x, float p1y, float p2x, float p2y, float p3x, float p3y) {
270  return p1x * (p3y - p2y) + p2x * (p1y - p3y) + p3x * (p2y - p1y) >= 0;
271  }
272 
273  static private int Winding (float p1x, float p1y, float p2x, float p2y, float p3x, float p3y) {
274  float px = p2x - p1x, py = p2y - p1y;
275  return p3x * py - p3y * px + px * p1y - p1x * py >= 0 ? 1 : -1;
276  }
277  }
