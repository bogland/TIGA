126  [MenuItem("CONTEXT/SkeletonGraphic/Match RectTransform with Mesh Bounds")]
127  static void MatchRectTransformWithBounds (MenuCommand command) {
128  var skeletonGraphic = (SkeletonGraphic)command.context;
129  Mesh mesh = skeletonGraphic.GetLastMesh();
130  if (mesh == null) {
131  Debug.Log("Mesh was not previously generated.");
132  return;
133  }
134 
135  if (mesh.vertexCount == 0) {
136  skeletonGraphic.rectTransform.sizeDelta = new Vector2(50f, 50f);
137  skeletonGraphic.rectTransform.pivot = new Vector2(0.5f, 0.5f);
138  return;
139  }
140 
141  mesh.RecalculateBounds();
142  var bounds = mesh.bounds;
143  var size = bounds.size;
144  var center = bounds.center;
145  var p = new Vector2(
146  0.5f - (center.x / size.x),
147  0.5f - (center.y / size.y)
148  );
149 
150  skeletonGraphic.rectTransform.sizeDelta = size;
151  skeletonGraphic.rectTransform.pivot = p;
152  }

178  public static SkeletonGraphic InstantiateSkeletonGraphic (SkeletonDataAsset skeletonDataAsset, Skin skin = null) {
179  string spineGameObjectName = string.Format("SkeletonGraphic ({0})", skeletonDataAsset.name.Replace("_SkeletonData", ""));
180  var go = NewSkeletonGraphicGameObject(spineGameObjectName);
181  var graphic = go.GetComponent<SkeletonGraphic>();
182  graphic.skeletonDataAsset = skeletonDataAsset;
183 
184  SkeletonData data = skeletonDataAsset.GetSkeletonData(true);
185 
186  if (data == null) {
187  for (int i = 0; i < skeletonDataAsset.atlasAssets.Length; i++) {
188  string reloadAtlasPath = AssetDatabase.GetAssetPath(skeletonDataAsset.atlasAssets[i]);
189  skeletonDataAsset.atlasAssets[i] = (AtlasAsset)AssetDatabase.LoadAssetAtPath(reloadAtlasPath, typeof(AtlasAsset));
190  }
191 
192  data = skeletonDataAsset.GetSkeletonData(true);
193  }
194 
195  skin = skin ?? data.DefaultSkin ?? data.Skins.Items[0];
196  graphic.MeshGenerator.settings.zSpacing = SpineEditorUtilities.defaultZSpacing;
197 
198  graphic.Initialize(false);
199  if (skin != null) graphic.Skeleton.SetSkin(skin);
200  graphic.initialSkinName = skin.Name;
201  graphic.Skeleton.UpdateWorldTransform();
202  graphic.UpdateMesh();
203 
204  return graphic;
205  }

214  public static Material DefaultSkeletonGraphicMaterial {
215  get {
216  var guids = AssetDatabase.FindAssets("SkeletonGraphicDefault t:material");
217  if (guids.Length <= 0) return null;
218 
219  var firstAssetPath = AssetDatabase.GUIDToAssetPath(guids[0]);
220  if (string.IsNullOrEmpty(firstAssetPath)) return null;
221 
222  var firstMaterial = AssetDatabase.LoadAssetAtPath<Material>(firstAssetPath);
223  return firstMaterial;
224  }
