229  public struct Hierarchy {
230  public string skin;
231  public string slot;
232  public string name;
233 
234  public Hierarchy (string fullPath) {
235  string[] chunks = fullPath.Split(new char[]{'/'}, System.StringSplitOptions.RemoveEmptyEntries);
236  if (chunks.Length == 0) {
237  skin = "";
238  slot = "";
239  name = "";
240  return;
241  }
242  else if (chunks.Length < 2) {
243  throw new System.Exception("Cannot generate Attachment Hierarchy from string! Not enough components! [" + fullPath + "]");
244  }
245  skin = chunks[0];
246  slot = chunks[1];
247  name = "";
248  for (int i = 2; i < chunks.Length; i++) {
249  name += chunks[i];
250  }
251  }
252  }
253  }
