58  void OnEnable () {
59  mode = this.serializedObject.FindProperty("mode");

70  skeletonUtility.skeletonRenderer.Initialize(false);
71 
72  canCreateHingeChain = CanCreateHingeChain();
73  boundingBoxTable.Clear();
74 
75  if (multiObject) return;
76  if (utilityBone.bone == null) return;
77 
78  var skeleton = utilityBone.bone.Skeleton;
79  int slotCount = skeleton.Slots.Count;
80  Skin skin = skeleton.Skin;
81  if (skeleton.Skin == null)
82  skin = skeleton.Data.DefaultSkin;
83 
84  //currentSkinName = skin.Name;
85  for(int i = 0; i < slotCount; i++){
86  Slot slot = skeletonUtility.skeletonRenderer.skeleton.Slots.Items[i];
87  if (slot.Bone == utilityBone.bone) {
88  var slotAttachments = new List<Attachment>();
89  skin.FindAttachmentsForSlot(skeleton.FindSlotIndex(slot.Data.Name), slotAttachments);
90  var boundingBoxes = new List<BoundingBoxAttachment>();
91  foreach (var att in slotAttachments) {
92  var boundingBoxAttachment = att as BoundingBoxAttachment;
93  if (boundingBoxAttachment != null)
94  boundingBoxes.Add(boundingBoxAttachment);
95  }
96 
97  if (boundingBoxes.Count > 0)
98  boundingBoxTable.Add(slot, boundingBoxes);
99  }
100  }

104  void EvaluateFlags () {
105  utilityBone = (SkeletonUtilityBone)target;
106  skeletonUtility = utilityBone.skeletonUtility;
107 
108  if (Selection.objects.Length == 1) {
109  containsFollows = utilityBone.mode == SkeletonUtilityBone.Mode.Follow;
110  containsOverrides = utilityBone.mode == SkeletonUtilityBone.Mode.Override;
111  } else {
112  int boneCount = 0;
113  foreach (Object o in Selection.objects) {
114  var go = o as GameObject;
115  if (go != null) {
116  SkeletonUtilityBone sub = go.GetComponent<SkeletonUtilityBone>();
117  if (sub != null) {
118  boneCount++;
119  containsFollows |= (sub.mode == SkeletonUtilityBone.Mode.Follow);
120  containsOverrides |= (sub.mode == SkeletonUtilityBone.Mode.Override);
121  }
122  }
123  }
124 
125  multiObject |= (boneCount > 1);
126  }

292  void CreateHingeChain () {
293  var utilBoneArr = utilityBone.GetComponentsInChildren<SkeletonUtilityBone>();
294 
295  foreach (var utilBone in utilBoneArr) {
296  if (utilBone.GetComponent<Collider2D>() == null) {
297  if (utilBone.bone.Data.Length == 0) {
298  var sphere = utilBone.gameObject.AddComponent<CircleCollider2D>();
299  sphere.radius = 0.1f;
300  } else {
301  float length = utilBone.bone.Data.Length;
302  var box = utilBone.gameObject.AddComponent<BoxCollider2D>();
303  box.size = new Vector3(length, length / 3f, 0.2f);
304  box.offset = new Vector3(length / 2f, 0, 0);
305  }
306  }
307 
308  utilBone.gameObject.AddComponent<Rigidbody2D>();
309  }
310 
311  utilityBone.GetComponent<Rigidbody2D>().isKinematic = true;
312 
313  foreach (var utilBone in utilBoneArr) {
314  if (utilBone == utilityBone)
315  continue;
316 
317  utilBone.mode = SkeletonUtilityBone.Mode.Override;
318 
319  var joint = utilBone.gameObject.AddComponent<HingeJoint2D>();
320  joint.connectedBody = utilBone.transform.parent.GetComponent<Rigidbody2D>();
321  joint.useLimits = true;
322  joint.limits = new JointAngleLimits2D {
323  min = -20,
324  max = 20
325  };
326  utilBone.GetComponent<Rigidbody2D>().mass = utilBone.transform.parent.GetComponent<Rigidbody2D>().mass * 0.75f;
327  }
328  }
329 #else
330  bool CanCreateHingeChain () {
331  if (utilityBone == null)
332  return false;
333  if (utilityBone.GetComponent<Rigidbody>() != null)
334  return false;
335  if (utilityBone.bone != null && utilityBone.bone.Children.Count == 0)
336  return false;

370  static void AttachRigidbody (SkeletonUtilityBone utilBone) {
371  if (utilBone.GetComponent<Collider>() == null) {
372  if (utilBone.bone.Data.Length == 0) {
373  SphereCollider sphere = utilBone.gameObject.AddComponent<SphereCollider>();
374  sphere.radius = 0.1f;
375  } else {
376  float length = utilBone.bone.Data.Length;
377  BoxCollider box = utilBone.gameObject.AddComponent<BoxCollider>();
378  box.size = new Vector3(length, length / 3f, 0.2f);
379  box.center = new Vector3(length / 2f, 0, 0);
380  }
381  }
382 
383  utilBone.gameObject.AddComponent<Rigidbody>();
384  }
