63  public static AtlasAsset CreateRuntimeInstance (TextAsset atlasText, Texture2D[] textures, Material materialPropertySource, bool initialize) {
64  // Get atlas page names.
65  string atlasString = atlasText.text;
66  atlasString = atlasString.Replace("\r", "");
67  string[] atlasLines = atlasString.Split('\n');
68  var pages = new List<string>();
69  for (int i = 0; i < atlasLines.Length - 1; i++) {
70  if (atlasLines[i].Trim().Length == 0)
71  pages.Add(atlasLines[i + 1].Trim().Replace(".png", ""));
72  }
73 
74  // Populate Materials[] by matching texture names with page names.
75  var materials = new Material[pages.Count];
76  for (int i = 0, n = pages.Count; i < n; i++) {
77  Material mat = null;
78 
79  // Search for a match.
80  string pageName = pages[i];
81  for (int j = 0, m = textures.Length; j < m; j++) {
82  if (string.Equals(pageName, textures[j].name, System.StringComparison.OrdinalIgnoreCase)) {
83  // Match found.
84  mat = new Material(materialPropertySource);
85  mat.mainTexture = textures[j];
86  break;
87  }
88  }
89 
90  if (mat != null)
91  materials[i] = mat;
92  else
93  throw new ArgumentException("Could not find matching atlas page in the texture array.");
94  }
95 
96  // Create AtlasAsset normally
97  return CreateRuntimeInstance(atlasText, materials, initialize);
98  }

147  public Mesh GenerateMesh (string name, Mesh mesh, out Material material, float scale = 0.01f) {
148  AtlasRegion region = atlas.FindRegion(name);
149  material = null;
150  if (region != null) {
151  if (mesh == null) {
152  mesh = new Mesh();
153  mesh.name = name;
154  }
155 
156  Vector3[] verts = new Vector3[4];
157  Vector2[] uvs = new Vector2[4];
158  Color[] colors = { Color.white, Color.white, Color.white, Color.white };
159  int[] triangles = { 0, 1, 2, 2, 3, 0 };
160 
161  float left, right, top, bottom;
162  left = region.width / -2f;
163  right = left * -1f;
164  top = region.height / 2f;
165  bottom = top * -1;
166 
167  verts[0] = new Vector3(left, bottom, 0) * scale;
168  verts[1] = new Vector3(left, top, 0) * scale;
169  verts[2] = new Vector3(right, top, 0) * scale;
170  verts[3] = new Vector3(right, bottom, 0) * scale;
171  float u, v, u2, v2;
172  u = region.u;
173  v = region.v;
174  u2 = region.u2;
175  v2 = region.v2;
176 
177  if (!region.rotate) {
178  uvs[0] = new Vector2(u, v2);
179  uvs[1] = new Vector2(u, v);
180  uvs[2] = new Vector2(u2, v);
181  uvs[3] = new Vector2(u2, v2);
182  } else {
183  uvs[0] = new Vector2(u2, v2);
184  uvs[1] = new Vector2(u, v2);
185  uvs[2] = new Vector2(u, v);
186  uvs[3] = new Vector2(u2, v);
187  }
188 
189  mesh.triangles = new int[0];
190  mesh.vertices = verts;
191  mesh.uv = uvs;
192  mesh.colors = colors;
193  mesh.triangles = triangles;
194  mesh.RecalculateNormals();
195  mesh.RecalculateBounds();
196 
197  material = (Material)region.page.rendererObject;
198  } else {
199  mesh = null;
200  }
201 
202  return mesh;
203  }

213  public void Load (AtlasPage page, string path) {
214  String name = Path.GetFileNameWithoutExtension(path);
215  Material material = null;
216  foreach (Material other in atlasAsset.materials) {
217  if (other.mainTexture == null) {
218  Debug.LogError("Material is missing texture: " + other.name, other);
219  return;
220  }
221  if (other.mainTexture.name == name) {
222  material = other;
223  break;
224  }
225  }
226  if (material == null) {
227  Debug.LogError("Material with texture name \"" + name + "\" not found for atlas asset: " + atlasAsset.name, atlasAsset);
228  return;
229  }
230  page.rendererObject = material;
231 
232  // Very old atlas files expected the texture's actual size to be used at runtime.
233  if (page.width == 0 || page.height == 0) {
234  page.width = material.mainTexture.width;
235  page.height = material.mainTexture.height;
236  }
237  }
