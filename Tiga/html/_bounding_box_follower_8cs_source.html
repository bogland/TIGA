82  public void Initialize (bool overwrite = false) {
83  if (skeletonRenderer == null)
84  return;
85 
86  skeletonRenderer.Initialize(false);
87 
88  if (string.IsNullOrEmpty(slotName))
89  return;
90 
91  // Don't reinitialize if the setup did not change.
92  if (!overwrite
93  &&
94  colliderTable.Count > 0 && slot != null // Slot is set and colliders already populated.
95  &&
96  skeletonRenderer.skeleton == slot.Skeleton // Skeleton object did not change.
97  &&
98  slotName == slot.data.name // Slot object did not change.
99  )
100  return;
101 
102  DisposeColliders();
103 
104  var skeleton = skeletonRenderer.skeleton;
105  slot = skeleton.FindSlot(slotName);
106  int slotIndex = skeleton.FindSlotIndex(slotName);
107 
108  if (slot == null) {
109  if (BoundingBoxFollower.DebugMessages)
110  Debug.LogWarning(string.Format("Slot '{0}' not found for BoundingBoxFollower on '{1}'. (Previous colliders were disposed.)", slotName, this.gameObject.name));
111  return;
112  }
113 
114  if (this.gameObject.activeInHierarchy) {
115  foreach (var skin in skeleton.Data.Skins)
116  AddSkin(skin, slotIndex);
117 
118  if (skeleton.skin != null)
119  AddSkin(skeleton.skin, slotIndex);
120  }
121 
122  if (BoundingBoxFollower.DebugMessages) {
123  bool valid = colliderTable.Count != 0;
124  if (!valid) {
125  if (this.gameObject.activeInHierarchy)
126  Debug.LogWarning("Bounding Box Follower not valid! Slot [" + slotName + "] does not contain any Bounding Box Attachments!");
127  else
128  Debug.LogWarning("Bounding Box Follower tried to rebuild as a prefab.");
129  }
130  }
131  }
132 
133  void AddSkin (Skin skin, int slotIndex) {
134  if (skin == null) return;
135  var attachmentNames = new List<string>();
136  skin.FindNamesForSlot(slotIndex, attachmentNames);
137 
138  foreach (var skinKey in attachmentNames) {
139  var attachment = skin.GetAttachment(slotIndex, skinKey);
140  var boundingBoxAttachment = attachment as BoundingBoxAttachment;
141 
142  if (BoundingBoxFollower.DebugMessages && attachment != null && boundingBoxAttachment == null)
143  Debug.Log("BoundingBoxFollower tried to follow a slot that contains non-boundingbox attachments: " + slotName);
144 
145  if (boundingBoxAttachment != null) {
146  if (!colliderTable.ContainsKey(boundingBoxAttachment)) {
147  var bbCollider = SkeletonUtility.AddBoundingBoxAsComponent(boundingBoxAttachment, slot, gameObject, isTrigger);
148  bbCollider.enabled = false;
149  bbCollider.hideFlags = HideFlags.NotEditable;
150  bbCollider.isTrigger = IsTrigger;
151  colliderTable.Add(boundingBoxAttachment, bbCollider);
152  nameTable.Add(boundingBoxAttachment, skinKey);
153  }
154  }
155  }
156  }

173  void DisposeColliders () {
174  var colliders = GetComponents<PolygonCollider2D>();
175  if (colliders.Length == 0) return;
176 
177  if (Application.isEditor) {
178  if (Application.isPlaying) {
179  foreach (var c in colliders) {
180  if (c != null)
181  Destroy(c);
182  }
183  } else {
184  foreach (var c in colliders)
185  if (c != null)
186  DestroyImmediate(c);
187  }
188  } else {
189  foreach (PolygonCollider2D c in colliders)
190  if (c != null)
191  Destroy(c);
192  }
193 
194  slot = null;
195  currentAttachment = null;
196  currentAttachmentName = null;
197  currentCollider = null;
198  colliderTable.Clear();
199  nameTable.Clear();
200  }

209  void MatchAttachment (Attachment attachment) {
210  var bbAttachment = attachment as BoundingBoxAttachment;
211 
212  if (BoundingBoxFollower.DebugMessages && attachment != null && bbAttachment == null)
213  Debug.LogWarning("BoundingBoxFollower tried to match a non-boundingbox attachment. It will treat it as null.");
214 
215  if (currentCollider != null)
216  currentCollider.enabled = false;
217 
218  if (bbAttachment == null) {
219  currentCollider = null;
220  currentAttachment = null;
221  currentAttachmentName = null;
222  } else {
223  PolygonCollider2D foundCollider;
224  colliderTable.TryGetValue(bbAttachment, out foundCollider);
225  if (foundCollider != null) {
226  currentCollider = foundCollider;
227  currentCollider.enabled = true;
228  currentAttachment = bbAttachment;
229  currentAttachmentName = nameTable[bbAttachment];
230  } else {
231  currentCollider = null;
232  currentAttachment = bbAttachment;
233  currentAttachmentName = null;
234  if (BoundingBoxFollower.DebugMessages) Debug.LogFormat("Collider for BoundingBoxAttachment named '{0}' was not initialized. It is possibly from a new skin. currentAttachmentName will be null. You may need to call BoundingBoxFollower.Initialize(overwrite: true);", bbAttachment.Name);
235  }
236  }
237  }
