49  void Reset () {
50  skeletonRenderer = GetComponent<SkeletonRenderer>();
51 
52  // Populate atlas list
53  if (skeletonRenderer != null && skeletonRenderer.skeletonDataAsset != null) {
54  AtlasAsset[] atlasAssets = skeletonRenderer.skeletonDataAsset.atlasAssets;
55 
56  var initialAtlasMaterialOverrides = new List<AtlasMaterialOverride>();
57  foreach (AtlasAsset atlasAsset in atlasAssets) {
58  foreach (Material atlasMaterial in atlasAsset.materials) {
59  var atlasMaterialOverride = new AtlasMaterialOverride();
60  atlasMaterialOverride.overrideDisabled = true;
61  atlasMaterialOverride.originalMaterial = atlasMaterial;
62 
63  initialAtlasMaterialOverrides.Add(atlasMaterialOverride);
64  }
65  }
66 
67  customMaterialOverrides = initialAtlasMaterialOverrides;
68  }
69  }
70  #endif

73  void SetCustomSlotMaterials () {
74  if (skeletonRenderer == null) {
75  Debug.LogError("skeletonRenderer == null");
76  return;
77  }
78 
79  for (int i = 0; i < customSlotMaterials.Count; i++) {
80  SlotMaterialOverride slotMaterialOverride = customSlotMaterials[i];
81  if (slotMaterialOverride.overrideDisabled || string.IsNullOrEmpty(slotMaterialOverride.slotName))
82  continue;
83 
84  Slot slotObject = skeletonRenderer.skeleton.FindSlot(slotMaterialOverride.slotName);
85  skeletonRenderer.CustomSlotMaterials[slotObject] = slotMaterialOverride.material;
86  }
87  }
88 
89  void RemoveCustomSlotMaterials () {
90  if (skeletonRenderer == null) {
91  Debug.LogError("skeletonRenderer == null");
92  return;
93  }
94 
95  for (int i = 0; i < customSlotMaterials.Count; i++) {
96  SlotMaterialOverride slotMaterialOverride = customSlotMaterials[i];
97  if (string.IsNullOrEmpty(slotMaterialOverride.slotName))
98  continue;
99 
100  Slot slotObject = skeletonRenderer.skeleton.FindSlot(slotMaterialOverride.slotName);
101 
102  Material currentMaterial;
103  if (!skeletonRenderer.CustomSlotMaterials.TryGetValue(slotObject, out currentMaterial))
104  continue;
105 
106  // Do not revert the material if it was changed by something else
107  if (currentMaterial != slotMaterialOverride.material)
108  continue;
109 
110  skeletonRenderer.CustomSlotMaterials.Remove(slotObject);
111  }
112  }
