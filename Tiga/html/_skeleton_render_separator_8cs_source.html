201 
202  int rendererIndex = 0;
203  var currentRenderer = partsRenderers[rendererIndex];
204  for (int si = 0, start = 0; si <= lastSubmeshInstruction; si++) {
205  if (submeshInstructionsItems[si].forceSeparate || si == lastSubmeshInstruction) {
206  // Apply properties

215
