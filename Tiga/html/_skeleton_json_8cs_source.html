89  public SkeletonData ReadSkeletonData (TextReader reader) {
90  if (reader == null) throw new ArgumentNullException("reader", "reader cannot be null.");
91 
92  float scale = this.Scale;
93  var skeletonData = new SkeletonData();
94 
95  var root = Json.Deserialize(reader) as Dictionary<string, Object>;
96  if (root == null) throw new Exception("Invalid JSON.");
97 
98  // Skeleton.
99  if (root.ContainsKey("skeleton")) {
100  var skeletonMap = (Dictionary<string, Object>)root["skeleton"];
101  skeletonData.hash = (string)skeletonMap["hash"];
102  skeletonData.version = (string)skeletonMap["spine"];
103  skeletonData.width = GetFloat(skeletonMap, "width", 0);
104  skeletonData.height = GetFloat(skeletonMap, "height", 0);
105  skeletonData.fps = GetFloat(skeletonMap, "fps", 0);
106  skeletonData.imagesPath = GetString(skeletonMap, "images", null);
107  }
108 
109  // Bones.
110  foreach (Dictionary<string, Object> boneMap in (List<Object>)root["bones"]) {
111  BoneData parent = null;
112  if (boneMap.ContainsKey("parent")) {
113  parent = skeletonData.FindBone((string)boneMap["parent"]);
114  if (parent == null)
115  throw new Exception("Parent bone not found: " + boneMap["parent"]);
116  }
117  var data = new BoneData(skeletonData.Bones.Count, (string)boneMap["name"], parent);
118  data.length = GetFloat(boneMap, "length", 0) * scale;
119  data.x = GetFloat(boneMap, "x", 0) * scale;
120  data.y = GetFloat(boneMap, "y", 0) * scale;
121  data.rotation = GetFloat(boneMap, "rotation", 0);
122  data.scaleX = GetFloat(boneMap, "scaleX", 1);
123  data.scaleY = GetFloat(boneMap, "scaleY", 1);
124  data.shearX = GetFloat(boneMap, "shearX", 0);
125  data.shearY = GetFloat(boneMap, "shearY", 0);
126 
127  string tm = GetString(boneMap, "transform", TransformMode.Normal.ToString());
128  data.transformMode = (TransformMode)Enum.Parse(typeof(TransformMode), tm, true);
129 
130  skeletonData.bones.Add(data);
131  }
132 
133  // Slots.
134  if (root.ContainsKey("slots")) {
135  foreach (Dictionary<string, Object> slotMap in (List<Object>)root["slots"]) {
136  var slotName = (string)slotMap["name"];
137  var boneName = (string)slotMap["bone"];
138  BoneData boneData = skeletonData.FindBone(boneName);
139  if (boneData == null) throw new Exception("Slot bone not found: " + boneName);
140  var data = new SlotData(skeletonData.Slots.Count, slotName, boneData);
141 
142  if (slotMap.ContainsKey("color")) {
143  string color = (string)slotMap["color"];
144  data.r = ToColor(color, 0);
145  data.g = ToColor(color, 1);
146  data.b = ToColor(color, 2);
147  data.a = ToColor(color, 3);
148  }
149 
150  if (slotMap.ContainsKey("dark")) {
151  var color2 = (string)slotMap["dark"];
152  data.r2 = ToColor(color2, 0, 6); // expectedLength = 6. ie. "RRGGBB"
153  data.g2 = ToColor(color2, 1, 6);
154  data.b2 = ToColor(color2, 2, 6);
155  data.hasSecondColor = true;
156  }
157 
158  data.attachmentName = GetString(slotMap, "attachment", null);
159  if (slotMap.ContainsKey("blend"))
160  data.blendMode = (BlendMode)Enum.Parse(typeof(BlendMode), (string)slotMap["blend"], true);
161  else
162  data.blendMode = BlendMode.Normal;
163  skeletonData.slots.Add(data);
164  }
165  }
166 
167  // IK constraints.
168  if (root.ContainsKey("ik")) {
169  foreach (Dictionary<string, Object> constraintMap in (List<Object>)root["ik"]) {
170  IkConstraintData data = new IkConstraintData((string)constraintMap["name"]);
171  data.order = GetInt(constraintMap, "order", 0);
172 
173  foreach (string boneName in (List<Object>)constraintMap["bones"]) {
174  BoneData bone = skeletonData.FindBone(boneName);
175  if (bone == null) throw new Exception("IK constraint bone not found: " + boneName);
176  data.bones.Add(bone);
177  }
178 
179  string targetName = (string)constraintMap["target"];
180  data.target = skeletonData.FindBone(targetName);
181  if (data.target == null) throw new Exception("Target bone not found: " + targetName);
182 
183  data.bendDirection = GetBoolean(constraintMap, "bendPositive", true) ? 1 : -1;
184  data.mix = GetFloat(constraintMap, "mix", 1);
185 
186  skeletonData.ikConstraints.Add(data);
187  }
188  }
189 
190  // Transform constraints.
191  if (root.ContainsKey("transform")) {
192  foreach (Dictionary<string, Object> constraintMap in (List<Object>)root["transform"]) {
193  TransformConstraintData data = new TransformConstraintData((string)constraintMap["name"]);
194  data.order = GetInt(constraintMap, "order", 0);
195 
196  foreach (string boneName in (List<Object>)constraintMap["bones"]) {
197  BoneData bone = skeletonData.FindBone(boneName);
198  if (bone == null) throw new Exception("Transform constraint bone not found: " + boneName);
199  data.bones.Add(bone);
200  }
201 
202  string targetName = (string)constraintMap["target"];
203  data.target = skeletonData.FindBone(targetName);
204  if (data.target == null) throw new Exception("Target bone not found: " + targetName);
205 
206  data.local = GetBoolean(constraintMap, "local", false);
207  data.relative = GetBoolean(constraintMap, "relative", false);
208 
209  data.offsetRotation = GetFloat(constraintMap, "rotation", 0);
210  data.offsetX = GetFloat(constraintMap, "x", 0) * scale;
211  data.offsetY = GetFloat(constraintMap, "y", 0) * scale;
212  data.offsetScaleX = GetFloat(constraintMap, "scaleX", 0);
213  data.offsetScaleY = GetFloat(constraintMap, "scaleY", 0);
214  data.offsetShearY = GetFloat(constraintMap, "shearY", 0);
215 
216  data.rotateMix = GetFloat(constraintMap, "rotateMix", 1);
217  data.translateMix = GetFloat(constraintMap, "translateMix", 1);
218  data.scaleMix = GetFloat(constraintMap, "scaleMix", 1);
219  data.shearMix = GetFloat(constraintMap, "shearMix", 1);
220 
221  skeletonData.transformConstraints.Add(data);
222  }
223  }
224 
225  // Path constraints.
226  if(root.ContainsKey("path")) {
227  foreach (Dictionary<string, Object> constraintMap in (List<Object>)root["path"]) {
228  PathConstraintData data = new PathConstraintData((string)constraintMap["name"]);
229  data.order = GetInt(constraintMap, "order", 0);
230 
231  foreach (string boneName in (List<Object>)constraintMap["bones"]) {
232  BoneData bone = skeletonData.FindBone(boneName);
233  if (bone == null) throw new Exception("Path bone not found: " + boneName);
234  data.bones.Add(bone);
235  }
236 
237  string targetName = (string)constraintMap["target"];
238  data.target = skeletonData.FindSlot(targetName);
239  if (data.target == null) throw new Exception("Target slot not found: " + targetName);
240 
241  data.positionMode = (PositionMode)Enum.Parse(typeof(PositionMode), GetString(constraintMap, "positionMode", "percent"), true);
242  data.spacingMode = (SpacingMode)Enum.Parse(typeof(SpacingMode), GetString(constraintMap, "spacingMode", "length"), true);
243  data.rotateMode = (RotateMode)Enum.Parse(typeof(RotateMode), GetString(constraintMap, "rotateMode", "tangent"), true);
244  data.offsetRotation = GetFloat(constraintMap, "rotation", 0);
245  data.position = GetFloat(constraintMap, "position", 0);
246  if (data.positionMode == PositionMode.Fixed) data.position *= scale;
247  data.spacing = GetFloat(constraintMap, "spacing", 0);
248  if (data.spacingMode == SpacingMode.Length || data.spacingMode == SpacingMode.Fixed) data.spacing *= scale;
249  data.rotateMix = GetFloat(constraintMap, "rotateMix", 1);
250  data.translateMix = GetFloat(constraintMap, "translateMix", 1);
251 
252  skeletonData.pathConstraints.Add(data);
253  }
254  }
255 
256  // Skins.
257  if (root.ContainsKey("skins")) {
258  foreach (KeyValuePair<string, Object> skinMap in (Dictionary<string, Object>)root["skins"]) {
259  var skin = new Skin(skinMap.Key);
260  foreach (KeyValuePair<string, Object> slotEntry in (Dictionary<string, Object>)skinMap.Value) {
261  int slotIndex = skeletonData.FindSlotIndex(slotEntry.Key);
262  foreach (KeyValuePair<string, Object> entry in ((Dictionary<string, Object>)slotEntry.Value)) {
263  try {
264  Attachment attachment = ReadAttachment((Dictionary<string, Object>)entry.Value, skin, slotIndex, entry.Key, skeletonData);
265  if (attachment != null) skin.AddAttachment(slotIndex, entry.Key, attachment);
266  } catch (Exception e) {
267  throw new Exception("Error reading attachment: " + entry.Key + ", skin: " + skin, e);
268  }
269  }
270  }
271  skeletonData.skins.Add(skin);
272  if (skin.name == "default") skeletonData.defaultSkin = skin;
273  }
274  }
275 
276  // Linked meshes.
277  for (int i = 0, n = linkedMeshes.Count; i < n; i++) {
278  LinkedMesh linkedMesh = linkedMeshes[i];
279  Skin skin = linkedMesh.skin == null ? skeletonData.defaultSkin : skeletonData.FindSkin(linkedMesh.skin);
280  if (skin == null) throw new Exception("Slot not found: " + linkedMesh.skin);
281  Attachment parent = skin.GetAttachment(linkedMesh.slotIndex, linkedMesh.parent);
282  if (parent == null) throw new Exception("Parent mesh not found: " + linkedMesh.parent);
283  linkedMesh.mesh.ParentMesh = (MeshAttachment)parent;
284  linkedMesh.mesh.UpdateUVs();
285  }
286  linkedMeshes.Clear();
287 
288  // Events.
289  if (root.ContainsKey("events")) {
290  foreach (KeyValuePair<string, Object> entry in (Dictionary<string, Object>)root["events"]) {
291  var entryMap = (Dictionary<string, Object>)entry.Value;
292  var data = new EventData(entry.Key);
293  data.Int = GetInt(entryMap, "int", 0);
294  data.Float = GetFloat(entryMap, "float", 0);
295  data.String = GetString(entryMap, "string", string.Empty);
296  skeletonData.events.Add(data);
297  }
298  }
299 
300  // Animations.
301  if (root.ContainsKey("animations")) {
302  foreach (KeyValuePair<string, Object> entry in (Dictionary<string, Object>)root["animations"]) {
303  try {
304  ReadAnimation((Dictionary<string, Object>)entry.Value, entry.Key, skeletonData);
305  } catch (Exception e) {
306  throw new Exception("Error reading animation: " + entry.Key, e);
307  }
308  }
309  }
310 
311  skeletonData.bones.TrimExcess();
312  skeletonData.slots.TrimExcess();
313  skeletonData.skins.TrimExcess();
314  skeletonData.events.TrimExcess();
315  skeletonData.animations.TrimExcess();
316  skeletonData.ikConstraints.TrimExcess();
317  return skeletonData;
318  }

320  private Attachment ReadAttachment (Dictionary<string, Object> map, Skin skin, int slotIndex, string name, SkeletonData skeletonData) {
321  float scale = this.Scale;
322  name = GetString(map, "name", name);
323 
324  var typeName = GetString(map, "type", "region");
325  if (typeName == "skinnedmesh") typeName = "weightedmesh";
326  if (typeName == "weightedmesh") typeName = "mesh";
327  if (typeName == "weightedlinkedmesh") typeName = "linkedmesh";
328  var type = (AttachmentType)Enum.Parse(typeof(AttachmentType), typeName, true);
329 
330  string path = GetString(map, "path", name);
331 
332  switch (type) {
333  case AttachmentType.Region:
334  RegionAttachment region = attachmentLoader.NewRegionAttachment(skin, name, path);
335  if (region == null) return null;
336  region.Path = path;
337  region.x = GetFloat(map, "x", 0) * scale;
338  region.y = GetFloat(map, "y", 0) * scale;
339  region.scaleX = GetFloat(map, "scaleX", 1);
340  region.scaleY = GetFloat(map, "scaleY", 1);
341  region.rotation = GetFloat(map, "rotation", 0);
342  region.width = GetFloat(map, "width", 32) * scale;
343  region.height = GetFloat(map, "height", 32) * scale;
344 
345  if (map.ContainsKey("color")) {
346  var color = (string)map["color"];
347  region.r = ToColor(color, 0);
348  region.g = ToColor(color, 1);
349  region.b = ToColor(color, 2);
350  region.a = ToColor(color, 3);
351  }
352 
353  region.UpdateOffset();
354  return region;
355  case AttachmentType.Boundingbox:
356  BoundingBoxAttachment box = attachmentLoader.NewBoundingBoxAttachment(skin, name);
357  if (box == null) return null;
358  ReadVertices(map, box, GetInt(map, "vertexCount", 0) << 1);
359  return box;
360  case AttachmentType.Mesh:
361  case AttachmentType.Linkedmesh: {
362  MeshAttachment mesh = attachmentLoader.NewMeshAttachment(skin, name, path);
363  if (mesh == null) return null;
364  mesh.Path = path;
365 
366  if (map.ContainsKey("color")) {
367  var color = (string)map["color"];
368  mesh.r = ToColor(color, 0);
369  mesh.g = ToColor(color, 1);
370  mesh.b = ToColor(color, 2);
371  mesh.a = ToColor(color, 3);
372  }
373 
374  mesh.Width = GetFloat(map, "width", 0) * scale;
375  mesh.Height = GetFloat(map, "height", 0) * scale;
376 
377  string parent = GetString(map, "parent", null);
378  if (parent != null) {
379  mesh.InheritDeform = GetBoolean(map, "deform", true);
380  linkedMeshes.Add(new LinkedMesh(mesh, GetString(map, "skin", null), slotIndex, parent));
381  return mesh;
382  }
383 
384  float[] uvs = GetFloatArray(map, "uvs", 1);
385  ReadVertices(map, mesh, uvs.Length);
386  mesh.triangles = GetIntArray(map, "triangles");
387  mesh.regionUVs = uvs;
388  mesh.UpdateUVs();
389 
390  if (map.ContainsKey("hull")) mesh.HullLength = GetInt(map, "hull", 0) * 2;
391  if (map.ContainsKey("edges")) mesh.Edges = GetIntArray(map, "edges");
392  return mesh;
393  }
394  case AttachmentType.Path: {
395  PathAttachment pathAttachment = attachmentLoader.NewPathAttachment(skin, name);
396  if (pathAttachment == null) return null;
397  pathAttachment.closed = GetBoolean(map, "closed", false);
398  pathAttachment.constantSpeed = GetBoolean(map, "constantSpeed", true);
399 
400  int vertexCount = GetInt(map, "vertexCount", 0);
401  ReadVertices(map, pathAttachment, vertexCount << 1);
402 
403  // potential BOZO see Java impl
404  pathAttachment.lengths = GetFloatArray(map, "lengths", scale);
405  return pathAttachment;
406  }
407  case AttachmentType.Point: {
408  PointAttachment point = attachmentLoader.NewPointAttachment(skin, name);
409  if (point == null) return null;
410  point.x = GetFloat(map, "x", 0) * scale;
411  point.y = GetFloat(map, "y", 0) * scale;
412  point.rotation = GetFloat(map, "rotation", 0);
413 
414  //string color = GetString(map, "color", null);
415  //if (color != null) point.color = color;
416  return point;
417  }
418  case AttachmentType.Clipping: {
419  ClippingAttachment clip = attachmentLoader.NewClippingAttachment(skin, name);
420  if (clip == null) return null;
421 
422  string end = GetString(map, "end", null);
423  if (end != null) {
424  SlotData slot = skeletonData.FindSlot(end);
425  if (slot == null) throw new Exception("Clipping end slot not found: " + end);
426  clip.EndSlot = slot;
427  }
428 
429  ReadVertices(map, clip, GetInt(map, "vertexCount", 0) << 1);
430 
431  //string color = GetString(map, "color", null);
432  // if (color != null) clip.color = color;
433  return clip;
434  }
435  }
436  return null;
437  }

439  private void ReadVertices (Dictionary<string, Object> map, VertexAttachment attachment, int verticesLength) {
440  attachment.WorldVerticesLength = verticesLength;
441  float[] vertices = GetFloatArray(map, "vertices", 1);
442  float scale = Scale;
443  if (verticesLength == vertices.Length) {
444  if (scale != 1) {
445  for (int i = 0; i < vertices.Length; i++) {
446  vertices[i] *= scale;
447  }
448  }
449  attachment.vertices = vertices;
450  return;
451  }
452  ExposedList<float> weights = new ExposedList<float>(verticesLength * 3 * 3);
453  ExposedList<int> bones = new ExposedList<int>(verticesLength * 3);
454  for (int i = 0, n = vertices.Length; i < n;) {
455  int boneCount = (int)vertices[i++];
456  bones.Add(boneCount);
457  for (int nn = i + boneCount * 4; i < nn; i += 4) {
458  bones.Add((int)vertices[i]);
459  weights.Add(vertices[i + 1] * this.Scale);
460  weights.Add(vertices[i + 2] * this.Scale);
461  weights.Add(vertices[i + 3]);
462  }
463  }
464  attachment.bones = bones.ToArray();
465  attachment.vertices = weights.ToArray();
466  }

468  private void ReadAnimation (Dictionary<string, Object> map, string name, SkeletonData skeletonData) {
469  var scale = this.Scale;
470  var timelines = new ExposedList<Timeline>();
471  float duration = 0;
472 
473  // Slot timelines.
474  if (map.ContainsKey("slots")) {
475  foreach (KeyValuePair<string, Object> entry in (Dictionary<string, Object>)map["slots"]) {
476  string slotName = entry.Key;
477  int slotIndex = skeletonData.FindSlotIndex(slotName);
478  var timelineMap = (Dictionary<string, Object>)entry.Value;
479  foreach (KeyValuePair<string, Object> timelineEntry in timelineMap) {
480  var values = (List<Object>)timelineEntry.Value;
481  var timelineName = (string)timelineEntry.Key;
482  if (timelineName == "attachment") {
483  var timeline = new AttachmentTimeline(values.Count);
484  timeline.slotIndex = slotIndex;
485 
486  int frameIndex = 0;
487  foreach (Dictionary<string, Object> valueMap in values) {
488  float time = (float)valueMap["time"];
489  timeline.SetFrame(frameIndex++, time, (string)valueMap["name"]);
490  }
491  timelines.Add(timeline);
492  duration = Math.Max(duration, timeline.frames[timeline.FrameCount - 1]);
493 
494  } else if (timelineName == "color") {
495  var timeline = new ColorTimeline(values.Count);
496  timeline.slotIndex = slotIndex;
497 
498  int frameIndex = 0;
499  foreach (Dictionary<string, Object> valueMap in values) {
500  float time = (float)valueMap["time"];
501  string c = (string)valueMap["color"];
502  timeline.SetFrame(frameIndex, time, ToColor(c, 0), ToColor(c, 1), ToColor(c, 2), ToColor(c, 3));
503  ReadCurve(valueMap, timeline, frameIndex);
504  frameIndex++;
505  }
506  timelines.Add(timeline);
507  duration = Math.Max(duration, timeline.frames[(timeline.FrameCount - 1) * ColorTimeline.ENTRIES]);
508 
509  } else if (timelineName == "twoColor") {
510  var timeline = new TwoColorTimeline(values.Count);
511  timeline.slotIndex = slotIndex;
512 
513  int frameIndex = 0;
514  foreach (Dictionary<string, Object> valueMap in values) {
515  float time = (float)valueMap["time"];
516  string light = (string)valueMap["light"];
517  string dark = (string)valueMap["dark"];
518  timeline.SetFrame(frameIndex, time, ToColor(light, 0), ToColor(light, 1), ToColor(light, 2), ToColor(light, 3),
519  ToColor(dark, 0, 6), ToColor(dark, 1, 6), ToColor(dark, 2, 6));
520  ReadCurve(valueMap, timeline, frameIndex);
521  frameIndex++;
522  }
523  timelines.Add(timeline);
524  duration = Math.Max(duration, timeline.frames[(timeline.FrameCount - 1) * TwoColorTimeline.ENTRIES]);
525 
526  } else
527  throw new Exception("Invalid timeline type for a slot: " + timelineName + " (" + slotName + ")");
528  }
529  }
530  }
531 
532  // Bone timelines.
533  if (map.ContainsKey("bones")) {
534  foreach (KeyValuePair<string, Object> entry in (Dictionary<string, Object>)map["bones"]) {
535  string boneName = entry.Key;
536  int boneIndex = skeletonData.FindBoneIndex(boneName);
537  if (boneIndex == -1) throw new Exception("Bone not found: " + boneName);
538  var timelineMap = (Dictionary<string, Object>)entry.Value;
539  foreach (KeyValuePair<string, Object> timelineEntry in timelineMap) {
540  var values = (List<Object>)timelineEntry.Value;
541  var timelineName = (string)timelineEntry.Key;
542  if (timelineName == "rotate") {
543  var timeline = new RotateTimeline(values.Count);
544  timeline.boneIndex = boneIndex;
545 
546  int frameIndex = 0;
547  foreach (Dictionary<string, Object> valueMap in values) {
548  timeline.SetFrame(frameIndex, (float)valueMap["time"], (float)valueMap["angle"]);
549  ReadCurve(valueMap, timeline, frameIndex);
550  frameIndex++;
551  }
552  timelines.Add(timeline);
553  duration = Math.Max(duration, timeline.frames[(timeline.FrameCount - 1) * RotateTimeline.ENTRIES]);
554 
555  } else if (timelineName == "translate" || timelineName == "scale" || timelineName == "shear") {
556  TranslateTimeline timeline;
557  float timelineScale = 1;
558  if (timelineName == "scale")
559  timeline = new ScaleTimeline(values.Count);
560  else if (timelineName == "shear")
561  timeline = new ShearTimeline(values.Count);
562  else {
