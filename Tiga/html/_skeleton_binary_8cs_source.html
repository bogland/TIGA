115  public static string GetVersionString (Stream input) {
116  if (input == null) throw new ArgumentNullException("input");
117 
118  try {
119  // Hash.
120  int byteCount = ReadVarint(input, true);
121  if (byteCount > 1) input.Position += byteCount - 1;
122 
123  // Version.
124  byteCount = ReadVarint(input, true);
125  if (byteCount > 1) {
126  byteCount--;
127  var buffer = new byte[byteCount];
128  ReadFully(input, buffer, 0, byteCount);
129  return System.Text.Encoding.UTF8.GetString(buffer, 0, byteCount);
130  }
131 
132  throw new ArgumentException("Stream does not contain a valid binary Skeleton Data.", "input");
133  } catch (Exception e) {
134  throw new ArgumentException("Stream does not contain a valid binary Skeleton Data.\n" + e, "input");
135  }
136  }

138  public SkeletonData ReadSkeletonData (Stream input) {
139  if (input == null) throw new ArgumentNullException("input");
140  float scale = Scale;
141 
142  var skeletonData = new SkeletonData();
143  skeletonData.hash = ReadString(input);
144  if (skeletonData.hash.Length == 0) skeletonData.hash = null;
145  skeletonData.version = ReadString(input);
146  if (skeletonData.version.Length == 0) skeletonData.version = null;
147  skeletonData.width = ReadFloat(input);
148  skeletonData.height = ReadFloat(input);
149 
150  bool nonessential = ReadBoolean(input);
151 
152  if (nonessential) {
153  skeletonData.fps = ReadFloat(input);
154  skeletonData.imagesPath = ReadString(input);
155  if (skeletonData.imagesPath.Length == 0) skeletonData.imagesPath = null;
156  }
157 
158  // Bones.
159  for (int i = 0, n = ReadVarint(input, true); i < n; i++) {
160  String name = ReadString(input);
161  BoneData parent = i == 0 ? null : skeletonData.bones.Items[ReadVarint(input, true)];
162  BoneData data = new BoneData(i, name, parent);
163  data.rotation = ReadFloat(input);
164  data.x = ReadFloat(input) * scale;
165  data.y = ReadFloat(input) * scale;
166  data.scaleX = ReadFloat(input);
167  data.scaleY = ReadFloat(input);
168  data.shearX = ReadFloat(input);
169  data.shearY = ReadFloat(input);
170  data.length = ReadFloat(input) * scale;
171  data.transformMode = TransformModeValues[ReadVarint(input, true)];
172  if (nonessential) ReadInt(input); // Skip bone color.
173  skeletonData.bones.Add(data);
174  }
175 
176  // Slots.
177  for (int i = 0, n = ReadVarint(input, true); i < n; i++) {
178  String slotName = ReadString(input);
179  BoneData boneData = skeletonData.bones.Items[ReadVarint(input, true)];
180  SlotData slotData = new SlotData(i, slotName, boneData);
181  int color = ReadInt(input);
182  slotData.r = ((color & 0xff000000) >> 24) / 255f;
183  slotData.g = ((color & 0x00ff0000) >> 16) / 255f;
184  slotData.b = ((color & 0x0000ff00) >> 8) / 255f;
185  slotData.a = ((color & 0x000000ff)) / 255f;
186 
187  int darkColor = ReadInt(input); // 0x00rrggbb
188  if (darkColor != -1) {
189  slotData.hasSecondColor = true;
190  slotData.r2 = ((darkColor & 0x00ff0000) >> 16) / 255f;
191  slotData.g2 = ((darkColor & 0x0000ff00) >> 8) / 255f;
192  slotData.b2 = ((darkColor & 0x000000ff)) / 255f;
193  }
194 
195  slotData.attachmentName = ReadString(input);
196  slotData.blendMode = (BlendMode)ReadVarint(input, true);
197  skeletonData.slots.Add(slotData);
198  }
199 
200  // IK constraints.
201  for (int i = 0, n = ReadVarint(input, true); i < n; i++) {
202  IkConstraintData data = new IkConstraintData(ReadString(input));
203  data.order = ReadVarint(input, true);
204  for (int ii = 0, nn = ReadVarint(input, true); ii < nn; ii++)
205  data.bones.Add(skeletonData.bones.Items[ReadVarint(input, true)]);
206  data.target = skeletonData.bones.Items[ReadVarint(input, true)];
207  data.mix = ReadFloat(input);
208  data.bendDirection = ReadSByte(input);
209  skeletonData.ikConstraints.Add(data);
210  }
211 
212  // Transform constraints.
213  for (int i = 0, n = ReadVarint(input, true); i < n; i++) {
214  TransformConstraintData data = new TransformConstraintData(ReadString(input));
215  data.order = ReadVarint(input, true);
216  for (int ii = 0, nn = ReadVarint(input, true); ii < nn; ii++)
217  data.bones.Add(skeletonData.bones.Items[ReadVarint(input, true)]);
218  data.target = skeletonData.bones.Items[ReadVarint(input, true)];
219  data.local = ReadBoolean(input);
220  data.relative = ReadBoolean(input);
221  data.offsetRotation = ReadFloat(input);
222  data.offsetX = ReadFloat(input) * scale;
223  data.offsetY = ReadFloat(input) * scale;
224  data.offsetScaleX = ReadFloat(input);
225  data.offsetScaleY = ReadFloat(input);
226  data.offsetShearY = ReadFloat(input);
227  data.rotateMix = ReadFloat(input);
228  data.translateMix = ReadFloat(input);
229  data.scaleMix = ReadFloat(input);
230  data.shearMix = ReadFloat(input);
231  skeletonData.transformConstraints.Add(data);
232  }
233 
234  // Path constraints
235  for (int i = 0, n = ReadVarint(input, true); i < n; i++) {
236  PathConstraintData data = new PathConstraintData(ReadString(input));
237  data.order = ReadVarint(input, true);
238  for (int ii = 0, nn = ReadVarint(input, true); ii < nn; ii++)
239  data.bones.Add(skeletonData.bones.Items[ReadVarint(input, true)]);
240  data.target = skeletonData.slots.Items[ReadVarint(input, true)];
241  data.positionMode = (PositionMode)Enum.GetValues(typeof(PositionMode)).GetValue(ReadVarint(input, true));
242  data.spacingMode = (SpacingMode)Enum.GetValues(typeof(SpacingMode)).GetValue(ReadVarint(input, true));
243  data.rotateMode = (RotateMode)Enum.GetValues(typeof(RotateMode)).GetValue(ReadVarint(input, true));
244  data.offsetRotation = ReadFloat(input);
245  data.position = ReadFloat(input);
246  if (data.positionMode == PositionMode.Fixed) data.position *= scale;
247  data.spacing = ReadFloat(input);
248  if (data.spacingMode == SpacingMode.Length || data.spacingMode == SpacingMode.Fixed) data.spacing *= scale;
249  data.rotateMix = ReadFloat(input);
250  data.translateMix = ReadFloat(input);
251  skeletonData.pathConstraints.Add(data);
252  }
253 
254  // Default skin.
255  Skin defaultSkin = ReadSkin(input, skeletonData, "default", nonessential);
256  if (defaultSkin != null) {
257  skeletonData.defaultSkin = defaultSkin;
258  skeletonData.skins.Add(defaultSkin);
259  }
260 
261  // Skins.
262  for (int i = 0, n = ReadVarint(input, true); i < n; i++)
263  skeletonData.skins.Add(ReadSkin(input, skeletonData, ReadString(input), nonessential));
264 
265  // Linked meshes.
266  for (int i = 0, n = linkedMeshes.Count; i < n; i++) {
267  SkeletonJson.LinkedMesh linkedMesh = linkedMeshes[i];
268  Skin skin = linkedMesh.skin == null ? skeletonData.DefaultSkin : skeletonData.FindSkin(linkedMesh.skin);
269  if (skin == null) throw new Exception("Skin not found: " + linkedMesh.skin);
270  Attachment parent = skin.GetAttachment(linkedMesh.slotIndex, linkedMesh.parent);
271  if (parent == null) throw new Exception("Parent mesh not found: " + linkedMesh.parent);
272  linkedMesh.mesh.ParentMesh = (MeshAttachment)parent;
273  linkedMesh.mesh.UpdateUVs();
274  }
275  linkedMeshes.Clear();
276 
277  // Events.
278  for (int i = 0, n = ReadVarint(input, true); i < n; i++) {
279  EventData data = new EventData(ReadString(input));
280  data.Int = ReadVarint(input, false);
281  data.Float = ReadFloat(input);
282  data.String = ReadString(input);
283  skeletonData.events.Add(data);
284  }
285 
286  // Animations.
287  for (int i = 0, n = ReadVarint(input, true); i < n; i++)
288  ReadAnimation(ReadString(input), input, skeletonData);
289 
290  skeletonData.bones.TrimExcess();
291  skeletonData.slots.TrimExcess();
292  skeletonData.skins.TrimExcess();
293  skeletonData.events.TrimExcess();
294  skeletonData.animations.TrimExcess();
295  skeletonData.ikConstraints.TrimExcess();
296  skeletonData.pathConstraints.TrimExcess();
297  return skeletonData;
298  }

302  private Skin ReadSkin (Stream input, SkeletonData skeletonData, String skinName, bool nonessential) {
303  int slotCount = ReadVarint(input, true);
304  if (slotCount == 0) return null;
305  Skin skin = new Skin(skinName);
306  for (int i = 0; i < slotCount; i++) {
307  int slotIndex = ReadVarint(input, true);
308  for (int ii = 0, nn = ReadVarint(input, true); ii < nn; ii++) {
309  String name = ReadString(input);
310  Attachment attachment = ReadAttachment(input, skeletonData, skin, slotIndex, name, nonessential);
311  if (attachment != null) skin.AddAttachment(slotIndex, name, attachment);
312  }
313  }
314  return skin;
315  }

317  private Attachment ReadAttachment (Stream input, SkeletonData skeletonData, Skin skin, int slotIndex, String attachmentName, bool nonessential) {
318  float scale = Scale;
319 
320  String name = ReadString(input);
321  if (name == null) name = attachmentName;
322 
323  AttachmentType type = (AttachmentType)input.ReadByte();
324  switch (type) {
325  case AttachmentType.Region: {
326  String path = ReadString(input);
327  float rotation = ReadFloat(input);
328  float x = ReadFloat(input);
329  float y = ReadFloat(input);
330  float scaleX = ReadFloat(input);
331  float scaleY = ReadFloat(input);
332  float width = ReadFloat(input);
333  float height = ReadFloat(input);
334  int color = ReadInt(input);
335 
336  if (path == null) path = name;
337  RegionAttachment region = attachmentLoader.NewRegionAttachment(skin, name, path);
338  if (region == null) return null;
339  region.Path = path;
340  region.x = x * scale;
341  region.y = y * scale;
342  region.scaleX = scaleX;
343  region.scaleY = scaleY;
344  region.rotation = rotation;
345  region.width = width * scale;
346  region.height = height * scale;
347  region.r = ((color & 0xff000000) >> 24) / 255f;
348  region.g = ((color & 0x00ff0000) >> 16) / 255f;
349  region.b = ((color & 0x0000ff00) >> 8) / 255f;
350  region.a = ((color & 0x000000ff)) / 255f;
351  region.UpdateOffset();
352  return region;
353  }
354  case AttachmentType.Boundingbox: {
355  int vertexCount = ReadVarint(input, true);
356  Vertices vertices = ReadVertices(input, vertexCount);
357  if (nonessential) ReadInt(input); //int color = nonessential ? ReadInt(input) : 0; // Avoid unused local warning.
358 
359  BoundingBoxAttachment box = attachmentLoader.NewBoundingBoxAttachment(skin, name);
360  if (box == null) return null;
361  box.worldVerticesLength = vertexCount << 1;
362  box.vertices = vertices.vertices;
363  box.bones = vertices.bones;
364  return box;
365  }
366  case AttachmentType.Mesh: {
367  String path = ReadString(input);
368  int color = ReadInt(input);
369  int vertexCount = ReadVarint(input, true);
370  float[] uvs = ReadFloatArray(input, vertexCount << 1, 1);
371  int[] triangles = ReadShortArray(input);
372  Vertices vertices = ReadVertices(input, vertexCount);
373  int hullLength = ReadVarint(input, true);
374  int[] edges = null;
375  float width = 0, height = 0;
376  if (nonessential) {
377  edges = ReadShortArray(input);
378  width = ReadFloat(input);
379  height = ReadFloat(input);
380  }
381 
382  if (path == null) path = name;
383  MeshAttachment mesh = attachmentLoader.NewMeshAttachment(skin, name, path);
384  if (mesh == null) return null;
385  mesh.Path = path;
386  mesh.r = ((color & 0xff000000) >> 24) / 255f;
387  mesh.g = ((color & 0x00ff0000) >> 16) / 255f;
388  mesh.b = ((color & 0x0000ff00) >> 8) / 255f;
389  mesh.a = ((color & 0x000000ff)) / 255f;
390  mesh.bones = vertices.bones;
391  mesh.vertices = vertices.vertices;
392  mesh.WorldVerticesLength = vertexCount << 1;
393  mesh.triangles = triangles;
394  mesh.regionUVs = uvs;
395  mesh.UpdateUVs();
396  mesh.HullLength = hullLength << 1;
397  if (nonessential) {
398  mesh.Edges = edges;
399  mesh.Width = width * scale;
400  mesh.Height = height * scale;
401  }
402  return mesh;
403  }
404  case AttachmentType.Linkedmesh: {
405  String path = ReadString(input);
406  int color = ReadInt(input);
407  String skinName = ReadString(input);
408  String parent = ReadString(input);
409  bool inheritDeform = ReadBoolean(input);
410  float width = 0, height = 0;
411  if (nonessential) {
412  width = ReadFloat(input);
413  height = ReadFloat(input);
414  }
415 
416  if (path == null) path = name;
417  MeshAttachment mesh = attachmentLoader.NewMeshAttachment(skin, name, path);
418  if (mesh == null) return null;
419  mesh.Path = path;
420  mesh.r = ((color & 0xff000000) >> 24) / 255f;
421  mesh.g = ((color & 0x00ff0000) >> 16) / 255f;
422  mesh.b = ((color & 0x0000ff00) >> 8) / 255f;
423  mesh.a = ((color & 0x000000ff)) / 255f;
424  mesh.inheritDeform = inheritDeform;
425  if (nonessential) {
426  mesh.Width = width * scale;
427  mesh.Height = height * scale;
428  }
429  linkedMeshes.Add(new SkeletonJson.LinkedMesh(mesh, skinName, slotIndex, parent));
430  return mesh;
431  }
432  case AttachmentType.Path: {
433  bool closed = ReadBoolean(input);
434  bool constantSpeed = ReadBoolean(input);
435  int vertexCount = ReadVarint(input, true);
436  Vertices vertices = ReadVertices(input, vertexCount);
437  float[] lengths = new float[vertexCount / 3];
438  for (int i = 0, n = lengths.Length; i < n; i++)
439  lengths[i] = ReadFloat(input) * scale;
440  if (nonessential) ReadInt(input); //int color = nonessential ? ReadInt(input) : 0;
441 
442  PathAttachment path = attachmentLoader.NewPathAttachment(skin, name);
443  if (path == null) return null;
444  path.closed = closed;
445  path.constantSpeed = constantSpeed;
446  path.worldVerticesLength = vertexCount << 1;
447  path.vertices = vertices.vertices;
448  path.bones = vertices.bones;
449  path.lengths = lengths;
450  return path;
451  }
452  case AttachmentType.Point: {
453  float rotation = ReadFloat(input);
454  float x = ReadFloat(input);
455  float y = ReadFloat(input);
456  if (nonessential) ReadInt(input); //int color = nonessential ? ReadInt(input) : 0;
457 
458  PointAttachment point = attachmentLoader.NewPointAttachment(skin, name);
459  if (point == null) return null;
460  point.x = x * scale;
461  point.y = y * scale;
462  point.rotation = rotation;
463  //if (nonessential) point.color = color;
464  return point;
465  }
466  case AttachmentType.Clipping: {
467  int endSlotIndex = ReadVarint(input, true);
468  int vertexCount = ReadVarint(input, true);
469  Vertices vertices = ReadVertices(input, vertexCount);
470  if (nonessential) ReadInt(input);
471 
472  ClippingAttachment clip = attachmentLoader.NewClippingAttachment(skin, name);
473  if (clip == null) return null;
474  clip.EndSlot = skeletonData.slots.Items[endSlotIndex];
475  clip.worldVerticesLength = vertexCount << 1;
476  clip.vertices = vertices.vertices;
477  clip.bones = vertices.bones;
478  return clip;
479  }
480  }
481  return null;
482  }

484  private Vertices ReadVertices (Stream input, int vertexCount) {
485  float scale = Scale;
486  int verticesLength = vertexCount << 1;
487  Vertices vertices = new Vertices();
488  if(!ReadBoolean(input)) {
489  vertices.vertices = ReadFloatArray(input, verticesLength, scale);
490  return vertices;
491  }
492  var weights = new ExposedList<float>(verticesLength * 3 * 3);
493  var bonesArray = new ExposedList<int>(verticesLength * 3);
494  for (int i = 0; i < vertexCount; i++) {
495  int boneCount = ReadVarint(input, true);
496  bonesArray.Add(boneCount);
497  for (int ii = 0; ii < boneCount; ii++) {
498  bonesArray.Add(ReadVarint(input, true));
499  weights.Add(ReadFloat(input) * scale);
500  weights.Add(ReadFloat(input) * scale);
501  weights.Add(ReadFloat(input));
502  }
503  }
504 
505  vertices.vertices = weights.ToArray();
506  vertices.bones = bonesArray.ToArray();
507  return vertices;
508  }

510  private float[] ReadFloatArray (Stream input, int n, float scale) {
511  float[] array = new float[n];
512  if (scale == 1) {
513  for (int i = 0; i < n; i++)
514  array[i] = ReadFloat(input);
515  } else {
516  for (int i = 0; i < n; i++)
517  array[i] = ReadFloat(input) * scale;
518  }
519  return array;
520  }

522  private int[] ReadShortArray (Stream input) {
523  int n = ReadVarint(input, true);
524  int[] array = new int[n];
525  for (int i = 0; i < n; i++)
526  array[i] = (input.ReadByte() << 8) | input.ReadByte();
527  return array;
528  }

530  private void ReadAnimation (String name, Stream input, SkeletonData skeletonData) {
531  var timelines = new ExposedList<Timeline>();
532  float scale = Scale;
533  float duration = 0;
534 
535  // Slot timelines.
536  for (int i = 0, n = ReadVarint(input, true); i < n; i++) {
537  int slotIndex = ReadVarint(input, true);
538  for (int ii = 0, nn = ReadVarint(input, true); ii < nn; ii++) {
539  int timelineType = input.ReadByte();
540  int frameCount = ReadVarint(input, true);
541  switch (timelineType) {
542  case SLOT_ATTACHMENT: {
543  AttachmentTimeline timeline = new AttachmentTimeline(frameCount);
544  timeline.slotIndex = slotIndex;
545  for (int frameIndex = 0; frameIndex < frameCount; frameIndex++)
546  timeline.SetFrame(frameIndex, ReadFloat(input), ReadString(input));
547  timelines.Add(timeline);
548  duration = Math.Max(duration, timeline.frames[frameCount - 1]);
549  break;
550  }
551  case SLOT_COLOR: {
552  ColorTimeline timeline = new ColorTimeline(frameCount);
553  timeline.slotIndex = slotIndex;
554  for (int frameIndex = 0; frameIndex < frameCount; frameIndex++) {
555  float time = ReadFloat(input);
556  int color = ReadInt(input);
557  float r = ((color & 0xff000000) >> 24) / 255f;
558  float g = ((color & 0x00ff0000) >> 16) / 255f;
559  float b = ((color & 0x0000ff00) >> 8) / 255f;
560  float a = ((color & 0x000000ff)) / 255f;
561  timeline.SetFrame(frameIndex, time, r, g, b, a);
562  if (frameIndex < frameCount - 1) ReadCurve(input, frameIndex, timeline);
563  }
564  timelines.Add(timeline);
565  duration = Math.Max(duration, timeline.frames[(timeline.FrameCount - 1) * ColorTimeline.ENTRIES]);
566  break;
567  }
568  case SLOT_TWO_COLOR: {
569  TwoColorTimeline timeline = new TwoColorTimeline(frameCount);
570  timeline.slotIndex = slotIndex;
571  for (int frameIndex = 0; frameIndex < frameCount; frameIndex++) {
572  float time = ReadFloat(input);
573  int color = ReadInt(input);
574  float r = ((color & 0xff000000) >> 24) / 255f;
575  float g = ((color & 0x00ff0000) >> 16) / 255f;
576  float b = ((color & 0x0000ff00) >> 8) / 255f;
577  float a = ((color & 0x000000ff)) / 255f;
578  int color2 = ReadInt(input); // 0x00rrggbb
579  float r2 = ((color2 & 0x00ff0000) >> 16) / 255f;
580  float g2 = ((color2 & 0x0000ff00) >> 8) / 255f;
581  float b2 = ((color2 & 0x000000ff)) / 255f;
582
