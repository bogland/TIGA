37  // This is an example script that shows you how to change images on your skeleton using UnityEngine.Sprites.
38  public class MixAndMatch : MonoBehaviour {
39 
40  #region Inspector
41  [SpineSkin]
42  public string templateAttachmentsSkin = "base";
43  public Material sourceMaterial; // This will be used as the basis for shader and material property settings.
44 
45  [Header("Visor")]
46  public Sprite visorSprite;
47  [SpineSlot] public string visorSlot;
48  [SpineAttachment(slotField:"visorSlot", skinField:"baseSkinName")] public string visorKey = "goggles";
49 
50  [Header("Gun")]
51  public Sprite gunSprite;
52  [SpineSlot] public string gunSlot;
53  [SpineAttachment(slotField:"gunSlot", skinField:"baseSkinName")] public string gunKey = "gun";
54 
55  [Header("Runtime Repack")]
56  public bool repack = true;
57  public BoundingBoxFollower bbFollower;
58 
59  [Header("Do not assign")]
60  public Texture2D runtimeAtlas;
61  public Material runtimeMaterial;
62  #endregion
63 
64  Skin customSkin;
65 
66  void OnValidate () {
67  if (sourceMaterial == null) {
68  var skeletonAnimation = GetComponent<SkeletonAnimation>();
69  if (skeletonAnimation != null)
70  sourceMaterial = skeletonAnimation.SkeletonDataAsset.atlasAssets[0].materials[0];
71  }
72  }
73 
74  IEnumerator Start () {
75  yield return new WaitForSeconds(1f); // Delay for one second before applying. For testing.
76  Apply();
77  }
78 
79  void Apply () {
80  var skeletonAnimation = GetComponent<SkeletonAnimation>();
81  var skeleton = skeletonAnimation.Skeleton;
82 
83  // STEP 0: PREPARE SKINS
84  // Let's prepare a new skin to be our custom skin with equips/customizations. We get a clone so our original skins are unaffected.
85  customSkin = customSkin ?? new Skin("custom skin"); // This requires that all customizations are done with skin placeholders defined in Spine.
86  //customSkin = customSkin ?? skeleton.UnshareSkin(true, false, skeletonAnimation.AnimationState); // use this if you are not customizing on the default skin.
87  var templateSkin = skeleton.Data.FindSkin(templateAttachmentsSkin);
88 
89  // STEP 1: "EQUIP" ITEMS USING SPRITES
90  // STEP 1.1 Find the original/template attachment.
91  // Step 1.2 Get a clone of the original/template attachment.
92  // Step 1.3 Apply the Sprite image to the clone.
93  // Step 1.4 Add the remapped clone to the new custom skin.
94 
95  // Let's do this for the visor.
96  int visorSlotIndex = skeleton.FindSlotIndex(visorSlot); // You can access GetAttachment and SetAttachment via string, but caching the slotIndex is faster.
97  Attachment templateAttachment = templateSkin.GetAttachment(visorSlotIndex, visorKey); // STEP 1.1
98  Attachment newAttachment = templateAttachment.GetRemappedClone(visorSprite, sourceMaterial); // STEP 1.2 - 1.3
99  customSkin.SetAttachment(visorSlotIndex, visorKey, newAttachment); // STEP 1.4
100 
101  // And now for the gun.
102  int gunSlotIndex = skeleton.FindSlotIndex(gunSlot);
103  Attachment templateGun = templateSkin.GetAttachment(gunSlotIndex, gunKey); // STEP 1.1
104  Attachment newGun = templateGun.GetRemappedClone(gunSprite, sourceMaterial); // STEP 1.2 - 1.3
105  if (newGun != null) customSkin.SetAttachment(gunSlotIndex, gunKey, newGun); // STEP 1.4
106 
107  // customSkin.RemoveAttachment(gunSlotIndex, gunKey); // To remove an item.
108  // customSkin.Clear()
109  // Use skin.Clear() To remove all customizations.
110  // Customizations will fall back to the value in the default skin if it was defined there.
111  // To prevent fallback from happening, make sure the key is not defined in the default skin.
112 
113  // STEP 3: APPLY AND CLEAN UP.
114  // Recommended: REPACK THE CUSTOM SKIN TO MINIMIZE DRAW CALLS
115  // Repacking requires that you set all source textures/sprites/atlases to be Read/Write enabled in the inspector.
116  // Combine all the attachment sources into one skin. Usually this means the default skin and the custom skin.
117  // call Skin.GetRepackedSkin to get a cloned skin with cloned attachments that all use one texture.
118  // Under the hood, this relies on
119  if (repack) {
120  var repackedSkin = new Skin("repacked skin");
121  repackedSkin.Append(skeleton.Data.DefaultSkin); // Include the "default" skin. (everything outside of skin placeholders)
122  repackedSkin.Append(customSkin); // Include your new custom skin.
123  repackedSkin = repackedSkin.GetRepackedSkin("repacked skin", sourceMaterial, out runtimeMaterial, out runtimeAtlas); // Pack all the items in the skin.
124  skeleton.SetSkin(repackedSkin); // Assign the repacked skin to your Skeleton.
125  if (bbFollower != null) bbFollower.Initialize(true);
126  } else {
127  skeleton.SetSkin(customSkin); // Just use the custom skin directly.
128  }
129 
130  skeleton.SetSlotsToSetupPose(); // Use the pose from setup pose.
131  skeletonAnimation.Update(0); // Use the pose in the currently active animation.
132 
133  Resources.UnloadUnusedAssets();
134  }
135  }
136 }
