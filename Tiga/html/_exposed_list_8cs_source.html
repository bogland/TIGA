91  public ExposedList<T> Resize (int newSize) {
92  int itemsLength = Items.Length;
93  var oldItems = Items;
94  if (newSize > itemsLength) {
95  Array.Resize(ref Items, newSize);
96 // var newItems = new T[newSize];
97 // Array.Copy(oldItems, newItems, Count);
98 // Items = newItems;
99  } else if (newSize < itemsLength) {
100  // Allow nulling of T reference type to allow GC.
101  for (int i = newSize; i < itemsLength; i++)
102  oldItems[i] = default(T);
103  }
104  Count = newSize;
105  return this;
106  }

428  public int RemoveAll (Predicate<T> match) {
429  CheckMatch(match);
430  int i = 0;
431  int j = 0;
432 
433  // Find the first item to remove
434  for (i = 0; i < Count; i++)
435  if (match(Items[i]))
436  break;
437 
438  if (i == Count)
439  return 0;
440 
441  version++;
442 
443  // Remove any additional items
444  for (j = i + 1; j < Count; j++) {
445  if (!match(Items[j]))
446  Items[i++] = Items[j];
447  }
448  if (j - i > 0)
449  Array.Clear(Items, i, j - i);
450 
451  Count = i;
452  return (j - i);
453  }
