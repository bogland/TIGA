53  internal static Material GetMaterialFor (Material materialSource, Texture2D texture) {
54  if (materialSource == null || texture == null) return null;
55 
56  var mt = SlotBlendModes.MaterialTable;
57  Material m;
58  var key = new MaterialTexturePair { material = materialSource, texture2D = texture };
59  if (!mt.TryGetValue(key, out m)) {
60  m = new Material(materialSource);
61  m.name = "(Clone)" + texture.name + "-" + materialSource.name;
62  m.mainTexture = texture;
63  mt[key] = m;
64  }
65 
66  return m;
67  }

87  public void Apply () {
88  GetTexture();
89  if (texture == null) return;
90 
91  var skeletonRenderer = GetComponent<SkeletonRenderer>();
92  if (skeletonRenderer == null) return;
93 
94  var slotMaterials = skeletonRenderer.CustomSlotMaterials;
95 
96  foreach (var s in skeletonRenderer.Skeleton.Slots) {
97  switch (s.data.blendMode) {
98  case BlendMode.Multiply:
99  if (multiplyMaterialSource != null) slotMaterials[s] = GetMaterialFor(multiplyMaterialSource, texture);
100  break;
101  case BlendMode.Screen:
102  if (screenMaterialSource != null) slotMaterials[s] = GetMaterialFor(screenMaterialSource, texture);
103  break;
104  }
105  }
106 
107  Applied = true;
108  skeletonRenderer.LateUpdate();
109  }
110 
111  public void Remove () {
112  GetTexture();
113  if (texture == null) return;
114 
115  var skeletonRenderer = GetComponent<SkeletonRenderer>();
116  if (skeletonRenderer == null) return;
117 
118  var slotMaterials = skeletonRenderer.CustomSlotMaterials;
119 
120  foreach (var s in skeletonRenderer.Skeleton.Slots) {
121  Material m = null;
122 
123  switch (s.data.blendMode) {
124  case BlendMode.Multiply:
125  if (slotMaterials.TryGetValue(s, out m) && Material.ReferenceEquals(m, GetMaterialFor(multiplyMaterialSource, texture)))
126  slotMaterials.Remove(s);
127  break;
128  case BlendMode.Screen:
129  if (slotMaterials.TryGetValue(s, out m) && Material.ReferenceEquals(m, GetMaterialFor(screenMaterialSource, texture)))
130  slotMaterials.Remove(s);
131  break;
132  }
133  }
134 
135  Applied = false;
136  if (skeletonRenderer.valid) skeletonRenderer.LateUpdate();
137  }

139  public void GetTexture () {
140  if (texture == null) {
141  var sr = GetComponent<SkeletonRenderer>(); if (sr == null) return;
142  var sda = sr.skeletonDataAsset; if (sda == null) return;
143  var aa = sda.atlasAssets[0]; if (aa == null) return;
144  var am = aa.materials[0]; if (am == null) return;
145  texture = am.mainTexture as Texture2D;
146  }
147  }
