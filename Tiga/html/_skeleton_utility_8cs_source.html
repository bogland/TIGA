43  public static PolygonCollider2D AddBoundingBoxGameObject (Skeleton skeleton, string skinName, string slotName, string attachmentName, Transform parent, bool isTrigger = true) {
44  Skin skin = string.IsNullOrEmpty(skinName) ? skeleton.data.defaultSkin : skeleton.data.FindSkin(skinName);
45  if (skin == null) {
46  Debug.LogError("Skin " + skinName + " not found!");
47  return null;
48  }
49 
50  var attachment = skin.GetAttachment(skeleton.FindSlotIndex(slotName), attachmentName);
51  if (attachment == null) {
52  Debug.LogFormat("Attachment in slot '{0}' named '{1}' not found in skin '{2}'.", slotName, attachmentName, skin.name);
53  return null;
54  }
55 
56  var box = attachment as BoundingBoxAttachment;
57  if (box != null) {
58  var slot = skeleton.FindSlot(slotName);
59  return AddBoundingBoxGameObject(box.Name, box, slot, parent, isTrigger);
60  } else {
61  Debug.LogFormat("Attachment '{0}' was not a Bounding Box.", attachmentName);
62  return null;
63  }
64  }

76  public static PolygonCollider2D AddBoundingBoxAsComponent (BoundingBoxAttachment box, Slot slot, GameObject gameObject, bool isTrigger = true, bool isKinematic = true, float gravityScale = 0f) {
77  if (box == null) return null;
78 
79  if (slot.bone != slot.Skeleton.RootBone) {
80  var rb = gameObject.GetComponent<Rigidbody2D>();
81  if (rb == null) {
82  rb = gameObject.AddComponent<Rigidbody2D>();
83  rb.isKinematic = isKinematic;
84  rb.gravityScale = gravityScale;
85  }
86  }
87 
88  var collider = gameObject.AddComponent<PolygonCollider2D>();
89  collider.isTrigger = isTrigger;
90  SetColliderPointsLocal(collider, slot, box);
91  return collider;
92  }

94  public static void SetColliderPointsLocal (PolygonCollider2D collider, Slot slot, BoundingBoxAttachment box) {
95  if (box == null) return;
96  if (box.IsWeighted()) Debug.LogWarning("UnityEngine.PolygonCollider2D does not support weighted or animated points. Collider points will not be animated and may have incorrect orientation. If you want to use it as a collider, please remove weights and animations from the bounding box in Spine editor.");
97  var verts = box.GetLocalVertices(slot, null);
98  collider.SetPath(0, verts);

101  public static Bounds GetBoundingBoxBounds (BoundingBoxAttachment boundingBox, float depth = 0) {
102  float[] floats = boundingBox.Vertices;
103  int floatCount = floats.Length;
104 
105  Bounds bounds = new Bounds();
106  bounds.center = new Vector3(floats[0], floats[1], 0);
107  for (int i = 2; i < floatCount; i += 2)
108  bounds.Encapsulate(new Vector3(floats[i], floats[i + 1], 0));
109 
110  Vector3 size = bounds.size;
111  size.z = depth;
112  bounds.size = size;
113 
114  return bounds;
115  }

219  public void CollectBones () {
220  var skeleton = skeletonRenderer.skeleton;
221  if (skeleton == null) return;
222 
223  if (boneRoot != null) {
224  var constraintTargets = new List<System.Object>();
225  var ikConstraints = skeleton.IkConstraints;
226  for (int i = 0, n = ikConstraints.Count; i < n; i++)
227  constraintTargets.Add(ikConstraints.Items[i].target);
228 
229  var transformConstraints = skeleton.TransformConstraints;
230  for (int i = 0, n = transformConstraints.Count; i < n; i++)
231  constraintTargets.Add(transformConstraints.Items[i].target);
232 
233  var utilityBones = this.utilityBones;
234  for (int i = 0, n = utilityBones.Count; i < n; i++) {
235  var b = utilityBones[i];
236  if (b.bone == null) continue;
237  hasTransformBones |= (b.mode == SkeletonUtilityBone.Mode.Override);
238  hasUtilityConstraints |= constraintTargets.Contains(b.bone);
239  }
240 
241  hasUtilityConstraints |= utilityConstraints.Count > 0;
242 
243  if (skeletonAnimation != null) {
244  skeletonAnimation.UpdateWorld -= UpdateWorld;
245  skeletonAnimation.UpdateComplete -= UpdateComplete;
246 
247  if (hasTransformBones || hasUtilityConstraints)
248  skeletonAnimation.UpdateWorld += UpdateWorld;
249 
250  if (hasUtilityConstraints)
251  skeletonAnimation.UpdateComplete += UpdateComplete;
252  }
253 
254  needToReprocessBones = false;
255  } else {
256  utilityBones.Clear();
257  utilityConstraints.Clear();
258  }
259  }

349  b.valid = true;
350 
351  if (mode == SkeletonUtilityBone.Mode.Override) {
352  if (rot)
353  go.transform.localRotation = Quaternion.Euler(0, 0, b.bone.AppliedRotation);
354 
355  if (pos)
356  go.transform.localPosition = new Vector3(b.bone.X, b.bone.Y, 0);
357 
358  go.transform.localScale = new Vector3(b.bone.scaleX, b.bone.scaleY, 0);
359  }
