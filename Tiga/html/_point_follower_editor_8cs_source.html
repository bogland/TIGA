70  void OnEnable () {

87  public void OnSceneGUI () {
88  var tbf = target as PointFollower;
89  var skeletonRendererComponent = tbf.skeletonRenderer;
90  if (skeletonRendererComponent == null)
91  return;
92 
93  var skeleton = skeletonRendererComponent.skeleton;
94  var skeletonTransform = skeletonRendererComponent.transform;
95 
96  if (string.IsNullOrEmpty(pointAttachmentName.stringValue)) {
97  // Draw all active PointAttachments in the current skin
98  var currentSkin = skeleton.Skin;
99  if (currentSkin != skeleton.Data.DefaultSkin) DrawPointsInSkin(skeleton.Data.DefaultSkin, skeleton, skeletonTransform);
100  if (currentSkin != null) DrawPointsInSkin(currentSkin, skeleton, skeletonTransform);
101  } else {
102  int slotIndex = skeleton.FindSlotIndex(slotName.stringValue);
103  if (slotIndex >= 0) {
104  var slot = skeleton.Slots.Items[slotIndex];
105  var point = skeleton.GetAttachment(slotIndex, pointAttachmentName.stringValue) as PointAttachment;
106  if (point != null) {
107  DrawPointAttachmentWithLabel(point, slot.Bone, skeletonTransform);
108  }
109  }
110  }
111  }

113  static void DrawPointsInSkin (Skin skin, Skeleton skeleton, Transform transform) {
114  foreach (var skinEntry in skin.Attachments) {
115  var attachment = skinEntry.Value as PointAttachment;
116  if (attachment != null) {
117  var skinKey = skinEntry.Key;
118  var slot = skeleton.Slots.Items[skinKey.slotIndex];
119  DrawPointAttachmentWithLabel(attachment, slot.Bone, transform);
120  }
121  }
122  }

124  static void DrawPointAttachmentWithLabel (PointAttachment point, Bone bone, Transform transform) {
125  Vector3 labelOffset = new Vector3(0f, -0.2f, 0f);
126  SpineHandles.DrawPointAttachment(bone, point, transform);
127  Handles.Label(labelOffset + point.GetWorldPosition(bone, transform), point.Name, SpineHandles.PointNameStyle);
128  }

130  override public void OnInspectorGUI () {

132  if (needsReset) {

144  needsReset |= EditorGUI.EndChangeCheck();

147 
148  if (needsReset && Event.current.type == EventType.Layout) {
149  targetPointFollower.Initialize();
150  targetPointFollower.LateUpdate();
151  needsReset = false;
152  SceneView.RepaintAll();
153  }
154  serializedObject.Update();
155 
156  DrawDefaultInspector();
157 
158  // Find Renderer
159  if (skeletonRenderer.objectReferenceValue == null) {
160  SkeletonRenderer parentRenderer = targetPointFollower.GetComponentInParent<SkeletonRenderer>();
161  if (parentRenderer != null && parentRenderer.gameObject != targetPointFollower.gameObject) {
162  skeletonRenderer.objectReferenceValue = parentRenderer;
163  Debug.Log("Inspector automatically assigned PointFollower.SkeletonRenderer");
164  }
165  }
166 
167  var skeletonRendererReference = skeletonRenderer.objectReferenceValue as SkeletonRenderer;
168  if (skeletonRendererReference != null) {
169  if (skeletonRendererReference.gameObject == targetPointFollower.gameObject) {
170  skeletonRenderer.objectReferenceValue = null;
171  EditorUtility.DisplayDialog("Invalid assignment.", "PointFollower can only follow a skeleton on a separate GameObject.\n\nCreate a new GameObject for your PointFollower, or choose a SkeletonRenderer from a different GameObject.", "Ok");
172  }
173  }
174 
175  if (!targetPointFollower.IsValid) {
176  needsReset = true;
177  }
178 
179  var current = Event.current;
180  bool wasUndo = (current.type == EventType.ValidateCommand && current.commandName == "UndoRedoPerformed");
181  if (wasUndo)
182  targetPointFollower.Initialize();
183 
184  serializedObject.ApplyModifiedProperties();
185  }
186  }
